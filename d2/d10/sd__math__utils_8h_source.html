91  static inline bool CardanoFormula(double a, double b, double c, double d, Vector& solution)
92  {
93  solution.resize(3,3);
94  noalias(solution)= ZeroVector(3);
95 
96  if(a==0)
97  {
98  std::cout<<"This is not a cubic equation: CardanoFormula"<<std::endl;
99 
100  return false;
101  }
102 
103  double p= (3.0*a*c-b*b)/(3.0*a*a);
104 
105  double q= 2.0*b*b*b/(27.0*a*a*a)-b*c/(3.0*a*a)+d/a;
106 
107  double discriminante= p*p*p/27.0+q*q/4.0;
108 
109  if(discriminante>0)
110  {
111  return false;
112  }
113 
114  if(discriminante==0)
115  {
116 
117  solution(0)= pow(q/2.0, 1.0/3.0)-b/(3*a);
118  solution(1)= pow(q/2.0, 1.0/3.0)-b/(3*a);
119  solution(2)= pow(-4.0*q, 1.0/3.0)-b/(3*a);
120 
121  if(isnan(solution(0)) || isnan(solution(1))|| isnan(solution(2)))
122  {
123  return(false);
124  }
125 
126  return true;
127  }
128 
129  solution(0)=
130  -sqrt(-4.0/3.0*p)*cos(1.0/3.0*acos(-q/2.0*sqrt(-27.0/(p*p*p)))+PI/3.0)
131  -b/(3*a);
132  solution(1)=
133  sqrt(-4.0/3.0*p)*cos(1.0/3.0*acos(-q/2.0*sqrt(-27.0/(p*p*p))))-b/(3*a)
134  ;
135  solution(2)=
136  -sqrt(-4.0/3.0*p)*cos(1.0/3.0*acos(-q/2.0*sqrt(-27.0/(p*p*p)))-PI/3.0)
137  -b/(3*a);
138 
139  if(isnan(solution(0)) || isnan(solution(1))|| isnan(solution(2)))
140  {
141  return false;
142  }
143 
144  return true;
145  }

253  static inline void QRFactorization(const MatrixType& A, MatrixType& Q, MatrixType& R)
254  {
255 
256  //QR Factorization with Householder-Algo
257  int dim= A.size1();
258 
259  Vector y(dim);
260 
261  Vector w(dim);
262 
263  R.resize(dim,dim);
264 
265  R=ZeroMatrix(dim,dim);
266 
267  Q.resize(dim,dim);
268 
269  Q=ZeroMatrix(dim,dim);
270 
271  Matrix Help= A;
272 
273  Matrix unity= ZeroMatrix(dim,dim);
274 
275  for(int j=0; j<dim; j++)
276  unity(j,j)=1.0;
277 
278  std::vector<Matrix> HelpQ(dim-1);
279 
280  std::vector<Matrix> HelpR(dim-1);
281 
282  for(int i=0; i< dim-1; i++)
283  {
284  HelpQ[i].resize(dim,dim);
285  HelpR[i].resize(dim,dim);
286  noalias(HelpQ[i])= unity;
287  noalias(HelpR[i])= ZeroMatrix(dim,dim);
288  }
289 
290  for(int iteration=0; iteration< dim-1; iteration++)
291  {
292  //Vector y
293  for(int i=iteration; i<dim; i++)
294  y(i)= Help(i,iteration);
295 
296 
297  //Helpvalue l
298  double normy=0.0;
299 
300  for(int i=iteration; i<dim; i++)
301  normy += y(i)*y(i);
302 
303  normy= sqrt(normy);
304 
305  double l= sqrt((normy*(normy+fabs(y(iteration))))/2);
306 
307  double k=0.0;
308 
309  if(y[iteration] !=0)
310  k= - y(iteration)/fabs(y(iteration))*normy;
311  else
312  k= -normy;
313 
314  for(int i=iteration; i<dim; i++)
315  {
316  double e=0;
317 
318  if(i==iteration)
319  e=1;
320 
321  w(i)= 1/(2*l)*(y(i)-k*e);
322  }
323 
324  for(int i=iteration; i<dim; i++)
325  for(int j=iteration; j<dim; j++)
326  HelpQ[iteration](i,j)= unity(i,j)- 2*w(i)*w(j);
327 
328 
329  for(int i=iteration; i<dim; i++)
330  for(int j=iteration; j<dim; j++)
331  for(int k=iteration; k<dim; k++)
332  HelpR[iteration](i,j)+= HelpQ[iteration](i,k)*Help(k,j);
333 
334  Help= HelpR[iteration];
335 
336  }
337 
338  //Assembling R
339  for(int k=0; k<dim-1; k++)
340  {
341  for(int i=k; i<dim; i++)
342  for(int j=k; j<dim; j++)
343  R(i,j) =HelpR[k](i,j);
344 
345  }
346 
347 
348  for(int k=1; k<dim-1; k++)
349  {
350  for(int i=0; i<dim; i++)
351  for(int j=0; j<dim; j++)
352  for(int l=0; l<dim; l++)
353  Q(i,j)+= HelpQ[(k-1)](i,l)*HelpQ[k](l,j);
354  noalias(HelpQ[k])=Q;
355  }
356  if(dim-1==1)
357  noalias(Q)=HelpQ[0];
358 
359  }

564  static inline void EigenVectors( MatrixType& A,
565  MatrixType& V,
566  TDataType& error_tolerance,
567  TDataType zero_tolerance)
568  {
569  //initial error
570  TDataType error = 1.0;
571  int n = A.size2();
572  //setting V to identity matrix
573  V = IdentityMatrix( V.size1() );
574  //calculation loop (as long as there is no convergence)
575  //WARNING: iteration never exceeds
576  while( error > error_tolerance )
577  {
578  for( int i=0; i<n; i++ )
579  {
580  for( int j=i+1; j<n; j++ )
581  {
582  double theta = 0.0;
583  if( MathUtilsType::Abs( A(i,j) ) >= zero_tolerance )
584  {
585  if( MathUtilsType::Abs( A(i,i)-A(j,j) ) > 0.0 )
586  {
587  theta = 0.5*atan(2*A(i,j)/(A(i,i)-A(j,j)));
588  }
589  else theta = 0.25*PI;
590  }
591  MatrixType T = IdentityMatrix( n );
592 
593  T(i,i) = cos(theta);
594  T(i,j) = -sin(theta);
595  T(j,i) = -T(i,j);
596  T(j,j) = T(i,i);
597 
598  A = Mult( A, T );
599  MatrixType TT = Transpose(T);
600  A = Mult( TT, A );
601  V = Mult( V, T );
602  }
603  }
604  double sTot = 0.0;
605  double sDiag = 0.0;
606  for( unsigned int i=0; i<A.size1(); i++ )
607  {
608  for( unsigned int j=0; j<A.size2(); j++ )
609  {
610  sTot += MathUtilsType::Abs(A(i,j));
611  }
612  sDiag+= MathUtilsType::Abs(A(i,i));
613  }
614  error=(sTot-sDiag)/sDiag;
615  }
616  //sorting eigenvalues
617  int maxIndex = 0;
618  TDataType maxEv = A(0,0);
619  for( unsigned int i=0; i<A.size1(); i++ )
620  {
621  for( unsigned int j=i; j<A.size1(); j++ )
622  {
623  //searching current maximum
624  if( A(j,j) > maxEv )
625  {
626  maxIndex = j;
627  maxEv = A(j,j);
628  }
629  //swapping eigenvalue matrix
630  TDataType dummy = A(i,i);
631  A(i,i) = A(maxIndex,maxIndex);
632  A(maxIndex,maxIndex) = dummy;
633  //swapping eigenvector matrix
634  for( unsigned int k=0; k<A.size2(); k++ )
635  {
636  dummy = V(k,i);
637  V(k,i) = V(k,maxIndex);
638  V(k,maxIndex) = dummy;
639  }
640  }
641 
642  }
643  }//EigenVectors
