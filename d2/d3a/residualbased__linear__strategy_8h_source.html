345  double Solve()
346  {
347  KRATOS_TRY
348 
349  //pointers needed in the solution
350  typename TSchemeType::Pointer pScheme = GetScheme();
351  typename TBuilderAndSolverType::Pointer pBuilderAndSolver = GetBuilderAndSolver();
352 
353 
354  ProcessInfo& pCurrentProcessInfo = BaseType::GetModelPart().GetProcessInfo();
355 
356  //OPERATIONS THAT SHOULD BE DONE ONCE - internal check to avoid repetitions
357  //if the operations needed were already performed this does nothing
358  if(mInitializeWasPerformed == false)
359  {
360  Initialize();
361  mInitializeWasPerformed = true;
362  }
363 
364  //prints informations about the current time
365  if (BaseType::GetEchoLevel()!=0)
366  {
367  std::cout << " " << std::endl;
368  std::cout << "CurrentTime = " << pCurrentProcessInfo[TIME] << std::endl;
369  }
370 
371  //initialize solution step
372  if (mSolutionStepIsInitialized == false)
373  {
374  InitializeSolutionStep();
375  mSolutionStepIsInitialized = true;
376  }
377 
378  //updates the database with a prediction of the solution
379  Predict();
380 
381  TSystemMatrixType& mA = *mpA;
382  TSystemVectorType& mDx = *mpDx;
383  TSystemVectorType& mb = *mpb;
384 
385 
386  if(BaseType::mRebuildLevel >0 || BaseType::mStiffnessMatrixIsBuilt == false)
387  {
388  TSparseSpace::SetToZero(mA);
389  TSparseSpace::SetToZero(mDx);
390  TSparseSpace::SetToZero(mb);
391  pBuilderAndSolver->BuildAndSolve(pScheme,BaseType::GetModelPart(),mA,mDx,mb);
392  BaseType::mStiffnessMatrixIsBuilt = true;
393  }
394  else
395  {
396  TSparseSpace::SetToZero(mDx);
397  TSparseSpace::SetToZero(mb);
398  pBuilderAndSolver->BuildRHSAndSolve(pScheme,BaseType::GetModelPart(),mA,mDx,mb);
399  }
400 
401  if (BaseType::GetEchoLevel()==3) //if it is needed to print the debug info
402  {
403  std::cout << "SystemMatrix = " << mA << std::endl;
404  std::cout << "solution obtained = " << mDx << std::endl;
405  std::cout << "RHS = " << mb << std::endl;
406  }
407 
408  //update results
409  DofsArrayType& rDofSet = pBuilderAndSolver->GetDofSet();
410  pScheme->Update(BaseType::GetModelPart(),rDofSet,mA,mDx,mb);
411 
412  //move the mesh if needed
413  if(BaseType::MoveMeshFlag() == true) BaseType::MoveMesh();
414 
415  //calculate if needed the norm of Dx
416  double normDx = 0.00;
417  if(mCalculateNormDxFlag == true)
418  {
419  normDx = TSparseSpace::TwoNorm(mDx);
420  }
421 
422  //calculate reactions if required
423  if (mCalculateReactionsFlag ==true)
424  {
425  pBuilderAndSolver->CalculateReactions(pScheme,BaseType::GetModelPart(),mA,mDx,mb);
426  }
427 
428  //Finalisation of the solution step,
429  //operations to be done after achieving convergence, for example the
430  //Final Residual Vector (mb) has to be saved in there
431  //to avoid error accumulation
432  pScheme->FinalizeSolutionStep(BaseType::GetModelPart(),mA,mDx,mb);
433  pBuilderAndSolver->FinalizeSolutionStep(BaseType::GetModelPart(),mA,mDx,mb);
434 
435  //deallocate the systemvectors if needed
436  if (mReformDofSetAtEachStep == true)
437  {
438  std::cout << "Clearing System" << std::endl;
439  this->Clear();
440  //std::cout << "Clearing System" << std::endl;
441  //TSparseSpace::ClearData(mA);
442  //TSparseSpace::ClearData(mDx);
443  //TSparseSpace::ClearData(mb);
444  }
445 
446  //Cleaning memory after the solution
447  pScheme->Clean();
448 
449  //reset flags for next step
450  mSolutionStepIsInitialized = false;
451 
452  return normDx;
453 
454  KRATOS_CATCH("")
455 
456  }

634  {
635  KRATOS_TRY
636 
637  typename TBuilderAndSolverType::Pointer pBuilderAndSolver = GetBuilderAndSolver();
638  typename TSchemeType::Pointer pScheme = GetScheme();

643 
644  //loop to reform the dofset

646  if (pBuilderAndSolver->GetDofSetIsInitializedFlag() == false ||

654

657  pBuilderAndSolver->SetUpSystem(BaseType::GetModelPart());

662  boost::timer system_matrix_resize_time;
663  pBuilderAndSolver->ResizeAndInitializeVectors(mpA,mpDx,mpb,BaseType::GetModelPart().Elements(),BaseType::GetModelPart().Conditions(),BaseType::GetModelPart().GetProcessInfo());

669 
670

672  TSystemVectorType& mDx = *mpDx;
673  TSystemVectorType& mb = *mpb;

677 
678  //initial operations ... things that are constant over the Solution Step
679  pScheme->InitializeSolutionStep(BaseType::GetModelPart(),mA,mDx,mb);
