51 /*
52  * *************************************************
53  * Global data structures used in LU factorization
54  * *************************************************
55  *
56  * nsuper: number of supernodes = nsuper+1, numbered between 0 and nsuper.
57  *
58  * (supno, xsup, xsup_end):
59  * supno[i] is the supernode number to which column i belongs;
60  * xsup[s] points to the first column of supernode s;
61  * xsup_end[s] points to one past the last column of supernode s.
62  * Example: supno 0 1 2 2 3 3 3 4 4 4 4 4 (n=12)
63  * xsup 0 1 2 4 7
64  * xsup_end 1 2 4 7 12
65  * Note: dfs will be performed on supernode rep. relative to the new
66  * row pivoting ordering
67  *
68  * (lsub, xlsub, xlsub_end):
69  * lsub[*] contains the compressed subscripts of the supernodes;
70  * xlsub[j] points to the starting location of the j-th column in
71  * lsub[*];
72  * xlsub_end[j] points to one past the ending location of the j-th
73  * column in lsub[*].
74  * Storage: original row subscripts in A.
75  *
76  * During the course of sparse LU factorization, we also use
77  * (lsub, xlsub, xlsub_end, xprune) to represent symmetrically
78  * pruned graph. Contention will occur when one processor is
79  * performing DFS on supernode S, while another processor is pruning
80  * supernode S. We use the following data structure to deal with
81  * this problem. Suppose each supernode contains columns {s,s+1,...,t},
82  * with first column s and last column t.
83  *
84  * (1) if t > s, only the subscript sets for column s and column t
85  * are stored. Column t represents pruned adjacency structure.
86  *
87  * --------------------------------------------
88  * lsub[*] ... | col s | col t | ...
89  * --------------------------------------------
90  * ^ ^ ^
91  * xlsub[s] xlsub_end[s] xlsub_end[s+1]
92  * xlsub[s+1] :
93  * : :
94  * : xlsub_end[t]
95  * xlsub[t] xprune[t]
96  * xprune[s]
97  *
98  * (2) if t == s, i.e., a singleton supernode, the subscript set
99  * is stored twice:
100  *
101  * --------------------------------------
102  * lsub[*] ... | s | s | ...
103  * --------------------------------------
104  * ^ ^ ^
105  * xlsub[s] xlsub_end[s] xprune[s]
106  *
107  * There are two subscript sets for each supernode, the last column
108  * structures (for pruning) will be removed after the numerical LU
109  * factorization phase:
110  * o lsub[j], j = xlsub[s], ..., xlsub_end[s]-1
111  * is the structure of column s (i.e. structure of this supernode).
112  * It is used for the storage of numerical values.
113  * o lsub[j], j = xlsub[t], ..., xlsub_end[t]-1
114  * is the structure of the last column t of this supernode.
115  * It is for the purpose of symmetric pruning. Therefore, the
116  * structural subscripts can be rearranged without making physical
117  * interchanges among the numerical values.
118  *
119  * DFS will traverse the first subscript set if the supernode
120  * has not been pruned; otherwise it will traverse the second
121  * subscript list, i.e., the part of the pruned graph.
122  *
123  * (lusup, xlusup, xlusup_end):
124  * lusup[*] contains the numerical values of the supernodes;
125  * xlusup[j] points to the starting location of the j-th column in
126  * storage vector lusup[*];
127  * xlusup_end[j] points to one past the ending location of the j-th
128  * column in lusup[*].
129  * Each supernode is stored in column-major, consistent with Fortran
130  * two-dimensional array storage.
131  *
132  * (ucol, usub, xusub, xusub_end):
133  * ucol[*] stores the numerical values of the U-columns above the
134  * supernodes.
135  * usub[k] stores the row subscripts of nonzeros ucol[k];
136  * xusub[j] points to the starting location of column j in ucol/usub[];
137  * xusub_end[j] points to one past the ending location column j in
138  * ucol/usub[].
139  * Storage: new row subscripts; that is indexed intp PA.
140  *
141  */
142 typedef struct {
143  int *xsup; /* supernode and column mapping */
144  int *xsup_end;
145  int *supno;
146  int *lsub; /* compressed L subscripts */
147  int *xlsub;
148  int *xlsub_end;
149  complex *lusup; /* L supernodes */
150  int *xlusup;
151  int *xlusup_end;
152  complex *ucol; /* U columns */
153  int *usub;
154  int *xusub;
155  int *xusub_end;
156  int nsuper; /* current supernode number */
157  int nextl; /* next position in lsub[] */
158  int nextu; /* next position in usub[]/ucol[] */
159  int nextlu; /* next position in lusup[] */
160  int nzlmax; /* current max size of lsub[] */
161  int nzumax; /* " " " ucol[] */
162  int nzlumax; /* " " " lusup[] */
163  /* ---------------------------------------------------------------
164  * Memory managemant for L supernodes
165  */
166  int *map_in_sup; /* size n+1 - the address offset of each column
167  * in lusup[*], which is divided into regions
168  * by the supernodes of Householder matrix H.
169  * If column k starts a supernode in H,
170  * map_in_sup[k] is the next open position in
171  * lusup[*]; otherwise map_in_sup[k] gives the
172  * offset (negative) to the leading column
173  * of the supernode in H.
174  */
175  int dynamic_snode_bound;
176  /* --------------------------------------------------------------- */
177 } GlobalLU_t;
