145  void CalculateBoundingBox() {
146  for(SizeType i = 0 ; i < TDimension ; i++){
147  mMinPoint[i] = (**mPointBegin)[i];
148  mMaxPoint[i] = (**mPointBegin)[i];
149  }
150  for(IteratorType Point = mPointBegin ; Point != mPointEnd ; Point++)
151  for(SizeType i = 0 ; i < TDimension ; i++){
152  if( (**Point)[i] < mMinPoint[i] ) mMinPoint[i] = (**Point)[i];
153  if( (**Point)[i] > mMaxPoint[i] ) mMaxPoint[i] = (**Point)[i];
154  }

159  void CalculateCellSize() {
160 
161  CoordinateType delta[TDimension];
162  CoordinateType alpha[TDimension];
163  CoordinateType mult_delta = 1.00;
164  for(SizeType i = 0 ; i < TDimension ; i++) {
165  delta[i] = mMaxPoint[i] - mMinPoint[i];
166  delta[i] = (delta[i] == 0.00) ? 1.00 : delta[i];
167  }
168 
169  alpha[0] = 1.00;
170  for(SizeType i = 1 ; i < TDimension ; i++){
171  alpha[i] = delta[i] / delta[0];
172  mult_delta *= alpha[i];
173  }
174 
175  mN[0] = static_cast<SizeType>( pow(static_cast<CoordinateType>(std::distance(mPointBegin,mPointEnd)/mult_delta), 1.00/3.00)+1 );
176 
177  for(SizeType i = 1 ; i < TDimension ; i++)
178  mN[i] = static_cast<SizeType>(alpha[i] * mN[0] + 1);
179 
180  for(SizeType i = 0 ; i < TDimension ; i++){
181  mCellSize[i] = delta[i] / mN[i];
182  mInvCellSize[i] = 1.00 / mCellSize[i];
183  }
184  }
