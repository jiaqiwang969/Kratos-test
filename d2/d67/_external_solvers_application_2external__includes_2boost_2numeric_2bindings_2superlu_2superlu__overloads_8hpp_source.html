379  template<typename T>
380  void Print_CompRow_Matrix(T, char *what, SuperMatrix *A)
381  {
382  NRformat *Astore;
383  register int i,n;
384  T *dp;
385 
386  printf("\nCompRow matrix %s:\n", what);
387  printf("Stype %d, Dtype %d, Mtype %d\n", A->Stype,A->Dtype,A->Mtype);
388  n = A->nrow;
389  Astore = (NRformat *) A->Store;
390  dp = (T *) Astore->nzval;
391  printf("nrow %d, ncol %d, nnz %d\n", A->nrow,A->ncol,Astore->nnz);
392  printf("nzval: ");
393  for (i = 0; i < Astore->rowptr[n]; ++i) printf("%f ", dp[i]);
394  printf("\ncolind: ");
395  for (i = 0; i < Astore->rowptr[n]; ++i) printf("%d ", Astore->colind[i]);
396  printf("\nrowptr: ");
397  for (i = 0; i <= n; ++i) printf("%d ", Astore->rowptr[i]);
398  printf("\n");
399  fflush(stdout);
400  }
