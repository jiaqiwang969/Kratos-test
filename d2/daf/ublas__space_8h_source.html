362  template<class TOtherMatrixType, class TEquationIdVectorType>
363  inline static void AssembleLHS(
364  MatrixType& A,
365  TOtherMatrixType& LHS_Contribution,
366  TEquationIdVectorType& EquationId
367  )
368  {
369  unsigned int system_size = A.size1();
370  unsigned int local_size = LHS_Contribution.size1();
371 
372  for (unsigned int i_local=0; i_local<local_size; i_local++)
373  {
374  unsigned int i_global=EquationId[i_local];
375  if ( i_global < system_size )
376  {
377  for (unsigned int j_local=0; j_local<local_size; j_local++)
378  {
379  unsigned int j_global=EquationId[j_local];
380  if ( j_global < system_size )
381  A(i_global,j_global) += LHS_Contribution(i_local,j_local);
382  }
383  }
384  }
385  }
