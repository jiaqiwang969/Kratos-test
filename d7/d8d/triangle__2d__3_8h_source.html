730  virtual Matrix& InverseOfJacobian( Matrix& rResult,
731  IndexType IntegrationPointIndex,
732  IntegrationMethod ThisMethod ) const
733  {
734  //current jacobian
735  Matrix tempMatrix = ZeroMatrix(2,2);
736  tempMatrix = Jacobian( tempMatrix, IntegrationPointIndex, ThisMethod );
737  //determinant of jacobian
738  double det_j = DeterminantOfJacobian( IntegrationPointIndex, ThisMethod );
739  //checking for singularity
740  if(det_j == 0.00)
741  KRATOS_ERROR(std::runtime_error,
742  "Zero determinant of jacobian during inversion of matrix!" ,
743  *this);
744  //setting up result matrix
745  rResult.resize(2,2);
746  //filling matrix
747  rResult(0,0) = (tempMatrix(1,1))/(det_j);
748  rResult(1,0) = -(tempMatrix(1,0))/(det_j);
749  rResult(0,1) = -(tempMatrix(0,1))/(det_j);
750  rResult(1,1) = (tempMatrix(0,0))/(det_j);
751  return rResult;
752  }
