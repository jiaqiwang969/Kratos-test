299  bool IterativeSolve(SparseMatrixType& rA, SparseVectorType& rX, SparseVectorType& rB)
300  {
301  const int size1 = TSparseSpaceType::Size(rX);
302  int size2 = size1 / mSeedPointNo;
303  if (size1 % mSeedPointNo != 0) size2++;
304 
305  std::cout << "********** mSeedPointNo = " << mSeedPointNo << std::endl;
306  std::cout << "********** size1 = " << size1 << std::endl;
307  std::cout << "********** size2 = " << size2 << std::endl;
308 
309  SparseMatrixType W(size1, size2, size1);
310 
311  // Building W
312  TSparseSpaceType::SetToZero(W);
313 
314  std::cout << "********** W set to zero!" << std::endl;
315 
316  for (int i = 0; i < size2; i++)
317  for (int j = 0; j < mSeedPointNo; j++)
318  if (i * mSeedPointNo + j < size1)
319  W(i * mSeedPointNo + j, i) = 1;
320 
321  std::cout << "********** W built!" << std::endl;
322 
323  // Use TSparseSpaceType::size_type?
324  std::vector<std::size_t> w(size1);
325 
326  // Building w
327  for (int i = 0; i < size2; i++)
328  for (int j = 0; j < mSeedPointNo; j++)
329  if (i * mSeedPointNo + j < size1)
330  w[i * mSeedPointNo + j] = i;
331 
332  std::cout << "********** w built!" << std::endl;
333 
334  // Non-zero structure of Ah
335  std::vector<std::set<std::size_t> > AhNZ(size2);
336 
337  // Loop over non-zero structure of A and build non-zero structure of Ah
338  typename SparseMatrixType::iterator1 a_iterator = rA.begin1();
339 
340  for (int i = 0; i < size1; i++)
341  {
342  #ifndef BOOST_UBLAS_NO_NESTED_CLASS_RELATION
343  for (typename SparseMatrixType::iterator2 row_iterator = a_iterator.begin() ;
344  row_iterator != a_iterator.end() ; ++row_iterator)
345  {
346  #else
347  for (typename SparseMatrixType::iterator2 row_iterator = begin(a_iterator,
348  boost::numeric::ublas::iterator1_tag());
349  row_iterator != end(a_iterator,
350  boost::numeric::ublas::iterator1_tag()); ++row_iterator )
351  {
352  #endif
353  AhNZ[w[a_iterator.index1()]].insert(w[row_iterator.index2()]);
354  }
355 
356  a_iterator++;
357  }
358 
359  std::cout << "********** NZS built!" << std::endl;
360 
361  // Count the number of non-zeros in Ah
362  int NZ = 0;
363  for (int i = 0; i < size2; i++)
364  NZ += AhNZ[i].size();
365 
366  std::cout << "********** NZ = " << NZ << std::endl;
367 
368  SparseMatrixType Ah(size2, size2, NZ);
369 
370  // Insert the non-zero structure into Ah
371  for(int i = 0 ; i < size2 ; i++)
372  {
373  for(std::set<std::size_t>::iterator j = AhNZ[i].begin() ; j != AhNZ[i].end() ; j++)
374  {
375  Ah.push_back(i,*j, 0.00);
376  }
377  }
378 
379  // Now building Ah
380  a_iterator = rA.begin1();
381 
382  for (int i = 0; i < size1; i++)
383  {
384  #ifndef BOOST_UBLAS_NO_NESTED_CLASS_RELATION
385  for (typename SparseMatrixType::iterator2 row_iterator = a_iterator.begin() ;
386  row_iterator != a_iterator.end() ; ++row_iterator)
387  {
388  #else
389  for (typename SparseMatrixType::iterator2 row_iterator = begin(a_iterator,
390  boost::numeric::ublas::iterator1_tag());
391  row_iterator != end(a_iterator,
392  boost::numeric::ublas::iterator1_tag()); ++row_iterator )
393  {
394  #endif
395  Ah(w[a_iterator.index1()], w[row_iterator.index2()]) += *row_iterator;
396  }
397 
398  a_iterator++;
399  }
400 
401  std::cout << "********** W^T * A * W built!" << std::endl;
402 
403  // To save some time, we do the factorization once, and do the solve several times.
404  // When this is available through the LinearSolver interface, replace this.
405 
406  LUSkylineFactorization<TSparseSpaceType, TDenseSpaceType> Factorization;
407 
408  //mpLinearSolver = LinearSolverType::Pointer(new SkylineLUFactorizationSolver<TSparseSpaceType, TDenseSpaceType>);
409 
410  Factorization.copyFromCSRMatrix(Ah);
411  Factorization.factorize();
412 
413  std::cout << "********** Factorization done!" << std::endl;
414 
415  SparseVectorType r(size1), t(size1), d(size1), p(size1), q(size1);
416  SparseVectorType th(size2), dh(size2);
417 
418  // r = rA * rX
419  PreconditionedMult(rA, rX, r);
420 
421  // r = rB - r
422  TSparseSpaceType::ScaleAndAdd(1.00, rB, -1.00, r);
423 
424  std::cout << "********** ||r|| = " << TSparseSpaceType::TwoNorm(r) << std::endl;
425 
426  // th = W^T * r
427  TSparseSpaceType::TransposeMult(W, r, th);
428 
429  // Solve Ah * th = dh
430  Factorization.backForwardSolve(size2, th, dh);
431 
432  // t = W * dh
433  TSparseSpaceType::Mult(W, dh, t);
434 
435  // rX = rX + t
436  TSparseSpaceType::ScaleAndAdd(1.00, t, 1.00, rX);
437 
438  //r = rA * rX
439  PreconditionedMult(rA, rX, r);
440 
441  // r = B - r
442  TSparseSpaceType::ScaleAndAdd(1.00, rB, -1.00, r);
443 
444  // t = A * r
445  //TSparseSpaceType::Mult(rA, r, t);
446  PreconditionedMult(rA, r, t);
447 
448  // th = W^T * t
449  TSparseSpaceType::TransposeMult(W, t, th);
450 
451  // Solve Ah * th = dh
452  Factorization.backForwardSolve(size2, th, dh);
453 
454  // p = W * dh
455  TSparseSpaceType::Mult(W, dh, p);
456 
457  // p = r - p
458  TSparseSpaceType::ScaleAndAdd(1.00, r, -1.00, p);
459 
460  // Iteration counter
461  BaseType::mIterationsNumber = 0;
462 
463  BaseType::mBNorm = TSparseSpaceType::TwoNorm(rB);
464 
465  double roh0 = TSparseSpaceType::Dot(r, r);
466  double roh1 = roh0;
467  double beta = 0;
468 
469  std::cout << "********** Just before the loop! " << (fabs(roh0) < 1.0e-30) << std::endl;
470 
471  if(fabs(roh0) < 1.0e-30) //modification by Riccardo
472 // if(roh0 == 0.00)
473 
474  return false;
475 
476  do
477  {
478  PreconditionedMult(rA, p, q);
479 
480  double pq = TSparseSpaceType::Dot(p, q);
481 
482  //std::cout << "********** pq = " << pq << std::endl;
483 
484  //if(pq == 0.00)
485  if(fabs(pq) <= 1.0e-30)
486  break;
487 
488  double alpha = roh0 / pq;
489 
490  TSparseSpaceType::ScaleAndAdd(alpha, p, 1.00, rX);
491  TSparseSpaceType::ScaleAndAdd(-alpha, q, 1.00, r);
492 
493  roh1 = TSparseSpaceType::Dot(r, r);
494 
495  beta = (roh1 / roh0);
496 
497  // t = A * r
498  //TSparseSpaceType::Mult(rA, r, t);
499  PreconditionedMult(rA, r, t);
500 
501  // th = W^T * t
502  TSparseSpaceType::TransposeMult(W, t, th);
503 
504  // Solve Ah * th = dh
505  Factorization.backForwardSolve(size2, th, dh);
506 
507  // t = W * dh
508  TSparseSpaceType::Mult(W, dh, t);
509 
510  // t = r - t
511  TSparseSpaceType::ScaleAndAdd(1.00, r, -1.00, t);
512 
513  // p = beta * p + t
514  TSparseSpaceType::ScaleAndAdd(1.00, t, beta, p);
515 
516  roh0 = roh1;
517 
518  BaseType::mResidualNorm = sqrt(roh1);
519 
520  BaseType::mIterationsNumber++;
521 
522  if (BaseType::mIterationsNumber % 100 == 0)
523  std::cout << "********** iteration = " << BaseType::mIterationsNumber << ", resnorm = " << BaseType::mResidualNorm << std::endl;
524 
525  } while(BaseType::IterationNeeded() && (fabs(roh0) > 1.0e-30)/*(roh0 != 0.00)*/);
526 
527  return BaseType::IsConverged();
528  }
