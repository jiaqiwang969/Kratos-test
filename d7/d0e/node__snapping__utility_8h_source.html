216  Point<3> Intersection(Point<3> k0, Point<3> k1, ClosedCylinder3D cylinder){
217  Point<3> b(0,0,0);
218  Point<3> m = cylinder.GetCenter();
219  Point<3> e1 = cylinder.GetR1();
220  Point<3> e2 = cylinder.GetR2();
221  Point<3> e3 = cylinder.GetR3();
222  Matrix A = ZeroMatrix(3,3);
223  Matrix invA = ZeroMatrix(3,3);
224  double detA = 0;
225  Point<3> solution(0,0,0);
226  b = k0 - m - e3;
227  for (int i=0; i<3; ++i){
228  A(0,i) = e1(i);
229  A(1,i) = e2(i);
230  A(2,i) = -(k1(i)-k0(i));
231  }
232  MathUtils<double>::InvertMatrix3(A, invA, detA);
233  if( fabs(detA) > 1.0e-6 ) solution = InvAb(invA, b);
234 
235  return k0 + solution(2) * (k1 - k0);
236  }
