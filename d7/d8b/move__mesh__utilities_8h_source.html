82  void BDF_MoveMesh(unsigned int time_order, ModelPart& rModelPart)
83  {
84  KRATOS_TRY
85  //calculating time integration coefficients
86  ProcessInfo& CurrentProcessInfo = rModelPart.GetProcessInfo();
87  double dt = CurrentProcessInfo[DELTA_TIME];
88 
89  Vector BDFcoeffs(time_order+1);
90 
91  if(time_order == 2)
92  {
93  if(rModelPart.GetBufferSize() < 3)
94  KRATOS_ERROR(std::logic_error,"insufficient buffer size for BDF2","")
95 
96  BDFcoeffs[0] = 1.5 / dt; //coefficient for step n+1
97  BDFcoeffs[1] = -2.0 / dt;//coefficient for step n
98  BDFcoeffs[2] = 0.5 / dt;//coefficient for step n-1
99  }
100  else
101  {
102  BDFcoeffs[0] = 1.0 / dt; //coefficient for step n+1
103  BDFcoeffs[1] = -1.0 / dt;//coefficient for step n
104  }
105 
106  //update nodal coordinates
107  array_1d<double,3> mesh_vel;
108  for(ModelPart::NodesContainerType::iterator i = rModelPart.NodesBegin();
109  i!=rModelPart.NodesEnd(); i++)
110  {
111  const array_1d<double,3>& disp = i->FastGetSolutionStepValue(DISPLACEMENT);
112  i->X() = i->X0() + disp[0];
113  i->Y() = i->Y0() + disp[1];
114  i->Z() = i->Z0() + disp[2];
115 
116  //calculating the mesh velocity
117  noalias(mesh_vel) = BDFcoeffs[0] * disp;
118  for(unsigned int step=1; step<time_order+1; step++)
119  noalias(mesh_vel) += BDFcoeffs[step]*i->FastGetSolutionStepValue(DISPLACEMENT,step);
120 
121  //saving the mesh velocity
122  noalias(i->FastGetSolutionStepValue(MESH_VELOCITY)) = mesh_vel;
123  }
124 
125 
126  KRATOS_CATCH("");
127  }
