278  void CalculateMeshVelocities()
279  {
280  KRATOS_TRY;
281 
282  double DeltaTime = (*mpMeshModelPart).GetProcessInfo()[DELTA_TIME];
283  double coeff = 1/DeltaTime;
284  if( mvel_order == 1) //mesh velocity calculated as (x(n+1)-x(n))/Dt
285  {
286  for(ModelPart::NodeIterator i = (*mpMeshModelPart).NodesBegin() ;
287  i != (*mpMeshModelPart).NodesEnd() ; ++i)
288  {
289  array_1d<double,3>& mesh_v = (i)->FastGetSolutionStepValue(MESH_VELOCITY);
290  array_1d<double,3>& disp = (i)->FastGetSolutionStepValue(DISPLACEMENT);
291  array_1d<double,3>& dispold = (i)->FastGetSolutionStepValue(DISPLACEMENT,1);
292  noalias(mesh_v) = disp - dispold;
293  mesh_v *= coeff;
294  }
295  }
296  else //mesh velocity calculated as (3*x(n+1)-4*x(n)+x(n-1))/(2*Dt)
297  {
298  double c1 = 1.50*coeff;
299  double c2 = -2.0*coeff;
300  double c3 = 0.50*coeff;
301 
302  for(ModelPart::NodeIterator i = (*mpMeshModelPart).NodesBegin() ;
303  i != (*mpMeshModelPart).NodesEnd() ; ++i)
304  {
305  array_1d<double,3>& mesh_v = (i)->FastGetSolutionStepValue(MESH_VELOCITY);
306  noalias(mesh_v) = c1 * (i)->FastGetSolutionStepValue(DISPLACEMENT);
307  noalias(mesh_v) += c2 * (i)->FastGetSolutionStepValue(DISPLACEMENT,1);
308  noalias(mesh_v) += c3 * (i)->FastGetSolutionStepValue(DISPLACEMENT,2);
309  }
310  }
