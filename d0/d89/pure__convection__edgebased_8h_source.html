156  void ComputeTimeStep(double CFLNumber)
157  {
158  KRATOS_TRY
159 
160  //local variable for time step size
161  double delta_t = 1e10;
162 
163  //getting value of current velocity
164  mr_matrix_container.FillVectorFromDatabase(VELOCITY, mUn1, mr_model_part.Nodes());
165 
166  //loop over all nodes
167  double n_nodes = mUn1.size();
168  for (unsigned int i_node = 0; i_node < n_nodes; i_node++)
169  {
170  //use CFL condition to compute time step size
171  double delta_t_i = CFLNumber * 1.0 / (norm_2(mUn1[i_node])/mHmin[i_node] ) ;
172  //choose the overall minimum of delta_t_i
173  if (delta_t_i < delta_t)
174  delta_t = delta_t_i;
175  }
176 
177  //perform MPI syncronization of the dt (minimum should be kept)
178 
179  //write time step size to Kratos
180  ProcessInfo& CurrentProcessInfo = mr_model_part.GetProcessInfo();
181  CurrentProcessInfo[DELTA_TIME] = delta_t;

183  KRATOS_CATCH("")

187  //function to solve fluid equations - fractional step 1: compute fractional momentum
188  Vector Solve()
189  {
190  KRATOS_TRY
191 
192  //PREREQUISITES
193 
194  //variables for node based data handling
195  ModelPart::NodesContainerType& rNodes = mr_model_part.Nodes();
196  int n_nodes = rNodes.size();
197 
198  //storage of nodal values in local variables
199  ValuesVectorType rhs;
200  rhs.resize(n_nodes);
201 
202  //read variables from Kratos
203  mr_matrix_container.FillVectorFromDatabase(VELOCITY, mUn1, mr_model_part.Nodes());
204  mr_matrix_container.FillOldVectorFromDatabase(VELOCITY, mUn, mr_model_part.Nodes());
205 
206  mr_matrix_container.FillScalarFromDatabase(DISTANCE, mphi_n1, mr_model_part.Nodes());
207  mr_matrix_container.FillOldScalarFromDatabase(DISTANCE, mphi_n, mr_model_part.Nodes());
208 
209 
210  //read time step size from Kratos
211  ProcessInfo& CurrentProcessInfo = mr_model_part.GetProcessInfo();
212  double delta_t = CurrentProcessInfo[DELTA_TIME];
213 
214  //compute advective velocity - area average of the current velocity
215  double coefficient = 1;
216  CalculateAdvectiveVelocity(mUn,mUn1,mA, coefficient);
217 
218  //compute intrinsic time
219  double time_inv = 1.0/delta_t;
220 
221  #pragma omp parallel for firstprivate(time_inv)
222  for (int i_node = 0; i_node < n_nodes; i_node++)
223  {
224  double& h_i = mHmin[i_node];
225  array_1d<double, TDim>& a_i = mA[i_node];
226 
227  double vel_norm = norm_2(a_i);
228  mTau[i_node] = 1.0 / (2.0 * vel_norm/h_i + 0.01*time_inv );
229  }
230  mr_matrix_container.AssignVectorToVector(mphi_n, mWork); //mWork = mphi_n
231 
232  //first step of Runge Kutta
233 // mr_matrix_container.AssignVectorToVector(mphi_n,mphi_n1); //mphi_n1 = mphi_n
234  mr_matrix_container.SetToZero(rhs);
235  CalculateRHS( mphi_n1,mA,rhs);
236 
237  mr_matrix_container.Add_Minv_value(mWork,mWork, delta_t/6.0 , mr_matrix_container.GetInvertedMass(), rhs);
238  mr_matrix_container.Add_Minv_value(mphi_n1, mphi_n, 0.5*delta_t , mr_matrix_container.GetInvertedMass(), rhs);
239 
240  //second step
241  mr_matrix_container.SetToZero(rhs);
242  CalculateRHS(mphi_n1,mA,rhs);
243  mr_matrix_container.Add_Minv_value(mWork,mWork, delta_t/3.0 , mr_matrix_container.GetInvertedMass(), rhs);
244  mr_matrix_container.Add_Minv_value(mphi_n1, mphi_n, 0.5*delta_t , mr_matrix_container.GetInvertedMass(),rhs);
245 
246  //third step
247  CalculateAdvectiveVelocity(mUn, mUn1,mA, coefficient);
248  mr_matrix_container.SetToZero(rhs);
249  CalculateRHS( mphi_n1,mA,rhs);
250  mr_matrix_container.Add_Minv_value(mWork,mWork, delta_t/3.0 , mr_matrix_container.GetInvertedMass(), rhs);
251  mr_matrix_container.Add_Minv_value(mphi_n1, mphi_n, delta_t , mr_matrix_container.GetInvertedMass(), rhs);
252 
253  //fourth step
254  CalculateAdvectiveVelocity(mUn, mUn1,mA, coefficient);
255  mr_matrix_container.SetToZero(rhs);
256  CalculateRHS( mphi_n1,mA,rhs );
257  mr_matrix_container.Add_Minv_value(mWork,mWork, delta_t/6.0 , mr_matrix_container.GetInvertedMass(), rhs);
258  //compute right-hand side
259  mr_matrix_container.AssignVectorToVector(mWork,mphi_n1);
260 
261  mr_matrix_container.WriteScalarToDatabase(DISTANCE, mphi_n1, mr_model_part.Nodes());
262 
263 
264 
265 // //compute ratio for iteration
266  Vector stop_criteria(TDim);
267  noalias(stop_criteria) = ZeroVector(TDim);
268 // stop_criteria[0] = 0.0;
269 // stop_criteria[1] = 0.0;
270 
271  return stop_criteria;
272 
273  KRATOS_CATCH("")

279  //function to calculate right-hand side of fractional momentum equation
280  void CalculateRHS(
281  const ValuesVectorType& mphi,
282  const CalcVectorType& convective_velocity,
283  ValuesVectorType& rhs)
284  {
285  KRATOS_TRY
286 
287  int n_nodes = mphi.size();
288 
289  //calculating the convective projection
290  #pragma omp parallel for
291  for (int i_node = 0; i_node < n_nodes; i_node++)
292  {
293  double& pi_i = mPi[i_node];
294  const double& phi_i = mphi[i_node];
295 
296  //set to zero the projection
297  pi_i = 0;
298 
299  const array_1d<double, TDim>& a_i = convective_velocity[i_node];
300 
301  //loop to all the edges surrounding node I
302  for (unsigned int csr_index=mr_matrix_container.GetRowStartIndex()[i_node]; csr_index!=mr_matrix_container.GetRowStartIndex()[i_node+1]; csr_index++)
303  {
304  unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
305 
306  const array_1d<double, TDim>& a_j = convective_velocity[j_neighbour];
307  const double& phi_j = mphi[j_neighbour];
308 
309 
310  CSR_Tuple& edge_ij = mr_matrix_container.GetEdgeValues()[csr_index];
311 
312  edge_ij.Add_ConvectiveContribution(pi_i,a_i,phi_i,a_j,phi_j);
313  }
314 
315  //apply inverted mass matrix
316  const double m_inv = mr_matrix_container.GetInvertedMass()[i_node];
317  pi_i *= m_inv;
318 // KRATOS_WATCH(pi_i);
319  }
320 
321  //perform MPI syncronization
322 
323  //calculating the RHS
324  double stab_low;
325  double stab_high;
326  #pragma omp parallel for private(stab_low,stab_high)
327  for ( int i_node = 0; i_node < n_nodes; i_node++)
328  {
329  double& rhs_i = rhs[i_node];
330  const double& phi_i = mphi[i_node];
331  const array_1d<double, TDim>& a_i = convective_velocity[i_node];
332  const double& pi_i = mPi[i_node];
333  //double& h_i = mHmin[i_node];
334 
335  //initializing with the external forces (e.g. gravity)
336  rhs_i = 0.0;
337 
338  //loop to all the edges surrounding node I
339  for (unsigned int csr_index=mr_matrix_container.GetRowStartIndex()[i_node]; csr_index!=mr_matrix_container.GetRowStartIndex()[i_node+1]; csr_index++)
340  {
341  unsigned int j_neighbour = mr_matrix_container.GetColumnIndex()[csr_index];
342 
343  //double& rhs_j = rhs[j_neighbour];
344  const double& phi_j = mphi[j_neighbour];
345  const array_1d<double, TDim>& a_j = convective_velocity[j_neighbour];
346  const double& pi_j = mPi[j_neighbour];
347  //double& h_j = mHmin[j_neighbour];
348 
349  CSR_Tuple& edge_ij = mr_matrix_container.GetEdgeValues()[csr_index];
350 
351  //convection operator
352  edge_ij.Sub_ConvectiveContribution(rhs_i,a_i,phi_i,a_j,phi_j);
353 
354  //calculate stabilization part
355  edge_ij.CalculateConvectionStabilization_LOW( stab_low,a_i,phi_i,a_j,phi_j);
356 
357  double edge_tau = mTau[i_node];
358 
359  edge_ij.CalculateConvectionStabilization_HIGH( stab_high,a_i,pi_i,a_j,pi_j);
360 
361  edge_ij.Sub_StabContribution( rhs_i, edge_tau, 1.0, stab_low, stab_high);
362 
363  }
364 
365 // KRATOS_WATCH(rhs_i);
366 
367  }

371  KRATOS_CATCH("")

375  void CalculateAdvectiveVelocity(
376  const CalcVectorType& mUn,
377  const CalcVectorType& mUn1,
378  CalcVectorType& mA,
379  double coefficient)
380  {
381  int n_nodes = mUn1.size();
382  #pragma omp parallel for
383  for (int i_node = 0; i_node < n_nodes; i_node++)
384  {
385 
386  //reference for advective velocity of node i
387  array_1d<double, TDim>& a_i = mA[i_node];
388  const array_1d<double, TDim>& Un_i = mUn[i_node];
389  const array_1d<double, TDim>& Un1_i = mUn1[i_node];
390 
391  for (unsigned int k_comp = 0; k_comp < TDim; k_comp++)
392  a_i[k_comp] = coefficient * Un1_i[k_comp] + (1.0 - coefficient)* Un_i[k_comp];
393 
394  }
