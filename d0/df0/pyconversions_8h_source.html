211  static void* convertible(PyObject* obj_ptr)
212  {
213  using namespace boost::python;
214  using boost::python::allow_null; // works around gcc 2.96 bug
215  {
216  // Restriction to list, tuple, iter, xrange until
217  // Boost.Python overload resolution is enhanced.
218  //
219  // add PySequence_Check() for numarray.
220  //
221  if (!( PyList_Check(obj_ptr)
222  || PyTuple_Check(obj_ptr)
223  || PyIter_Check(obj_ptr)
224  || PyRange_Check(obj_ptr)
225  || PySequence_Check(obj_ptr) )) return 0;
226  }
227  handle<> obj_iter(allow_null(PyObject_GetIter(obj_ptr)));
228  if (!obj_iter.get()) { // must be convertible to an iterator
229  PyErr_Clear();
230  return 0;
231  }
232  if (ConversionPolicy::check_convertibility_per_element()) {
233  int obj_size = PyObject_Length(obj_ptr);
234  if (obj_size < 0) { // must be a measurable sequence
235  PyErr_Clear();
236  return 0;
237  }
238  if (!ConversionPolicy::check_size(
239  boost::type<ContainerType>(), obj_size)) return 0;
240  bool is_range = PyRange_Check(obj_ptr);
241  //std::size_t i=0;
242  int i = 0;
243 #ifndef _MSC_VER // because it causes c1001: internal compiler error
244  for(;;i++) {
245  handle<> py_elem_hdl(allow_null(PyIter_Next(obj_iter.get())));
246  if (PyErr_Occurred()) {
247  PyErr_Clear();
248  return 0;
249  }
250  if (!py_elem_hdl.get()) break; // end of iteration
251  object py_elem_obj(py_elem_hdl);
252  extract<container_element_type> elem_proxy(py_elem_obj);
253  if (!elem_proxy.check()) return 0;
254  if (is_range) break; // in a range all elements are of the same type
255  }
256  if (!is_range) assert(i == obj_size );
257 #endif
258  }
259  return obj_ptr;
260  }
