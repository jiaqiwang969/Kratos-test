44  /*
45  * geev() computes the eigendecomposition of a N x N matrix,
46  * where V is a N x N matrix and D is a diagonal matrix. The
47  * diagonal element D(i,i) is an eigenvalue of A and Q(:,i) is
48  * a corresponding eigenvector.
49  *
50  *
51  * int geev (char jobz, char uplo, A& a, W& w, V* vl, V* vr, optimal_workspace);
52  *
53  * a is the matrix whose eigendecomposition you're interested in. (input)
54  *
55  * w contains the diagonal of D, above. w must always be complex. (output)
56  *
57  * vl is an N x N matrix containing the left eigenvectors of a in its
58  * columns. See remark on complex vs. real below. May be left NULL to indicate
59  * that you do not want left eigenvectors.
60  *
61  * vr is an N x N matrix containing the right ("usual") eigenvectors of a in its
62  * columns. See remark on complex vs. real below. As a matrix, vr fulfills
63  * A * VR = VR * D. (except if real, see below). May be left NULL to indicate
64  * that you do not want right eigenvectors.
65  *
66  *
67  * For real A, vr and vl may be either complex or real, at your option.
68  * If you choose to leave them real, you have to pick apart the complex-conjugate
69  * eigenpairs as per the LAPACK documentation. If you choose them complex,
70  * the code will do the picking-apart on your behalf, at the expense of 4*N
71  * extra storage. Only if vr is complex, it will really fulfill its invariant
72  * on exit to the code in all cases, since complex pairs spoil that relation.
73  */

140  template <typename A, typename W, typename V>
141  int geev(real_case, const char jobvl, const char jobvr, A& a, W& w,
142  V* vl, V *vr)
143  {
144  int const n = traits::matrix_size1(a);
145  typedef typename A::value_type value_type;
146  traits::detail::array<value_type> wr(n);
147  traits::detail::array<value_type> wi(n);
148 
149  traits::detail::array<value_type> vl2(vl ? 0 : n);
150  traits::detail::array<value_type> vr2(vr ? 0 : n);
151  value_type* vl_real = vl ? traits::matrix_storage(*vl) : vl2.storage();
152  const int ldvl = vl ? traits::matrix_size2(*vl) : 1;
153  value_type* vr_real = vr ? traits::matrix_storage(*vr) : vr2.storage();
154  const int ldvr = vr ? traits::matrix_size2(*vr) : 1;
155 
156 
157  // workspace query
158  int lwork = -1;
159  value_type work_temp;
160  int result = geev_backend(&jobvl, &jobvr, &n,
161  traits::matrix_storage(a), &n,
162  wr.storage(), wi.storage(),
163  vl_real, &ldvl, vr_real, &ldvr,
164  &work_temp, &lwork);
165  if (result != 0)
166  return result;
167 
168  lwork = (int) work_temp;
169  traits::detail::array<value_type> work(lwork);
170  result = geev_backend(&jobvl, &jobvr, &n,
171  traits::matrix_storage(a), &n,
172  wr.storage(), wi.storage(),
173  vl_real, &ldvl, vr_real, &ldvr,
174  work.storage(), &lwork);
175 
176  for (int i = 0; i < n; i++)
177  w[i] = std::complex<value_type>(wr[i], wi[i]);
178  return result;
179  }

262  if (vr)
263  vr_stor[i*ldvr+j] = vr2[i*n+j];
264  }
265  }
266  }
