195  virtual VectorType& ApplyLeft(VectorType& rX)
196  {
197  const int size = TSparseSpaceType::Size(rX);
198  VectorType temp(size);
199  double sum;
200  int i, indexj;
201  for (i=0; i<size; i++) {
202  sum=rX[i];
203  for (indexj=iL[i]; indexj<iL[i+1]; indexj++) {
204  sum=sum-L[indexj]*temp[jL[indexj]];
205  }
206  temp[i]=sum;
207  }
208  for (i=size-1; i>=0; i--) {
209  sum=temp[i];
210  for (indexj=iU[i]+1; indexj<iU[i+1]; indexj++) {
211  sum=sum-U[indexj]*rX[jU[indexj]];
212  }
213  rX[i]=sum/U[iU[i]];
214  }
215  return rX;
216  }

222  virtual VectorType& ApplyTransposeLeft(VectorType& rX)
223  {
224  const int size = TSparseSpaceType::Size(rX);
225  VectorType temp(size);
226  int i, indexj;
227  double tempi, rxi;
228  for (i=0; i<size; i++) temp[i]=rX[i];
229  for (i=0; i<size; i++) {
230  temp[i]=temp[i]/U[iU[i]];
231  tempi=temp[i];
232  for (indexj=iU[i]+1; indexj<iU[i+1]; indexj++) {
233  temp[jU[indexj]]=temp[jU[indexj]]-tempi*U[indexj];
234  }
235  }
236  for (i=0; i<size; i++) rX[i]=temp[i];
237  for (i=size-1; i>=0; i--) {
238  rxi=rX[i];
239  for (indexj=iL[i]; indexj<iL[i+1]; indexj++) {
240  rX[jL[indexj]]=rX[jL[indexj]]-rxi*L[indexj];
241  }
242  }
243  return rX;
