492  virtual CoordinatesArrayType& PointLocalCoordinates( CoordinatesArrayType& rResult,
493  const CoordinatesArrayType& rPoint )
494  {
495  double tol = 1.0e-8;
496  int maxiter = 1000;
497  //check orientation of surface
498  std::vector< unsigned int> orientation(3);
499 
500  double dummy= this->GetPoint(0).X();
501  if(fabs(this->GetPoint(1).X()-dummy)<=tol&& fabs(this->GetPoint(2).X()-dummy)<=tol&& fabs(this->GetPoint(3).X()-dummy)<=tol)
502  orientation[0]=0;
503 
504  dummy= this->GetPoint(0).Y();
505  if(fabs(this->GetPoint(1).Y()-dummy)<=tol&& fabs(this->GetPoint(2).Y()-dummy)<=tol&& fabs(this->GetPoint(3).Y()-dummy)<=tol)
506  orientation[0]=1;
507 
508  dummy= this->GetPoint(0).Z();
509  if(fabs(this->GetPoint(1).Z()-dummy)<=tol&& fabs(this->GetPoint(2).Z()-dummy)<=tol&& fabs(this->GetPoint(3).Z()-dummy)<=tol)
510  orientation[0]=2;
511 
512  switch(orientation[0])
513  {
514  case 0:
515  orientation[0]= 1; orientation[1]= 2; orientation[2]= 0;
516  break;
517  case 1:
518  orientation[0]= 0; orientation[1]= 2; orientation[2]= 1;
519  break;
520  case 2:
521  orientation[0]= 0; orientation[1]= 1; orientation[2]= 2;
522  break;
523  default:
524  orientation[0]= 0; orientation[1]= 1; orientation[2]= 2;
525  }
526 
527  Matrix J = ZeroMatrix( 2, 2 );
528  Matrix invJ = ZeroMatrix( 2, 2 );
529 
530  if( rResult.size() != 2 )
531  rResult.resize( 2 );
532  //starting with xi = 0
533  rResult = ZeroVector( 2 );
534  Vector DeltaXi = ZeroVector( 2 );
535  CoordinatesArrayType CurrentGlobalCoords( ZeroVector(3) );
536  //Newton iteration:
537  for( int k=0; k<maxiter; k++ )
538  {
539  CurrentGlobalCoords = ZeroVector(3);
540  GlobalCoordinates( CurrentGlobalCoords, rResult );
541  noalias( CurrentGlobalCoords ) = rPoint - CurrentGlobalCoords;
542 
543  //Caluclate Inverse of Jacobian
544  J.resize(2,2);
545  //derivatives of shape functions
546  Matrix shape_functions_gradients;
547  shape_functions_gradients = ShapeFunctionsLocalGradients(
548  shape_functions_gradients, rResult );
549 
550  //Elements of jacobian matrix (e.g. J(1,1) = dX1/dXi1)
551  //loop over all nodes
552  for( unsigned int i=0; i<this->PointsNumber(); i++ )
553  {
554  Point<3> dummyPoint= this->GetPoint(i);
555  J(0,0) += (dummyPoint[orientation[0]])*(shape_functions_gradients(i,0));
556  J(0,1) += (dummyPoint[orientation[0]])*(shape_functions_gradients(i,1));
557  J(1,0) += (dummyPoint[orientation[1]])*(shape_functions_gradients(i,0));
558  J(1,1) += (dummyPoint[orientation[1]])*(shape_functions_gradients(i,1));
559  }
560  //deteminant of Jacobian
561  double det_j = J(0,0)*J(1,1)-J(0,1)*J(1,0);
562  //filling matrix
563  invJ(0,0) = (J(1,1))/(det_j);
564  invJ(1,0) = -(J(1,0))/(det_j);
565  invJ(0,1) = -(J(0,1))/(det_j);
566  invJ(1,1) = (J(0,0))/(det_j);
567 
568 
569  DeltaXi(0) = invJ(0,0)*CurrentGlobalCoords(orientation[0])+invJ(0,1)*CurrentGlobalCoords(orientation[1]);
570  DeltaXi(1) = invJ(1,0)*CurrentGlobalCoords(orientation[0])+invJ(1,1)*CurrentGlobalCoords(orientation[1]);
571 
572  noalias( rResult ) += DeltaXi;
573 
574  if( MathUtils<double>::Norm3(DeltaXi) > 30 )
575  {
576  break;
577  }
578  if( MathUtils<double>::Norm3(DeltaXi) < tol )
579  {
580  if(!(fabs(CurrentGlobalCoords(orientation[2]))<=tol))
581  rResult(0)=2.0;
582  break;
583  }
584  }
585  return( rResult );
586  }

616  CalculateShapeFunctionsIntegrationPointsValues(ThisMethod);

621  // vector resize, I have to put this beside resizing!!
622  JacobiansType temp(this->IntegrationPointsNumber(ThisMethod));
623  rResult.swap(temp);
624  }
625  //loop over all integration points
626  for( unsigned int pnt=0; pnt < this->IntegrationPointsNumber(ThisMethod); pnt++ )
627  {
628  //defining single jacobian matrix
629  Matrix jacobian = ZeroMatrix(3,2);
630  //loop over all nodes
631  for( unsigned int i=0; i<this->PointsNumber(); i++ )

639  jacobian(1,1) +=
640  (this->GetPoint(i).Y())*(shape_functions_gradients[pnt](i,1));
641  jacobian(2,0) +=
642  (this->GetPoint(i).Z())*(shape_functions_gradients[pnt](i,0));
643  jacobian(2,1) +=
644  (this->GetPoint(i).Z())*(shape_functions_gradients[pnt](i,1));
645  }
646  rResult[pnt] = jacobian;
647  }//end of loop over all integration points
