133  {
134  using namespace boost::python;
135  using boost::python::allow_null; // works around gcc 2.96 bug
136  using boost::python::handle; // works around intel 9.1
137  {
138  // Restriction to list, tuple, iter, xrange until
139  // Boost.Python overload resolution is enhanced.
140  if (!( PyList_Check(obj_ptr)
141  || PyTuple_Check(obj_ptr)
142  || PyIter_Check(obj_ptr)
143  || PyRange_Check(obj_ptr))){ return 0; }
144  }
145 
146  boost::python::handle<> obj_iter( allow_null(PyObject_GetIter(obj_ptr) ) );
147 
148  if (!obj_iter.get()) { // must be convertible to an iterator
149  PyErr_Clear();
150  return 0;
151  }
152  if (true) {
153  int obj_size = PyObject_Length(obj_ptr);
154  if (obj_size < 0) { // must be a measurable sequence
155  PyErr_Clear();
156  return 0;
157  }
158 /* if (!TConversionPolicyType::check_size( */
159 /* boost::type<TContainerType>(), obj_size)) return 0; */
160  bool is_range = PyRange_Check(obj_ptr);
161  std::size_t i=0;
162  for(;;i++) {
163 
164  boost::python::handle<> py_elem_hdl(allow_null(PyIter_Next(obj_iter.get())));
165 
166  if (PyErr_Occurred()) {
167  PyErr_Clear();
168  return 0;
169  }
170  if (!py_elem_hdl.get()) break; // end of iteration
171  boost::python::object py_elem_obj(py_elem_hdl);
172  boost::python::extract<typename TContainerType::value_type> elem_proxy(py_elem_obj);
