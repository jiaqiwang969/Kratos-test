399  void CalculateElementData(const array_1d<double,3>& c1,
400  const array_1d<double,3>& c2,
401  const array_1d<double,3>& c3,
402  const array_1d<double,3>& c4,
403  double& volume,
404  array_1d<double,3>& xc,
405  double& radius,
406  double& hmin,
407  double& hmax
408  )
409  {
410  KRATOS_TRY
411  const double& x0 = c1[0]; const double& y0 = c1[1]; const double& z0 = c1[2];
412  const double& x1 = c2[0]; const double& y1 = c2[1]; const double& z1 = c2[2];
413  const double& x2 = c3[0]; const double& y2 = c3[1]; const double& z2 = c3[2];
414  const double& x3 = c4[0]; const double& y3 = c4[1]; const double& z3 = c4[2];
415 
416  //calculate min side lenght
417  //(use xc as a auxiliary vector) !!!!!!!!!!!!
418  double aux;
419  noalias(xc) = c4; noalias(xc)-=c1; aux = inner_prod(xc,xc); hmin = aux; hmax = aux;
420  noalias(xc) = c3; noalias(xc)-=c1; aux = inner_prod(xc,xc); if(aux<hmin) hmin = aux; else if(aux>hmax) hmax = aux;
421  noalias(xc) = c2; noalias(xc)-=c1; aux = inner_prod(xc,xc); if(aux<hmin) hmin = aux;
422  noalias(xc) = c4; noalias(xc)-=c2; aux = inner_prod(xc,xc); if(aux<hmin) hmin = aux;
423  noalias(xc) = c3; noalias(xc)-=c2; aux = inner_prod(xc,xc); if(aux<hmin) hmin = aux;
424  noalias(xc) = c4; noalias(xc)-=c3; aux = inner_prod(xc,xc); if(aux<hmin) hmin = aux;
425  hmin = sqrt(hmin);
426  hmax = sqrt(hmax);
427 
428  //calculation of the jacobian
429  mJ(0,0) = x1-x0; mJ(0,1) = y1-y0; mJ(0,2) = z1-z0;
430  mJ(1,0) = x2-x0; mJ(1,1) = y2-y0; mJ(1,2) = z2-z0;
431  mJ(2,0) = x3-x0; mJ(2,1) = y3-y0; mJ(2,2) = z3-z0;
432 
433  //inverse of the jacobian
434  //first column
435  mJinv(0,0) = mJ(1,1)*mJ(2,2) - mJ(1,2)*mJ(2,1);
436  mJinv(1,0) = -mJ(1,0)*mJ(2,2) + mJ(1,2)*mJ(2,0);
437  mJinv(2,0) = mJ(1,0)*mJ(2,1) - mJ(1,1)*mJ(2,0);
438  //second column
439  mJinv(0,1) = -mJ(0,1)*mJ(2,2) + mJ(0,2)*mJ(2,1);
440  mJinv(1,1) = mJ(0,0)*mJ(2,2) - mJ(0,2)*mJ(2,0);
441  mJinv(2,1) = -mJ(0,0)*mJ(2,1) + mJ(0,1)*mJ(2,0);
442  //third column
443  mJinv(0,2) = mJ(0,1)*mJ(1,2) - mJ(0,2)*mJ(1,1);
444  mJinv(1,2) = -mJ(0,0)*mJ(1,2) + mJ(0,2)*mJ(1,0);
445  mJinv(2,2) = mJ(0,0)*mJ(1,1) - mJ(0,1)*mJ(1,0);
446  //calculation of determinant (of the input matrix)
447 
448  double detJ = mJ(0,0)*mJinv(0,0)
449  + mJ(0,1)*mJinv(1,0)
450  + mJ(0,2)*mJinv(2,0);
451 
452  volume = detJ * 0.16666666667;
453 
454  if(volume < 1e-3 * hmax*hmax*hmax) //this is a sliver and we will remove it
455  {
456  //very bad element // ser a very high center for it to be eliminated
457  xc[0]=0.0; xc[1]=0.0; xc[2]=0.0;
458  radius = 10000000;
459  }
460  else
461  {
462 
463  double x0_2 = x0*x0 + y0*y0 + z0*z0;
464  double x1_2 = x1*x1 + y1*y1 + z1*z1;
465  double x2_2 = x2*x2 + y2*y2 + z2*z2;
466  double x3_2 = x3*x3 + y3*y3 + z3*z3;
467 
468  //finalizing the calculation of the inverted matrix
469  mJinv /= detJ;
470 
471  //calculating the RHS
472  //calculating the RHS
473  mRhs[0] = 0.5*(x1_2 - x0_2);
474  mRhs[1] = 0.5*(x2_2 - x0_2);
475  mRhs[2] = 0.5*(x3_2 - x0_2);
476 
477  //calculate position of the center
478  noalias(xc) = prod(mJinv,mRhs);
479 
480  //calculate radius
481  radius = pow(xc[0] - x0,2);
482  radius += pow(xc[1] - y0,2);
483  radius += pow(xc[2] - z0,2);
484  radius = sqrt(radius);
485  }
486  KRATOS_CATCH("")
487  }
