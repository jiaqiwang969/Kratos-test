358  MatrixType& A,
359  Matrix& LHS_Contribution,
360  std::vector<std::size_t>& EquationId
361  )
362  {
363  unsigned int system_size = Size1(A);
364  //unsigned int local_size = LHS_Contribution.size1();
365 
366  //count active indices
367  unsigned int active_indices = 0;
368  for(unsigned int i=0; i<EquationId.size(); i++)
369  if ( EquationId[i] < system_size )
370  active_indices += 1;
371 
372  if(active_indices > 0)
373  {
374  //size Epetra vectors
375  Epetra_IntSerialDenseVector indices(active_indices);
376  Epetra_SerialDenseMatrix values(active_indices,active_indices);
377 
378  //fill epetra vectors
379  int loc_i = 0;
380  for(unsigned int i=0; i<EquationId.size(); i++)
381  {
382  if ( EquationId[i] < system_size )
383  {
384  indices[loc_i] = EquationId[i];
385 
386  int loc_j = 0;
387  for(unsigned int j=0; j<EquationId.size(); j++)
388  {
389  if ( EquationId[j] < system_size )
390  {
391  values(loc_i,loc_j) = LHS_Contribution(i,j);
392  loc_j += 1;
393  }
394  }
395  loc_i += 1;
396  }
397  }
398 
399  int ierr = A.SumIntoGlobalValues(indices,values);
400  assert(ierr);
