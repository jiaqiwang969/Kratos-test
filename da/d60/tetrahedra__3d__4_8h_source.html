491  double z3 = this->Points()[3].Z();

493  //calculation of the jacobian
494  jacobian(0,0) = x1-x0; jacobian(0,1) = x2-x0; jacobian(0,2) = x3-x0;
495  jacobian(1,0) = y1-y0; jacobian(1,1) = y2-y0; jacobian(1,2) = y3-y0;
496  jacobian(2,0) = z1-z0; jacobian(2,1) = z2-z0; jacobian(2,2) = z3-z0;
497 
498  if(rResult.size() != this->IntegrationPointsNumber(ThisMethod))
499  {
500  // KLUDGE: While there is a bug in ublas vector resize, I have to put this beside resizing!!
501  JacobiansType temp(this->IntegrationPointsNumber(ThisMethod));
502  rResult.swap(temp);
503  }
504  std::fill(rResult.begin(), rResult.end(), jacobian);
505  return rResult;
