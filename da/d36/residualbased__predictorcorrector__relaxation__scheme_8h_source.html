450  void InitializeSolutionStep(
451  ModelPart& r_model_part,
452  TSystemMatrixType& A,
453  TSystemVectorType& Dx,
454  TSystemVectorType& b
455  )
456  {
457  ProcessInfo& CurrentProcessInfo = r_model_part.GetProcessInfo();
458 
459  Scheme<TSparseSpace,TDenseSpace>::InitializeSolutionStep(r_model_part,A,Dx,b);
460 
461  double DeltaTime = CurrentProcessInfo[DELTA_TIME];
462 
463  if(DeltaTime == 0)
464  KRATOS_ERROR(std::logic_error,"detected delta_time = 0 in the Bossak Scheme ... check if the time step is created correctly for the current model part","");
465 
466  //initializing constants
467  ma0 = 1.0/(mBetaNewmark*pow(DeltaTime,2));
468  ma1 = mGammaNewmark / (mBetaNewmark*DeltaTime);
469  ma2 = 1.0/(mBetaNewmark*DeltaTime);
470  ma3 = 1.0/(2.0*mBetaNewmark) - 1.0;
471  ma4 = mGammaNewmark/mBetaNewmark - 1.0;
472  ma5 = DeltaTime*0.5*(mGammaNewmark/mBetaNewmark-2.0);
473  mam = (1.0-mAlphaBossak)/(mBetaNewmark*pow(DeltaTime,2));
474  }
