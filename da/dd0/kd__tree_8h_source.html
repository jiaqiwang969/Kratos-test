347  static CoordinateType Spread( IteratorType PointsBegin, IteratorType PointsEnd, IndexType CoordinateIndex )
348  {
349  if(PointsBegin == PointsEnd)
350  return CoordinateType(); // If there is no point so the distance is zero!!
351 
352  CoordinateType min = (**PointsBegin)[CoordinateIndex];
353  CoordinateType max = (**PointsBegin)[CoordinateIndex];
354  for (; PointsBegin != PointsEnd; PointsBegin++)
355  {
356  CoordinateType c = (**PointsBegin)[CoordinateIndex];
357  if (c < min)
358  min = c;
359  else if (c > max)
360  max = c;
361  }
362  return (max - min);
363  }

366  static SizeType MaxSpread( IteratorType PointsBegin, IteratorType PointsEnd )
367  {
368  SizeType max_dimension = 0; // dimension of max spread

385  static void MedianSplit( IteratorType PointsBegin, IteratorType PartitionPosition, IteratorType PointsEnd,
386  IndexType CuttingDimension, CoordinateType& rCuttingValue )
387  {
388  IteratorType left = PointsBegin;
389  IteratorType right = PointsEnd - 1;
390  while (left < right) { // Iterating to find the partition point
391 
392  IteratorType i = left; // selecting left as pivot
393  if ((**i)[CuttingDimension] > (**right)[CuttingDimension])
394  std::swap(*i,*right);
395 
396  CoordinateType value = (**i)[CuttingDimension];
397  IteratorType j = right;
398  for(;;)
399  {
400  while ((**(++i))[CuttingDimension] < value);
401  while ((**(--j))[CuttingDimension] > value);
402  if (i < j) std::swap(*i,*j); else break;
403  }
404  std::swap(*left,*j);
405 
406  if (j > PartitionPosition)
407  right = j - 1;
408  else if (j < PartitionPosition)
409  left = j + 1;
410  else break;
411  }
412 
413  CoordinateType max = (**PointsBegin)[CuttingDimension];
414  IteratorType k = PointsBegin;
415  IteratorType last = PartitionPosition;
416 
417  for(IteratorType i = PointsBegin ; i != PartitionPosition ; ++i)
418  if((**i)[CuttingDimension] > max)
419  {
420  max = (**i)[CuttingDimension];
421  k = i;
422  }
423 
424  if(k != PointsBegin)
425  std::swap(--last, k);
426 
427  rCuttingValue = ((**last)[CuttingDimension] + (**PartitionPosition)[CuttingDimension])/2.0;
428  }
