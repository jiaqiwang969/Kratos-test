192  VectorType x(TDenseSpaceType::Size1(rX));
193  VectorType b(TDenseSpaceType::Size1(rB));
194  for(unsigned int i = 0 ; i < TDenseSpaceType::Size2(rX) ; i++)
195  {

310  bool IterativeSolve(SparseMatrixType& rA, VectorType& rX, VectorType& rB)
311  {
312  const int size = TSparseSpaceType::Size(rX);
313 
314  BaseType::mIterationsNumber = 0;
315 
316  VectorType r(size);
317 
318  PreconditionedMult(rA,rX,r);
319  TSparseSpaceType::ScaleAndAdd(1.00, rB, -1.00, r);
320 
321  BaseType::mBNorm = TSparseSpaceType::TwoNorm(rB);
322 
323  VectorType p(r);
324  VectorType q(size);
325 
326  double roh0 = TSparseSpaceType::Dot(r, r);
327  double roh1 = roh0;
328  double beta = 0;
329 
330  if(fabs(roh0) < 1.0e-30) //modification by Riccardo
331 // if(roh0 == 0.00)
332  return false;
333 
334  do
335  {
336  PreconditionedMult(rA,p,q);
337 
338  double pq = TSparseSpaceType::Dot(p,q);
339 
340  //if(pq == 0.00)
341  if(fabs(pq) <= 1.0e-30)
342  break;
343 
344  double alpha = roh0 / pq;
345 
346  TSparseSpaceType::ScaleAndAdd(alpha, p, 1.00, rX);
347  TSparseSpaceType::ScaleAndAdd(-alpha, q, 1.00, r);
348 
349  roh1 = TSparseSpaceType::Dot(r,r);
350 
351  beta = (roh1 / roh0);
352  TSparseSpaceType::ScaleAndAdd(1.00, r, beta, p);
353 
354  roh0 = roh1;
355 
356  BaseType::mResidualNorm = sqrt(roh1);
357  BaseType::mIterationsNumber++;
358  } while(BaseType::IterationNeeded() && (fabs(roh0) > 1.0e-30)/*(roh0 != 0.00)*/);
359 
360  return BaseType::IsConverged();
361  }
