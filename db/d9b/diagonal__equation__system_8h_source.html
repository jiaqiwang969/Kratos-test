218  static IndexType AssignDofsEquationIds(DofsArrayType & rDofs)
219  {
220  SizeType free_id = 0;
221  SizeType fix_id = rDofs.size();
222 
223  for (typename DofsArrayType::iterator dof_iterator = rDofs.begin(); dof_iterator != rDofs.end(); ++dof_iterator)
224  if (dof_iterator->IsFixed())
225  dof_iterator->SetEquationId(--fix_id);
226  else
227  dof_iterator->SetEquationId(free_id++);
228 
229  return fix_id;

266  static void AssembleMatrix(SystemMatrixType& rSystemMatrix, LocalMatrixType const& LocalMatrix, DofsArrayType const& Dofs)
267  {
268  KRATOS_TRY_LEVEL_4
269 
270  SizeType global_size = TSystemSpace::Size(rSystemMatrix);
271 
272  SizeType local_size = TLocalSpace::Size1(LocalMatrix);
273 
274  for(SizeType i_local=0; i_local<local_size; i_local++)
275  {
276  SizeType i_global= Dofs[i_local].EquationId();
277  if ( i_global < global_size)
278  {
279  rSystemMatrix[i_global] += LocalMatrix(i_local, i_local);
280  }
281  }
282  KRATOS_CATCH_LEVEL_4("")
283  }

392  void Resize(SizeType NewSize)
393  {
394  mEquationSystemSize = NewSize;
395  mDirichletSize = (mDofs.size() > mEquationSystemSize) ? (mDofs.size() - mEquationSystemSize) : 0;
396  TSystemSpace::Resize(mSystemMatrix, mEquationSystemSize, mEquationSystemSize);
397  TSystemSpace::Resize(mDirichletMatrix, mEquationSystemSize, mDirichletSize);
398  TSystemSpace::Resize(mResults, mEquationSystemSize);
399  TSystemSpace::Resize(mRightHandSide, mEquationSystemSize);
400  std::cout << "Equation system size : " << mEquationSystemSize << std::endl;
401  std::cout << "Dirichlet system size : " << mDirichletSize << std::endl;
