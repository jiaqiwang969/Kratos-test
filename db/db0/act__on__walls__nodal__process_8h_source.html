143  //removes the velocity towards the wall for all the nodes which are getting to close
144  virtual void Execute()
145  {
146  KRATOS_TRY
147 
148  array_1d<double,3> n, final_pos;
149  array_1d<double,3> dist, corrective_disp, delta_disp_wall;
150 
151  for(ModelPart::NodesContainerType::iterator i = mr_model_part.NodesBegin();
152  i!=mr_model_part.NodesEnd(); i++)
153  {
154  // const WeakPointerVector<Node<3> >& rN = i->GetValue(NEIGHBOUR_NODES);
155 
156  //if the conditions is composed only by wall nodes remove a percentage of the velocity
157  if( i->FastGetSolutionStepValue(IS_STRUCTURE) == 1
158  && (i->GetValue(NEIGHBOUR_NODES)).size() != 0)
159  {
160 
161  noalias(n) = i->FastGetSolutionStepValue(NORMAL_TO_WALL);
162  double A = inner_prod(n,n);
163  A = sqrt(A);
164  const array_1d<double,3>& disp_wall = i->FastGetSolutionStepValue(DISPLACEMENT);
165  const array_1d<double,3>& disp_wall_old = i->FastGetSolutionStepValue(DISPLACEMENT,1);
166  noalias(delta_disp_wall) = disp_wall;
167  noalias(delta_disp_wall) -= disp_wall_old;
168  if(A != 0)
169  {
170  n /= A;
171 
172  double x = i->X();
173  double y = i->Y();
174  double z = i->Z();
175  double h = i->FastGetSolutionStepValue(NODAL_H);
176 
177  WeakPointerVector<Node<3> >& rN = i->GetValue(NEIGHBOUR_NODES);
178  for(WeakPointerVector<Node<3> >::iterator in = rN.begin(); in != rN.end(); in++)
179  {
180  if(in->FastGetSolutionStepValue(IS_STRUCTURE) == 0) //for nodes that are not of structure
181  {
182  //calculating the distance along the normal and orhtogonal to the normal
183  array_1d<double,3>& disp = in->FastGetSolutionStepValue(DISPLACEMENT);
184  array_1d<double,3>& olddisp = in->FastGetSolutionStepValue(DISPLACEMENT,1);
185 
186  array_1d<double,3> delta_disp = disp;
187  noalias(delta_disp) -= olddisp;
188 
189  //calculating the end of the step position
190  final_pos[0] = in->X0(); final_pos[1] = in->Y0(); final_pos[2] = in->Z0();
191  noalias(final_pos) += disp;
192 
193  //calculating the distance at the end of the step
194  noalias(dist) = final_pos;
195  dist[0] = final_pos[0] - x;
196  dist[1] = final_pos[1] - y;
197  dist[2] = final_pos[2] - z;
198 
199  double dist_2 = inner_prod(dist,dist);
200  double dist_n = inner_prod(dist,n);
201  double dist_n2 = dist_n * dist_n;
202  double dist_ort2 = dist_2 - dist_n2;
203 
204  //verify that the final position of the point is close to the normal in the wall node
205  if(dist_ort2 < 0.5*h*h)
206  {
207  //verify if the final position of the normal closer than 0.75h to the face (distance along the normal)
208  if(dist_n > -0.5 * h)
209  {
210  double a = inner_prod(delta_disp,n);
211 
212  //if it is moving towards the outside put it back
213  // to the allowed distance of 0.75h on the interior
214  if(a > 0)
215  {
216  double coeff = - 0.5*h - dist_n;
217  noalias(disp) += coeff * n;
218  noalias(disp) += delta_disp_wall;
219  }
220 
221  }
222 
223  }

225  }
226  }
