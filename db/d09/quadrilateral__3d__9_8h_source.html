472  virtual CoordinatesArrayType& PointLocalCoordinates( CoordinatesArrayType& rResult,
473  const CoordinatesArrayType& rPoint )
474  {
475  double tol = 1.0e-8;
476  int maxiter = 1000;
477  //check orientation of surface
478  std::vector< unsigned int> orientation(3);
479 
480  double dummy= this->GetPoint(0).X();
481  if(fabs(this->GetPoint(1).X()-dummy)<=tol&& fabs(this->GetPoint(2).X()-dummy)<=tol&& fabs(this->GetPoint(3).X()-dummy)<=tol)
482  orientation[0]=0;
483 
484  dummy= this->GetPoint(0).Y();
485  if(fabs(this->GetPoint(1).Y()-dummy)<=tol&& fabs(this->GetPoint(2).Y()-dummy)<=tol&& fabs(this->GetPoint(3).Y()-dummy)<=tol)
486  orientation[0]=1;
487 
488  dummy= this->GetPoint(0).Z();
489  if(fabs(this->GetPoint(1).Z()-dummy)<=tol&& fabs(this->GetPoint(2).Z()-dummy)<=tol&& fabs(this->GetPoint(3).Z()-dummy)<=tol)
490  orientation[0]=2;
491 
492  switch(orientation[0])
493  {
494  case 0:
495  orientation[0]= 1; orientation[1]= 2; orientation[2]= 0;
496  break;
497  case 1:
498  orientation[0]= 0; orientation[1]= 2; orientation[2]= 1;
499  break;
500  case 2:
501  orientation[0]= 0; orientation[1]= 1; orientation[2]= 2;
502  break;
503  default:
504  orientation[0]= 0; orientation[1]= 1; orientation[2]= 2;
505  }
506 
507  Matrix J = ZeroMatrix( 2, 2 );
508  Matrix invJ = ZeroMatrix( 2, 2 );
509 
510  if( rResult.size() != 2 )
511  rResult.resize( 2 );
512  //starting with xi = 0
513  rResult = ZeroVector( 2 );
514  Vector DeltaXi = ZeroVector( 2 );
515  CoordinatesArrayType CurrentGlobalCoords( ZeroVector(3) );
516  //Newton iteration:
517  for( int k=0; k<maxiter; k++ )
518  {
519  CurrentGlobalCoords = ZeroVector(3);
520  GlobalCoordinates( CurrentGlobalCoords, rResult );
521  noalias( CurrentGlobalCoords ) = rPoint - CurrentGlobalCoords;
522 
523  //Caluclate Inverse of Jacobian
524  J.resize(2,2);
525  //derivatives of shape functions
526  Matrix shape_functions_gradients;
527  shape_functions_gradients = ShapeFunctionsLocalGradients(
528  shape_functions_gradients, rResult );
529 
530  //Elements of jacobian matrix (e.g. J(1,1) = dX1/dXi1)
531  //loop over all nodes
532  for( unsigned int i=0; i<this->PointsNumber(); i++ )
533  {
534  Point<3> dummyPoint= this->GetPoint(i);
535  J(0,0) += (dummyPoint[orientation[0]])*(shape_functions_gradients(i,0));
536  J(0,1) += (dummyPoint[orientation[0]])*(shape_functions_gradients(i,1));
537  J(1,0) += (dummyPoint[orientation[1]])*(shape_functions_gradients(i,0));
538  J(1,1) += (dummyPoint[orientation[1]])*(shape_functions_gradients(i,1));
539  }
540  //deteminant of Jacobian
541  double det_j = J(0,0)*J(1,1)-J(0,1)*J(1,0);
542  //filling matrix
543  invJ(0,0) = (J(1,1))/(det_j);
544  invJ(1,0) = -(J(1,0))/(det_j);
545  invJ(0,1) = -(J(0,1))/(det_j);
546  invJ(1,1) = (J(0,0))/(det_j);
547 
548 
549  DeltaXi(0) = invJ(0,0)*CurrentGlobalCoords(orientation[0])+invJ(0,1)*CurrentGlobalCoords(orientation[1]);
550  DeltaXi(1) = invJ(1,0)*CurrentGlobalCoords(orientation[0])+invJ(1,1)*CurrentGlobalCoords(orientation[1]);
551 
552  noalias( rResult ) += DeltaXi;
553 
554  if( MathUtils<double>::Norm3(DeltaXi) > 30 )
555  {
556  break;
557  }
558  if( MathUtils<double>::Norm3(DeltaXi) < tol )
559  {
560  if(!(fabs(CurrentGlobalCoords(orientation[2]))<=tol))
561  rResult(0)=2.0;
562  break;
563  }
564  }
565  return( rResult );
566  }
