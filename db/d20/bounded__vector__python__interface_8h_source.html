134  static void* convertible(PyObject* obj_ptr)
135  {
136  using namespace boost::python;
137  using boost::python::allow_null; // works around gcc 2.96 bug
138  {
139  // Restriction to list, tuple, iter, xrange until
140  // Boost.Python overload resolution is enhanced.
141  if (!( PyList_Check(obj_ptr)
142  || PyTuple_Check(obj_ptr)
143  || PyIter_Check(obj_ptr)
144  || PyRange_Check(obj_ptr))) return 0;
145  }
146  handle<> obj_iter(allow_null(PyObject_GetIter(obj_ptr)));
147  if (!obj_iter.get()) { // must be convertible to an iterator
148  PyErr_Clear();
149  return 0;
150  }
151  if (true) {
152  int obj_size = PyObject_Length(obj_ptr);
153  if (obj_size < 0) { // must be a measurable sequence
154  PyErr_Clear();
155  return 0;
156  }
157  if (TSize < obj_size) return 0;
158  bool is_range = PyRange_Check(obj_ptr);
159  std::size_t i=0;
160  for(;;i++) {
161  handle<> py_elem_hdl(allow_null(PyIter_Next(obj_iter.get())));
162  if (PyErr_Occurred()) {
163  PyErr_Clear();
164  return 0;
165  }
166  if (!py_elem_hdl.get()) break; // end of iteration
167  object py_elem_obj(py_elem_hdl);
168  extract<typename TContainerType::value_type> elem_proxy(py_elem_obj);
169  if (!elem_proxy.check()) return 0;
170  if (is_range) break; // in a range all elements are of the same type
171  }
172  if (!is_range) assert(i == obj_size);
173  }
174  return obj_ptr;
175  }
