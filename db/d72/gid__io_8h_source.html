416  virtual void InitializeResults( double name, MeshType rThisMesh )
417  {
418  if( mMode == GiD_PostAscii && ! mResultFileOpen )
419  {
420  std::stringstream file_name;
421  file_name << mResultFileName << "_" << name << ".post.res";
422  GiD_OpenPostResultFile((char*)(file_name.str()).c_str(), mMode);
423  mResultFileOpen = true;
424  }
425  if( mUseMultiFile == SingleFile && mMode == GiD_PostBinary )
426  {
427  std::stringstream step_index;
428  step_index << "step" << name;
429  GiD_BeginOnMeshGroup( (char *)(step_index.str()).c_str() );
430  }
431  //initializing gauss points containers
432  for( MeshType::ElementIterator element_iterator = rThisMesh.ElementsBegin();
433  element_iterator != rThisMesh.ElementsEnd(); ++element_iterator )
434  for( typename std::vector<TGaussPointContainer>::iterator it =
435  mGidGaussPointContainers.begin();
436  it != mGidGaussPointContainers.end(); it++ )
437  if( it->AddElement( element_iterator ) )
438  break;
439 
440  if( mWriteConditions == WriteConditions )
441  for( MeshType::ConditionsContainerType::iterator conditions_iterator =
442  rThisMesh.ConditionsBegin(); conditions_iterator
443  != rThisMesh.ConditionsEnd(); conditions_iterator++ )
444  for( typename std::vector<TGaussPointContainer>::iterator it =
445  mGidGaussPointContainers.begin();
446  it != mGidGaussPointContainers.end(); it++ )
447  if( it->AddCondition( conditions_iterator ) )
448  break;
449  }
