541  static void ParallelProductNoAdd( const MatrixType& A, const VectorType& in, VectorType& out)
542  {
543 //std::cout << "in function ParallelProductNoAdd" << std::endl;
544  typedef unsigned int size_type;
545  typedef double value_type;
546 
547  //create partition
548  vector<size_type> partition;
549  int number_of_threads = omp_get_max_threads();
550  CreatePartition(number_of_threads, A.size1(), partition);
551  //parallel loop
552  size_type processor_row_begin, processor_row_end;
553  int proc_id = 0;
554 
555  #pragma omp parallel
556  {
557  int thread_id = omp_get_thread_num();
558  int number_of_rows = partition[thread_id+1] - partition[thread_id];
559  typename MatrixType::index_array_type::const_iterator row_iter_begin = A.index1_data().begin()+partition[thread_id];
560  typename MatrixType::index_array_type::const_iterator index_2_begin = A.index2_data().begin()+*row_iter_begin;
561  typename MatrixType::value_array_type::const_iterator value_begin = A.value_data().begin()+*row_iter_begin;
562  typename VectorType::iterator output_vec_begin = out.begin()+partition[thread_id];
563 
564 
565  partial_product_no_add( number_of_rows,
566  row_iter_begin,
567  index_2_begin,
568  value_begin,
569  in,
570  output_vec_begin
571  );
572  }

575  static void CreatePartition(unsigned int number_of_threads,const int number_of_rows, vector<unsigned int>& partitions)
576  {
577  partitions.resize(number_of_threads+1);
578  int partition_size = number_of_rows / number_of_threads;
579  partitions[0] = 0;
580  partitions[number_of_threads] = number_of_rows;
581  for(int i = 1; i<number_of_threads; i++)
582  partitions[i] = partitions[i-1] + partition_size ;
583  }

588  static void partial_product_no_add(
589  int number_of_rows,
590  typename TMatrixType::index_array_type::const_iterator row_begin,
591  typename TMatrixType::index_array_type::const_iterator index2_begin,
592  typename TMatrixType::value_array_type::const_iterator value_begin,
593  const VectorType& input_vec,
594  typename VectorType::iterator output_vec_begin
595  )
596  {
597  int row_size;
598  typename MatrixType::index_array_type::const_iterator row_it = row_begin;
599  for(int k = 0; k < number_of_rows; k++)
600  {
601  row_size= *(row_it+1)-*row_it;
602  row_it++;
603  TDataType t = TDataType();
604 
605  for(int i = 0; i<row_size; i++)
606  t += *value_begin++ * ( input_vec[*index2_begin++]);
607 
608  *output_vec_begin++ = t;
609 
610  }
611  }
