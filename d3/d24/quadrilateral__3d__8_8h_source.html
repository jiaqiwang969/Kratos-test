465  virtual CoordinatesArrayType& PointLocalCoordinates( CoordinatesArrayType& rResult,
466  const CoordinatesArrayType& rPoint )
467  {
468  double tol = 1.0e-8;
469  int maxiter = 1000;
470  //check orientation of surface
471  std::vector< unsigned int> orientation(3);
472 
473  double dummy= this->GetPoint(0).X();
474  if(fabs(this->GetPoint(1).X()-dummy)<=tol&& fabs(this->GetPoint(2).X()-dummy)<=tol&& fabs(this->GetPoint(3).X()-dummy)<=tol)
475  orientation[0]=0;
476 
477  dummy= this->GetPoint(0).Y();
478  if(fabs(this->GetPoint(1).Y()-dummy)<=tol&& fabs(this->GetPoint(2).Y()-dummy)<=tol&& fabs(this->GetPoint(3).Y()-dummy)<=tol)
479  orientation[0]=1;
480 
481  dummy= this->GetPoint(0).Z();
482  if(fabs(this->GetPoint(1).Z()-dummy)<=tol&& fabs(this->GetPoint(2).Z()-dummy)<=tol&& fabs(this->GetPoint(3).Z()-dummy)<=tol)
483  orientation[0]=2;
484 
485  switch(orientation[0])
486  {
487  case 0:
488  orientation[0]= 1; orientation[1]= 2; orientation[2]= 0;
489  break;
490  case 1:
491  orientation[0]= 0; orientation[1]= 2; orientation[2]= 1;
492  break;
493  case 2:
494  orientation[0]= 0; orientation[1]= 1; orientation[2]= 2;
495  break;
496  default:
497  orientation[0]= 0; orientation[1]= 1; orientation[2]= 2;
498  }
499 
500  Matrix J = ZeroMatrix( 2, 2 );
501  Matrix invJ = ZeroMatrix( 2, 2 );
502 
503  if( rResult.size() != 2 )
504  rResult.resize( 2 );
505  //starting with xi = 0
506  rResult = ZeroVector( 2 );
507  Vector DeltaXi = ZeroVector( 2 );
508  CoordinatesArrayType CurrentGlobalCoords( ZeroVector(3) );
509  //Newton iteration:
510  for( int k=0; k<maxiter; k++ )
511  {
512  CurrentGlobalCoords = ZeroVector(3);
513  GlobalCoordinates( CurrentGlobalCoords, rResult );
514  noalias( CurrentGlobalCoords ) = rPoint - CurrentGlobalCoords;
515 
516  //Caluclate Inverse of Jacobian
517  J.resize(2,2);
518  //derivatives of shape functions
519  Matrix shape_functions_gradients;
520  shape_functions_gradients = ShapeFunctionsLocalGradients(
521  shape_functions_gradients, rResult );
522 
523  //Elements of jacobian matrix (e.g. J(1,1) = dX1/dXi1)
524  //loop over all nodes
525  for( unsigned int i=0; i<this->PointsNumber(); i++ )
526  {
527  Point<3> dummyPoint= this->GetPoint(i);
528  J(0,0) += (dummyPoint[orientation[0]])*(shape_functions_gradients(i,0));
529  J(0,1) += (dummyPoint[orientation[0]])*(shape_functions_gradients(i,1));
530  J(1,0) += (dummyPoint[orientation[1]])*(shape_functions_gradients(i,0));
531  J(1,1) += (dummyPoint[orientation[1]])*(shape_functions_gradients(i,1));
532  }
533  //deteminant of Jacobian
534  double det_j = J(0,0)*J(1,1)-J(0,1)*J(1,0);
535  //filling matrix
536  invJ(0,0) = (J(1,1))/(det_j);
537  invJ(1,0) = -(J(1,0))/(det_j);
538  invJ(0,1) = -(J(0,1))/(det_j);
539  invJ(1,1) = (J(0,0))/(det_j);
540 
541 
542  DeltaXi(0) = invJ(0,0)*CurrentGlobalCoords(orientation[0])+invJ(0,1)*CurrentGlobalCoords(orientation[1]);
543  DeltaXi(1) = invJ(1,0)*CurrentGlobalCoords(orientation[0])+invJ(1,1)*CurrentGlobalCoords(orientation[1]);
544 
545  noalias( rResult ) += DeltaXi;
546 
547  if( MathUtils<double>::Norm3(DeltaXi) > 30 )
548  {
549  break;
550  }
551  if( MathUtils<double>::Norm3(DeltaXi) < tol )
552  {
553  if(!(fabs(CurrentGlobalCoords(orientation[2]))<=tol))
554  rResult(0)=2.0;
555  break;
556  }
557  }
558  return( rResult );
559  }
