477  void CalculatePressureForces(Matrix crd,ModelPart::NodesContainerType::iterator& ctrnd, WeakPointerVector< Element >::iterator& host, Element::WeakPointer& neighbor, int flag,int pressurefac )
478  {
479  int patchsize = TDim*(TDim+1);
480  Matrix volstiffness = ZeroMatrix(patchsize,patchsize);
481  Matrix B = ZeroMatrix(TDim +1, patchsize);
482  Vector H = ZeroVector(patchsize);
483 
484  double Xc = ctrnd->X();
485  double Yc = ctrnd->Y();
486  double area = 0.0;
487 
488  area = 0.5 *( Xc*(crd(0,1)-crd(1,1))+crd(1,0)*(Yc-crd(0,1))+crd(0,0)*(crd(1,1)-Yc));
489 
490 
491  //fill B for linear element
492  B(0,0) = 0.5*(crd(0,1)-crd(1,1)); B(0,1) = 0.0;
493  B(1,0) = 0.0; B(1,1) = 0.5*(crd(1,0)-crd(0,0));
494  B(2,0) = 0.5*(crd(1,0)-crd(0,0)); B(2,1) = 0.5*(crd(0,1)-crd(1,1));
495 
496  B(0,2) = 0.5*(crd(1,1)-Yc); B(0,3) = 0.0;
497  B(1,2) = 0.0; B(1,3) = 0.5*(Xc-crd(1,0));
498  B(2,2) = 0.5*(Xc-crd(1,0)); B(2,3) = 0.5*(crd(1,1)-Yc);
499 
500  B(0,4) = 0.5*(Yc-crd(0,1)); B(0,5) = 0.0;
501  B(1,4) = 0.0; B(1,5) = 0.5*(crd(0,0)-Xc);
502  B(2,4) = 0.5*(crd(0,0)-Xc); B(2,5) = 0.5*(Yc-crd(0,1));
503 
504  //fill H
505  H[0] = B(0,0); H[1] = B(1,1); H[2] = B(0,2); H[3] = B(1,3); H[4] = B(0,4); H[5] = B(1,5);
506 
507 
508  //get properties
509  double NU = host->GetProperties()[POISSON_RATIO];
510  double EE = host->GetProperties()[YOUNG_MODULUS];
511  double bulk = EE/(3.0*(1-2*NU));
512  double factor = bulk ;
513  double accelerator = 3.0/(2.0*(1.0 + NU));
514 
515  //KRATOS_WATCH(bulk);
516  //KRATOS_WATCH(accelerator);
517 
518 
519  //build volstiffness
520  volstiffness = accelerator*factor/area*outer_prod(H, H);
521 
522 
523 
524  Matrix AV;
525  AV = outer_prod(H,H);
526  double denom =0.0;
527  denom = inner_prod(H,H);
528 
529  AV/=denom;
530  volstiffness=prod(volstiffness, AV);
531 
532  //fill displacement vector
533  Vector disp=ZeroVector(patchsize);
534  disp[0] = ctrnd->GetSolutionStepValue(DISPLACEMENT_X);
535  disp[1] = ctrnd->GetSolutionStepValue(DISPLACEMENT_Y);
536 
537  Geometry< Node<3> >& hostgeom = host->GetGeometry();
538  Geometry< Node<3> >& nghgeom = neighbor.lock()->GetGeometry();
539 
540  double meanhostX = 0.0;
541  double meanhostY = 0.0;
542  double meannghX = 0.0;
543  double meannghY = 0.0;
544 
545  for(int ii = 0; ii<TDim+1; ii++)
546  {
547 
548  meanhostX += 1.0/3*(hostgeom[ii].GetSolutionStepValue(DISPLACEMENT_X));
549  meanhostY += 1.0/3*(hostgeom[ii].GetSolutionStepValue(DISPLACEMENT_Y));
550  meannghX += 1.0/3*(nghgeom[ii].GetSolutionStepValue(DISPLACEMENT_X));
551  meannghY += 1.0/3*(nghgeom[ii].GetSolutionStepValue(DISPLACEMENT_Y));
552 
553 
554  }
555 
556  if(flag ==1 )
557  {
558  disp[2] = meanhostX;
559  disp[3] = meanhostY;
560  disp[4] = meannghX;
561  disp[5] = meannghY;
562  }
563  else
564  {
565  disp[2] = meannghX ;
566  disp[3] = meannghY;
567  disp[4] = meanhostX;
568  disp[5] = meanhostY;
569  }
570 
571 
572  //calculate pressure force & assembling
573  Vector volforce = ZeroVector(patchsize);
574  volforce = -1*prod( volstiffness, disp);
575 
576 
577  for(int ii=0; ii<TDim; ++ii)
578  ctrnd->FastGetSolutionStepValue(FORCE)[ii] += volforce[ii];
579 
580 
581 
582  for(int jj=0;jj<TDim+1; jj++)
583  {
584  int index =TDim;
585  for(int kk=0;kk<TDim; kk++)
586  if(flag == 1)
587  {
588 
589  hostgeom[jj].FastGetSolutionStepValue(FORCE)[kk] += 1.0/3* volforce[index+kk];
590 
591  nghgeom[jj].FastGetSolutionStepValue(FORCE)[kk] += 1.0/3* volforce[index+TDim+kk];
592 
593  }
594  else
595  {
596  nghgeom[jj].FastGetSolutionStepValue(FORCE)[kk] += 1.0/3* volforce[index+kk];
597 
598  hostgeom[jj].FastGetSolutionStepValue(FORCE)[kk] += 1.0/3* volforce[index+TDim+kk];
599 
600  }
601 
602 
603 
604  }
605 
606 
607  //calculate & update pressure
608  double pressure=0.0;
609  for(int ii=0; ii< patchsize; ii++)
610  if ( H[ii]*H[ii]> 10e-14)
611  pressure +=volforce[ii]/(pressurefac*H[ii]);
612 
613 
614  ctrnd->FastGetSolutionStepValue(PRESSURE) += pressure;
615 
616  }
