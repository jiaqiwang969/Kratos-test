496  virtual JacobiansType& Jacobian( JacobiansType& rResult,
497  IntegrationMethod ThisMethod) const
498  {
499  //getting derivatives of shape functions
500  ShapeFunctionsGradientsType shape_functions_gradients =
501  CalculateShapeFunctionsIntegrationPointsLocalGradients(ThisMethod);
502  //getting values of shape functions
503  Matrix shape_functions_values =
504  CalculateShapeFunctionsIntegrationPointsValues(ThisMethod);
505  //workaround by riccardo...
506  if(rResult.size() != this->IntegrationPointsNumber(ThisMethod))
507  {
508  // KLUDGE: While there is a bug in ublas
509  // vector resize, I have to put this beside resizing!!
510  JacobiansType temp(this->IntegrationPointsNumber(ThisMethod));
511  rResult.swap(temp);
512  }
513  //loop over all integration points
514  for( unsigned int pnt=0; pnt < this->IntegrationPointsNumber(ThisMethod); pnt++ )
515  {
516  //defining single jacobian matrix
517  Matrix jacobian = ZeroMatrix(3,3);
518  //loop over all nodes
519  for( unsigned int i=0; i<this->PointsNumber(); i++ )
520  {
521  jacobian(0,0) += (this->GetPoint(i).X())*(shape_functions_gradients[pnt](i,0));
522  jacobian(0,1) += (this->GetPoint(i).X())*(shape_functions_gradients[pnt](i,1));
523  jacobian(0,2) += (this->GetPoint(i).X())*(shape_functions_gradients[pnt](i,2));
524  jacobian(1,0) += (this->GetPoint(i).Y())*(shape_functions_gradients[pnt](i,0));
525  jacobian(1,1) += (this->GetPoint(i).Y())*(shape_functions_gradients[pnt](i,1));
526  jacobian(1,2) += (this->GetPoint(i).Y())*(shape_functions_gradients[pnt](i,2));
527  jacobian(2,0) += (this->GetPoint(i).Z())*(shape_functions_gradients[pnt](i,0));
528  jacobian(2,1) += (this->GetPoint(i).Z())*(shape_functions_gradients[pnt](i,1));
529  jacobian(2,2) += (this->GetPoint(i).Z())*(shape_functions_gradients[pnt](i,2));
530  }
531  rResult[pnt] = jacobian;
532  }//end of loop over all integration points
533  return rResult;
534  }
