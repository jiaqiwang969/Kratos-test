776  {

778  array_1d<double, 2> temp, proj_aux;

780  array_1d<double, 3> elemental_stabilization;

815  nu *= 0.333333333333333333333;

837  noalias(temp) = prod(trans(DN_DX),pressures);

841  for(int iii = 0; iii<2; iii++)

845  const array_1d<double,3>& proj_temp = geom[kk].FastGetSolutionStepValue(PRESS_PROJ);
846  for(int iii = 0; iii<2; iii++)
847  proj_aux[iii] += N[kk]*proj_temp[iii];
848  }
849  proj_aux *= tau;
850  noalias(elemental_stabilization) += prod(DN_DX , proj_aux);
851 
852  geom[0].FastGetSolutionStepValue(PRESSURE) += (elemental_stabilization[0] - Gaux*N[0])*(volume*density);
853  geom[1].FastGetSolutionStepValue(PRESSURE) += (elemental_stabilization[1] - Gaux*N[1])*(volume*density);
854  geom[2].FastGetSolutionStepValue(PRESSURE) += (elemental_stabilization[2] - Gaux*N[2])*(volume*density);
855 
856  double nodal_vol = volume/3.0;
857  geom[0].FastGetSolutionStepValue(NODAL_MASS) += nodal_vol * density;
858  geom[1].FastGetSolutionStepValue(NODAL_MASS) += nodal_vol * density;
859  geom[2].FastGetSolutionStepValue(NODAL_MASS) += nodal_vol * density;
860  }
861  }
862  else
863  {
864  KRATOS_ERROR(std::logic_error,"not yet implemented the 3D","");
865 
866  }
867 
868  //correct pressure
869  //double temp;
870  for(ModelPart::NodeIterator i = BaseType::GetModelPart().NodesBegin() ;
871  i != BaseType::GetModelPart().NodesEnd() ; ++i)
872  {
873  double& p = (i)->FastGetSolutionStepValue(PRESSURE);
874  const double p_old_it = (i)->FastGetSolutionStepValue(PRESSURE_OLD_IT);
875  const double rho = (i)->FastGetSolutionStepValue(DENSITY);
876  const double nu = (i)->FastGetSolutionStepValue(VISCOSITY);
877  const double mass = (i)->FastGetSolutionStepValue(NODAL_MASS);
