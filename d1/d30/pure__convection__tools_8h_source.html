181  typename TLinearSolver::Pointer linear_solver,
182  Variable<double>& rScalarVar,
183  const Variable< array_1d<double,3> >& rTransportVel,
184  const Variable< array_1d<double,3> >& rMeshVel,
185  const Variable< double >& rProjVar,
186  unsigned int time_order )

188  KRATOS_TRY
189 
190  TSparseSpace::SetToZero(mA);
191  TSparseSpace::SetToZero(mb);
192  TSparseSpace::SetToZero(mDx);
193 
194  ProcessInfo& CurrentProcessInfo = model_part.GetProcessInfo();
195  double dt = CurrentProcessInfo[DELTA_TIME];
196 
197  Vector BDFcoeffs(time_order+1);
198 
199  if(time_order == 2)
200  {
201  if(model_part.GetBufferSize() < 3)
202  KRATOS_ERROR(std::logic_error,"insufficient buffer size for BDF2","")
203 
204  BDFcoeffs[0] = 1.5 / dt; //coefficient for step n+1
205  BDFcoeffs[1] = -2.0 / dt;//coefficient for step n
206  BDFcoeffs[2] = 0.5 / dt;//coefficient for step n-1
207  }
208  else
209  {
210  BDFcoeffs[0] = 1.0 / dt; //coefficient for step n+1
211  BDFcoeffs[1] = -1.0 / dt;//coefficient for step n
212  }
213 
214  //**********************************
215  //BUILD PHASE
216  //**********************************
217  boost::numeric::ublas::bounded_matrix<double,TDim+1,TDim> DN_DX;
218  boost::numeric::ublas::bounded_matrix<double,TDim+1,TDim+1> lhs_contribution;
219  array_1d<double,TDim+1> N;
220  array_1d<unsigned int ,TDim+1> local_indices;
221  array_1d<double,TDim+1> rhs_contribution;
222  array_1d<double,TDim+1> dp_vector;
223  array_1d<double,TDim+1> MassFactors;
224  array_1d<double,TDim> vel_gauss;
225  array_1d<double,TDim> proj;
226  array_1d<double,TDim+1> u_DN;
227  array_1d<double,TDim+1> temp_vec_np;
228 
229 
230 
231  for(unsigned int i=0; i<TDim+1; i++)
232  MassFactors[i] = 1.0/double(TDim+1);
233 
234  //getting the dof position
235 // unsigned int dof_position = (model_part.NodesBegin())->GetDofPosition(rScalarVar);
236 
237  double lumping_factor = 1.0/double(TDim+1.0);
238  const unsigned int number_of_points = TDim+1;
239 
240  array_1d<double,3> vg;
241  for(ModelPart::ElementsContainerType::iterator i = model_part.ElementsBegin();
242  i!=model_part.ElementsEnd(); i++)
243  {
244  Geometry< Node<3> >& geom = i->GetGeometry();
245 
246  //calculating elemental values
247  double Volume;
248  GeometryUtils::CalculateGeometryData(geom, DN_DX, N, Volume);
249 
250 
251  //finiding local indices
252  for(int ii = 0; ii<TDim+1; ii++)
253  {
254  local_indices[ii] = geom[ii].GetDof(rScalarVar).EquationId();
255 
256 /* dp_vector[ii] = geom[ii].FastGetSolutionStepValue(rScalarVar)
257  - geom[ii].FastGetSolutionStepValue(rScalarVar,1);*/
258  }
259 
260  //calculating convective velocity
261  noalias(vel_gauss) = ZeroVector(TDim);
262  double proj = 0.0;
263  for(unsigned int i = 0; i<number_of_points; i++)
264  {
265  const array_1d<double,3>& v = geom[i].FastGetSolutionStepValue(rTransportVel);
266  const array_1d<double,3>& w = geom[i].FastGetSolutionStepValue(rMeshVel);
267  proj += geom[i].FastGetSolutionStepValue(rProjVar);
268  for(unsigned int j = 0; j<TDim; j++)
269  {
270  vel_gauss[j] += v[j] - w[j];
271  }
272  }
273  vel_gauss *= lumping_factor;
274  proj *= lumping_factor;
275 
276 
277  //CONVECTIVE CONTRIBUTION TO THE STIFFNESS MATRIX
278  noalias(u_DN) = prod(DN_DX , vel_gauss);
279  noalias(lhs_contribution) = outer_prod(N,u_DN);
280 
281  //CONVECTION STABILIZING CONTRIBUTION (Suu)
282  double tau = CalculateTAU(vel_gauss,Volume);
283  noalias(lhs_contribution) += (tau) * outer_prod(u_DN,u_DN);
284 
285  //INERTIA CONTRIBUTION
286  for(unsigned int iii = 0; iii<number_of_points; iii++)
287  lhs_contribution(iii,iii) += BDFcoeffs[0] * MassFactors[iii];
288 
289  //adding projection contribution
290  //RHS += Suy * proj[component]
291  noalias(rhs_contribution) = (tau*proj)*u_DN;
292 
293 
294  //adding the inertia terms
295  // RHS += M*vhistory
296  //calculating the historical velocity
297  for(unsigned int iii = 0; iii<number_of_points; iii++)
298  {
299  temp_vec_np[iii] = BDFcoeffs[1]*geom[iii].FastGetSolutionStepValue(rScalarVar,1);
300  }
301  for(unsigned int step = 2; step<BDFcoeffs.size(); step++)
302  {
303  for(unsigned int iii = 0; iii<number_of_points; iii++)
304  temp_vec_np[iii] += BDFcoeffs[step]*geom[iii].FastGetSolutionStepValue(rScalarVar,step);
305  }
306  for(unsigned int iii = 0; iii<number_of_points; iii++)
307  rhs_contribution[iii] -= MassFactors[iii]*temp_vec_np[iii];
308 // rhs_contribution[iii] = - MassFactors[iii]*temp_vec_np[iii];
309 
310 
311  //subtracting the dirichlet term
312  // RHS -= LHS*rScalarVars
313  for(unsigned int iii = 0; iii<number_of_points; iii++)
314  temp_vec_np[iii] = geom[iii].FastGetSolutionStepValue(rScalarVar);
315  noalias(rhs_contribution) -= prod(lhs_contribution,temp_vec_np);
316 
317  //multiplying by Volume, rho and density
318  rhs_contribution *= Volume;
319  lhs_contribution *= Volume;
320 
321  AssembleLHS(mA,lhs_contribution,local_indices);
322  AssembleRHS(mb,rhs_contribution,local_indices);
323 
324 
325 
326  }

329  //SOLVE PHASE

331  // KRATOS_WATCH(mA);
332  // KRATOS_WATCH(mb);

360  void CalculateProjection(ModelPart& model_part,
361  const Variable<double>& rScalarVar,
362  const Variable<double>& rNodalArea,
363  const Variable< array_1d<double,3> >& rTransportVel,
364  const Variable< array_1d<double,3> >& rMeshVel,
365  Variable< double >& rProjVar )
366  {
367  KRATOS_TRY
368 
369 
370  //**********************************
371  //BUILD PHASE
372  //**********************************
373  boost::numeric::ublas::bounded_matrix<double,TDim+1,TDim> DN_DX;
374  array_1d<double,TDim+1> N;
375  array_1d<double,TDim> vel_gauss;
376  array_1d<double,TDim+1> temp_vec_np;
377  array_1d<double,TDim+1> u_DN;
378 
379  double lumping_factor = 1.0/double(TDim+1.0);
380  const unsigned int number_of_points = TDim+1;
381 
382  //set to zero nodal areas
383  for(ModelPart::NodesContainerType::iterator i = model_part.NodesBegin();
384  i!=model_part.NodesEnd(); i++)
385  {
386  i->FastGetSolutionStepValue(rNodalArea) = 0.0;
387  i->FastGetSolutionStepValue(rProjVar) = 0.0;
388  }
389 
390  //build up nodal areas and projections
391  array_1d<double,3> vg;
392  for(ModelPart::ElementsContainerType::iterator i = model_part.ElementsBegin();
393  i!=model_part.ElementsEnd(); i++)
394  {
395 
396  Geometry< Node<3> >& geom = i->GetGeometry();
397 
398  //calculating elemental values
399  double Volume;
400  GeometryUtils::CalculateGeometryData(geom, DN_DX, N, Volume);
401 
402  //calculating convective velocity
403  double nodal_area = Volume*lumping_factor;
404  noalias(vel_gauss) = ZeroVector(TDim);
405  for(unsigned int i = 0; i<number_of_points; i++)
406  {
407  const array_1d<double,3>& v = geom[i].FastGetSolutionStepValue(rTransportVel);
408  const array_1d<double,3>& w = geom[i].FastGetSolutionStepValue(rMeshVel);
409 
410  //adding up nodal area as needed
411  geom[i].FastGetSolutionStepValue(rNodalArea) += nodal_area;
412 
413  for(unsigned int j = 0; j<TDim; j++)
414  {
415  vel_gauss[j] += v[j] - w[j];
416  }
417  }
418  vel_gauss *= lumping_factor;
419 
420  //filling up a vector with all of the nodal values
421  for(unsigned int iii = 0; iii<number_of_points; iii++)
422  temp_vec_np[iii] = geom[iii].FastGetSolutionStepValue(rScalarVar);
423 
424  //
425  noalias(u_DN) = prod(DN_DX , vel_gauss);
426  double conv_proj = inner_prod( u_DN , temp_vec_np);
427  conv_proj *= Volume * lumping_factor;
428 
429  for(unsigned int iii = 0; iii<number_of_points; iii++)
430  geom[iii].FastGetSolutionStepValue(rProjVar) += conv_proj;
431  }
432 
433  //dividing by the overall mass matrix
434  for(ModelPart::NodesContainerType::iterator i = model_part.NodesBegin();
435  i!=model_part.NodesEnd(); i++)
436  {
437  i->FastGetSolutionStepValue(rProjVar) /= i->FastGetSolutionStepValue(rNodalArea);
438  }
439 
440  KRATOS_CATCH("")
441  }

516  double CalculateTAU(const array_1d<double,2>& vel, const double& Volume)
517  {
518  //calculating parameter tau
519 // double c1 = 4.00;
520  double c2 = 2.00;
521  double h = sqrt(2.00*Volume);
522  double norm_u =norm_2(vel);
523  if(norm_u > 1e-15)
524  return h / ( c2*norm_u );
525  return 0.0;
526  }

528  double CalculateTAU(const array_1d<double,3>& vel, const double& Volume)
529  {
530  //calculating parameter tau
531 // double c1 = 4.00;
532  double c2 = 2.00;
533  double h = pow(6.00*Volume,0.3333333);
534  double norm_u =norm_2(vel);
535  if(norm_u > 1e-15)
536  return h / ( c2*norm_u );
537  return 0.0;
