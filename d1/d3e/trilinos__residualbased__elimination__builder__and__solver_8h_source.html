243  //getting the array of the conditions

245 
246  //resetting to zero the vector of reactions
247  TSparseSpace::SetToZero(*BaseType::mpReactionsVector);
248 
249  //contributions to the system

262  {
263  //calculate elemental contribution
264  pScheme->CalculateSystemContributions(*it,LHS_Contribution,RHS_Contribution,EquationId,CurrentProcessInfo);
