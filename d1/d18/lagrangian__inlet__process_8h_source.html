144  virtual void Execute()
145  {

147 
148  double time = mr_model_part.GetProcessInfo()[TIME];

150  if(minsertion_time == 0.00)

152  minsertion_time_step = EstimateInsertionTime();

154  }
155 
156  if(time >= minsertion_time)
157  {
158 
159  //detecting number of new nodes
160  int new_nodes_number = 0;
161  for(ModelPart::NodesContainerType::iterator in = mr_model_part.NodesBegin(); in!=mr_model_part.NodesEnd(); in++)
162  {
163  if(in->FastGetSolutionStepValue(IS_LAGRANGIAN_INLET) == 1)
164  new_nodes_number += 1;
165  }
166 
167  if(new_nodes_number != 0)
168  {
169  //allocating the memory needed
170  int old_size = mr_model_part.Nodes().size();
171  (mr_model_part.Nodes()).reserve(old_size + new_nodes_number);
172 
173  for(int i = 0; i<old_size; i++)
174  {
175  ModelPart::NodesContainerType::iterator in = mr_model_part.NodesBegin() + i;
176 
177  if(in->FastGetSolutionStepValue(IS_LAGRANGIAN_INLET) == 1)
178  {
179  //first of all set the node as normal fluid (not structure)
180  in->FastGetSolutionStepValue(IS_STRUCTURE) = 0;
181  in->FastGetSolutionStepValue(IS_BOUNDARY) = 0;
182  in->FastGetSolutionStepValue(IS_LAGRANGIAN_INLET) = 0;
183  in->Free(VELOCITY_X);
184  in->Free(VELOCITY_Y);
185  in->Free(VELOCITY_Z);
186  in->Free(FRACT_VEL_X);
187  in->Free(FRACT_VEL_Y);
188  in->Free(FRACT_VEL_Z);
189 
190  //make a copy of the
191  Node<3>::Pointer new_node = AdaptivityUtils::AddNode(in->X(),in->Y(),in->Z(),mr_model_part);
192  array_1d<double,3>& vnew = new_node->FastGetSolutionStepValue(VELOCITY);
193  const array_1d<double,3>& v = in->FastGetSolutionStepValue(VELOCITY);
194  noalias(vnew) = v;
195  new_node->FastGetSolutionStepValue(VELOCITY,1) = in->FastGetSolutionStepValue(VELOCITY,1);
196  new_node->FastGetSolutionStepValue(PRESSURE) = in->FastGetSolutionStepValue(PRESSURE);
197  new_node->FastGetSolutionStepValue(CONV_PROJ) = in->FastGetSolutionStepValue(CONV_PROJ);
198  new_node->FastGetSolutionStepValue(PRESS_PROJ) = in->FastGetSolutionStepValue(PRESS_PROJ);
199  new_node->Fix(VELOCITY_X);
200  new_node->Fix(VELOCITY_Y);
201  new_node->Fix(VELOCITY_Z);
202  new_node->Fix(FRACT_VEL_X);
203  new_node->Fix(FRACT_VEL_Y);
204  new_node->Fix(FRACT_VEL_Z);
205  new_node->pAddDof(PRESSURE);
206 
207  new_node->FastGetSolutionStepValue(IS_BOUNDARY) = 1;
208  new_node->FastGetSolutionStepValue(IS_STRUCTURE) = 1;
209  new_node->FastGetSolutionStepValue(IS_LAGRANGIAN_INLET) = 1;
210  double nodal_h = in->FastGetSolutionStepValue(NODAL_H);
211  new_node->FastGetSolutionStepValue(NODAL_H) = nodal_h;
212  }

216

220  }

320  double EstimateInsertionTime()
321  {
322  KRATOS_TRY
323 
324  double dt_estimate = 100.0;
325 
326  for(ModelPart::NodesContainerType::iterator in = mr_model_part.NodesBegin(); in!=mr_model_part.NodesEnd(); in++)
327  {
328  if(in->FastGetSolutionStepValue(IS_LAGRANGIAN_INLET) == 1)
329  {
330  const array_1d<double,3>& v = in->FastGetSolutionStepValue(VELOCITY);
331  double nodal_h = in->FastGetSolutionStepValue(NODAL_H);
332 
333  //estimating the next time step
334  double normv = norm_2(v);
335  double dtcandidate = nodal_h / normv;
336  if( dtcandidate < dt_estimate)
337  dt_estimate = dtcandidate;
338  }
339  }
340 
341  return dt_estimate;
342 
343  KRATOS_CATCH("");
344  }
