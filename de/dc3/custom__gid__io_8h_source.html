139  virtual void PrintResults( Variable<Vector> rVariable, ModelPart& r_model_part,
140  double SolutionTag, unsigned int parameter_index )
141  {
142  if( mMeshElements.size() != 0 || mMeshConditions.size() != 0 )
143  {
144  WriteGaussPoints();
145 
146  if( rVariable == INSITU_STRESS )
147  GiD_BeginResult( (char *)(rVariable.Name()).c_str(), "Kratos", SolutionTag,
148  GiD_Matrix, GiD_OnGaussPoints, mGPTitle, NULL, 0, NULL );
149  else if( (rVariable == MATERIAL_PARAMETERS) || (rVariable == INTERNAL_VARIABLES) )
150  {
151  std::stringstream param_index;
152  param_index << parameter_index;
153  GiD_BeginResult( (char *)(rVariable.Name() + param_index.str() ).c_str(),
154  "Kratos", SolutionTag, GiD_Scalar, GiD_OnGaussPoints,
155  mGPTitle, NULL, 0, NULL );
156  }
157  else
158  GiD_BeginResult( (char *)(rVariable.Name()).c_str(), "Kratos", SolutionTag,
159  GiD_Vector, GiD_OnGaussPoints, mGPTitle, NULL, 0, NULL );
160  std::vector<Vector> ValuesOnIntPoint(mSize);
161  if( mMeshElements.size() != 0 )
162  {
163  for( ModelPart::ElementsContainerType::iterator it = mMeshElements.begin();
164  it != mMeshElements.end(); ++it )
165  {
166  it->GetValueOnIntegrationPoints( rVariable, ValuesOnIntPoint,
167  r_model_part.GetProcessInfo() );
168  for(unsigned int i=0; i<mIndexContainer.size(); i++)
169  {
170  int index = mIndexContainer[i];
171  if( rVariable == INSITU_STRESS )
172  {
173  if(ValuesOnIntPoint[i].size() ==6 )
174  GiD_Write3DMatrix( it->Id(), ValuesOnIntPoint[index](0),
175  ValuesOnIntPoint[index](1), ValuesOnIntPoint[index](2),
176  ValuesOnIntPoint[index](3), ValuesOnIntPoint[index](4),
177  ValuesOnIntPoint[index](5) );
178  }
179  else if( (rVariable == MATERIAL_PARAMETERS)
180  || (rVariable == INTERNAL_VARIABLES) )
181  {
182  double value = 0.0;
183  if( ValuesOnIntPoint[index].size() > parameter_index )
184  value = ValuesOnIntPoint[index][parameter_index];
185  GiD_WriteScalar( it->Id(), value );
186  }
187  else
188  if( ValuesOnIntPoint[0].size() == 3 )
189  GiD_WriteVector( it->Id(), ValuesOnIntPoint[index][0],
190  ValuesOnIntPoint[index][1], ValuesOnIntPoint[index][2] );
191  }
192  }
193  }
194  if( mMeshConditions.size() != 0 )
195  {
196  for( ModelPart::ConditionsContainerType::iterator it = mMeshConditions.begin();
