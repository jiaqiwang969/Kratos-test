277  template<class TDataType>
278  void Add(Variable<TDataType> const& ThisVariable)
279  {
280  if(ThisVariable.Key()== 0)
281  KRATOS_ERROR(std::logic_error,
282  "Adding uninitialize variable to this variable list. Check if all variables are registered before kernel initialization","");
283 
284  if(Has(ThisVariable))
285  return;
286 
287  if(mPositions.size() <= ThisVariable.Key())
288  mPositions.resize(ThisVariable.Key()+1/*KratosComponents<VariableData>::Size()*/, static_cast<IndexType>(-1));
289 
290  mPositions[ThisVariable.Key()] = mDataSize;
291  mVariables.push_back(&ThisVariable);
292  const SizeType block_size = sizeof(BlockType);
293  mDataSize += static_cast<SizeType>(((block_size - 1) + sizeof(TDataType)) / block_size);
294  }
