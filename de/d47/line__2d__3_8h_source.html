400  virtual JacobiansType& Jacobian(JacobiansType& rResult, IntegrationMethod ThisMethod) const
401  {
402  //getting derivatives of shape functions
403  ShapeFunctionsGradientsType shape_functions_gradients =
404  CalculateShapeFunctionsIntegrationPointsLocalGradients(ThisMethod);
405  //getting values of shape functions
406  Matrix shape_functions_values =
407  CalculateShapeFunctionsIntegrationPointsValues(ThisMethod);
408  if(rResult.size() != this->IntegrationPointsNumber(ThisMethod))
409  {
410  JacobiansType temp(this->IntegrationPointsNumber(ThisMethod));
411  rResult.swap(temp);
412  }
413  //loop over all integration points
414  for(unsigned int pnt=0; pnt < this->IntegrationPointsNumber(ThisMethod); pnt++ )
415  {
416  //defining single jacobian matrix
417  Matrix jacobian = ZeroMatrix(2,1);
418  //loop over all nodes
419  for(unsigned int i=0; i<this->PointsNumber(); i++ )
420  {
421  jacobian(0,0) += (this->GetPoint(i).X())*(shape_functions_gradients[pnt](i,0));
422  jacobian(1,0) += (this->GetPoint(i).Y())*(shape_functions_gradients[pnt](i,0));
423  }
424  rResult[pnt] = jacobian;
425  }//end of loop over all integration points
426  return rResult;
427  }

841  static Matrix CalculateShapeFunctionsIntegrationPointsValues(typename BaseType::IntegrationMethod ThisMethod)
842  {
843  const IntegrationPointsContainerType& all_integration_points = AllIntegrationPoints();
844  const IntegrationPointsArrayType& IntegrationPoints = all_integration_points[ThisMethod];
845  int integration_points_number = IntegrationPoints.size();
846  Matrix N(integration_points_number,3);
847  for(int it_gp = 0; it_gp < integration_points_number; it_gp++)
848  {
849  double e = IntegrationPoints[it_gp].X();
850  N(it_gp,0) = 0.5*(e-1)*e;
851  N(it_gp,1) = 1.0-e*e;
852  N(it_gp,2) = 0.5*(1+e)*e;
853  }
854  return N;
855  }
