927  virtual double ShapeFunctionValue( IndexType ShapeFunctionIndex,
928  const CoordinatesArrayType& rPoint) const
929  {
930  double fourthCoord= 1.0-(rPoint[0]+rPoint[1]+rPoint[2]);
931 
932  switch( ShapeFunctionIndex )
933  {
934  case 0:
935  return( fourthCoord*(2*fourthCoord-1) );
936  case 1:
937  return( rPoint[0]*(2*rPoint[0]-1));
938  case 2:
939  return( rPoint[1]*(2*rPoint[1]-1) );
940  case 3:
941  return( rPoint[2]*(2*rPoint[2]-1));
942  case 4:
943  return( 4*fourthCoord*rPoint[0]);
944  case 5:
945  return( 4*rPoint[0]*rPoint[1]);
946  case 6:
947  return( 4*fourthCoord*rPoint[1]);
948  case 7:
949  return( 4*fourthCoord*rPoint[2]);
950  case 8:
951  return( 4*rPoint[0]*rPoint[2]);
952  case 9:
953  return( 4*rPoint[1]*rPoint[2]);
954  default:
955  KRATOS_ERROR(std::logic_error,
956  "Wrong index of shape function!" , *this);
957  }
