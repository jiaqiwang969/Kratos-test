106  int solve (int sys, int n,

109  void *Numeric, double const* Control, double* Info)
110  {

112  traits::detail::array<double> Axr (nnz);
113  if (!Axr.valid()) return UMFPACK_ERROR_out_of_memory;
114  traits::detail::array<double> Axi (nnz);
115  if (!Axi.valid()) return UMFPACK_ERROR_out_of_memory;
116  traits::detail::disentangle (Ax, Ax+nnz,
117  Axr.storage(), Axi.storage());
118  traits::detail::array<double> Br (n);
119  if (!Br.valid()) return UMFPACK_ERROR_out_of_memory;
120  traits::detail::array<double> Bi (n);
121  if (!Bi.valid()) return UMFPACK_ERROR_out_of_memory;
122  traits::detail::disentangle (B, B+n,
123  Br.storage(), Bi.storage());
124  traits::detail::array<double> Xr (n);
125  if (!Xr.valid()) return UMFPACK_ERROR_out_of_memory;
126  traits::detail::array<double> Xi (n);
127  if (!Xi.valid()) return UMFPACK_ERROR_out_of_memory;
128 
129  int status = umfpack_zi_solve (sys, Ap, Ai,
130  Axr.storage(), Axi.storage(),
131  Xr.storage(), Xi.storage(),
132  Br.storage(), Bi.storage(),
133  Numeric, Control, Info);
134  if (status != UMFPACK_OK) return status;
135  traits::detail::interlace (Xr.storage(), Xr.storage() + n,
136  Xi.storage(), X);
137  return status;

228  inline
229  int triplet_to_col (int n_row, int n_col, int nz,
230  int const* Ti, int const* Tj,
231  traits::complex_d const* Tx,
232  int* Ap, int* Ai, traits::complex_d* Ax,
233  int* Map)
234  {
235  assert (Tx == 0 && Ax == 0 || Tx != 0 && Ax != 0);
236  double *Txr = 0, *Txi = 0;
237  if (Tx != 0) {
238  traits::detail::array<double> ATxr (nz);
239  if (!ATxr.valid()) return UMFPACK_ERROR_out_of_memory;
240  traits::detail::array<double> ATxi (nz);
241  if (!ATxi.valid()) return UMFPACK_ERROR_out_of_memory;
242  Txr = ATxr.storage();
243  Txi = ATxi.storage();
244  traits::detail::disentangle (Tx, Tx+nz, Txr, Txi);
245  }
246  double *Axr = 0, *Axi = 0;
247  if (Ax != 0) {
248  traits::detail::array<double> AAxr (nz);
249  if (!AAxr.valid()) return UMFPACK_ERROR_out_of_memory;
250  traits::detail::array<double> AAxi (nz);
251  if (!AAxi.valid()) return UMFPACK_ERROR_out_of_memory;
252  Axr = AAxr.storage();
253  Axi = AAxi.storage();
254  }
255  int status;
256  status = umfpack_zi_triplet_to_col (n_row, n_col, nz,
257  Ti, Tj, Txr, Txi,
258  Ap, Ai, Axr, Axi, Map);
259  if (Ax != 0) {
260  if (status != UMFPACK_OK) return status;
261  traits::detail::interlace (Axr, Axr + nz, Axi, Ax);
262  }
263  return status;
264  }
