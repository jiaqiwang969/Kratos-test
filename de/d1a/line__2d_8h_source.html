395  virtual JacobiansType& Jacobian(JacobiansType& rResult, IntegrationMethod ThisMethod) const
396  {
397  Matrix jacobian(2,1);
398  jacobian(0,0)=(BaseType::GetPoint(1).X()-BaseType::GetPoint(0).X())*0.5; //on the Gauss points (J is constant at each element)
399  jacobian(1,0)=(BaseType::GetPoint(1).Y()-BaseType::GetPoint(0).Y())*0.5;
400 
401  if(rResult.size() != BaseType::IntegrationPointsNumber(ThisMethod))
402  {
403  // KLUDGE: While there is a bug in ublas vector resize, I have to put this beside resizing!!
404  JacobiansType temp(BaseType::IntegrationPointsNumber(ThisMethod));
405  rResult.swap(temp);
406  }
407  std::fill(rResult.begin(), rResult.end(), jacobian);

699  static Matrix CalculateShapeFunctionsIntegrationPointsValues(typename BaseType::IntegrationMethod ThisMethod)
700  {
701  const IntegrationPointsContainerType& all_integration_points = AllIntegrationPoints();
702  const IntegrationPointsArrayType& IntegrationPoints = all_integration_points[ThisMethod];
703  int integration_points_number = IntegrationPoints.size();
704  Matrix N(integration_points_number,2);
705  for(int it_gp = 0; it_gp < integration_points_number; it_gp++)
706  {
707  double e = IntegrationPoints[it_gp].X();
708  N(it_gp,0) = 0.5*(1-e);
709  N(it_gp,1) = 0.5*(1+e);
710  }
711 
712  return N;
713  }
