137  //building up the matrix graph row by row

140  {
141  unsigned int index_i = it->GetDof(DISPLACEMENT_X).EquationId();
142  unsigned int index_j = it->GetDof(DISPLACEMENT_Y).EquationId();
143  unsigned int index_k = it->GetDof(DISPLACEMENT_Z).EquationId();

145  //if(index_i < mEquationSystemSize && (it->GetValue(NEIGHBOUR_NODES)).size() != 0)

158  int index_s = i->GetDof(DISPLACEMENT_Z).EquationId();
159  work_array.push_back(index_l);
160  work_array.push_back(index_r);
161  work_array.push_back(index_s);
162  }
163 
164  //sorting the indices and elminating the duplicates
165  std::sort(work_array.begin(),work_array.end());
166  unsigned int number_of_entries = work_array.size();
167 
168  //filling up the matrix
169  for(unsigned int j=0; j<number_of_entries; j++)
170  {
171  mA.push_back(index_i,work_array[j] , 0.00);
172  }
173  for(unsigned int j=0; j<number_of_entries; j++)
174  {
175  mA.push_back(index_j,work_array[j] , 0.00);
176  }
177  for(unsigned int j=0; j<number_of_entries; j++)
178  {
179  mA.push_back(index_k,work_array[j] , 0.00);
180  }
181  //clearing the array for the next step
182  work_array.erase(work_array.begin(),work_array.end());
183  total_size += number_of_entries;
184  }

192  mb.resize(mA.size1(),false);

333  TSystemMatrixType& A,
334  const boost::numeric::ublas::bounded_matrix<double,(TDim+1)*TDim,(TDim+1)*TDim> & LHS_Contribution,
335  const array_1d<unsigned int,(TDim+1)*TDim>& EquationId
336  )

339 
340  for (unsigned int i_local=0; i_local<local_size; i_local++)
341  {
342  unsigned int i_global=EquationId[i_local];
343  if ( i_global < mEquationSystemSize )
344  {
345  for (unsigned int j_local=0; j_local<local_size; j_local++)
346  {
347  unsigned int j_global=EquationId[j_local];
348  if ( j_global < mEquationSystemSize )
349  A(i_global,j_global) += LHS_Contribution(i_local,j_local);

378  void BallVertex3D( const Geometry< Node<3> >& geom, boost::numeric::ublas::bounded_matrix<double,(TDim+1)*TDim,(TDim+1)*TDim>& K_matrix ){
379 
380  array_1d<double,4> x,y,z;
381 
382  x[0] = geom[0].X();
383  y[0] = geom[0].Y();
384  z[0] = geom[0].Z();
385  x[1] = geom[1].X();
386  y[1] = geom[1].Y();
387  z[1] = geom[1].Z();
388  x[2] = geom[2].X();
389  y[2] = geom[2].Y();
390  z[2] = geom[2].Z();
391  x[3] = geom[3].X();
392  y[3] = geom[3].Y();
393  z[3] = geom[3].Z();
394 
395  noalias(K_matrix) = ZeroMatrix(12,12);
396 
397  boost::numeric::ublas::bounded_matrix<int,6,2> index;
398  index(0,0)=0;
399  index(0,1)=1;
400  index(1,0)=0;
401  index(1,1)=2;
402  index(2,0)=0;
403  index(2,1)=3;
404  index(3,0)=1;
405  index(3,1)=2;
406  index(4,0)=1;
407  index(4,1)=3;
408  index(5,0)=2;
409  index(5,1)=3;
410 
411  boost::numeric::ublas::bounded_matrix<double,3,3> vij_mat;
412  double invlijq;
413  double invlij;
414  int i;
415  int j;
416  for (unsigned int k=0; k<6; k++)
417  {
418  i=index(k,0);
419  j=index(k,1);
420 
421  // edge ij e ji
422  invlijq=1.0/(pow((x[j]-x[i]),2)+pow((y[j]-y[i]),2)+pow((z[j]-z[i]),2));
423  invlij=1.0/sqrt(pow((x[j]-x[i]),2)+pow((y[j]-y[i]),2)+pow((z[j]-z[i]),2));
424 
425  vij_mat(0,0)=pow((x[j]-x[i]),2);
426  vij_mat(0,1)=((x[j]-x[i])*(y[j]-y[i]));
427  vij_mat(0,2)=((x[j]-x[i])*(z[j]-z[i]));
428  vij_mat(1,0)=((x[j]-x[i])*(y[j]-y[i]));
429  vij_mat(1,1)=pow((y[j]-y[i]),2);
430  vij_mat(1,2)=((x[j]-x[i])*(z[j]-z[i]));
431  vij_mat(2,0)=((x[j]-x[i])*(z[j]-z[i]));
432  vij_mat(2,1)=((y[j]-y[i])*(z[j]-z[i]));
433  vij_mat(2,2)=pow((z[j]-z[i]),2);
434 
435  vij_mat=vij_mat*invlijq;
436 
437  K_matrix(3*i,3*i)+=invlij*vij_mat(0,0);
438  K_matrix(3*i,3*i+1)+=invlij*vij_mat(0,1);
439  K_matrix(3*i,3*i+2)+=invlij*vij_mat(0,2);
440  K_matrix(3*i+1,3*i)+=invlij*vij_mat(1,0);
441  K_matrix(3*i+1,3*i+1)+=invlij*vij_mat(1,1);
442  K_matrix(3*i+1,3*i+2)+=invlij*vij_mat(1,2);
443  K_matrix(3*i+2,3*i)+=invlij*vij_mat(2,0);
444  K_matrix(3*i+2,3*i+1)+=invlij*vij_mat(2,1);
445  K_matrix(3*i+2,3*i+2)+=invlij*vij_mat(2,2);
446 
447  K_matrix(3*i,3*j)+=-invlij*vij_mat(0,0);
448  K_matrix(3*i,3*j+1)+=-invlij*vij_mat(0,1);
449  K_matrix(3*i,3*j+2)+=-invlij*vij_mat(0,2);
450  K_matrix(3*i+1,3*j)+=-invlij*vij_mat(1,0);
451  K_matrix(3*i+1,3*j+1)+=-invlij*vij_mat(1,1);
452  K_matrix(3*i+1,3*j+2)+=-invlij*vij_mat(1,2);
453  K_matrix(3*i+2,3*j)+=-invlij*vij_mat(2,0);
454  K_matrix(3*i+2,3*j+1)+=-invlij*vij_mat(2,1);
455  K_matrix(3*i+2,3*j+2)+=-invlij*vij_mat(2,2);
456 
457  K_matrix(3*j,3*i)+=-invlij*vij_mat(0,0);
458  K_matrix(3*j,3*i+1)+=-invlij*vij_mat(0,1);
459  K_matrix(3*j,3*i+2)+=-invlij*vij_mat(0,2);
460  K_matrix(3*j+1,3*i)+=-invlij*vij_mat(1,0);
461  K_matrix(3*j+1,3*i+1)+=-invlij*vij_mat(1,1);
462  K_matrix(3*j+1,3*i+2)+=-invlij*vij_mat(1,2);
463  K_matrix(3*j+2,3*i)+=-invlij*vij_mat(2,0);
464  K_matrix(3*j+2,3*i+1)+=-invlij*vij_mat(2,1);
465  K_matrix(3*j+2,3*i+2)+=-invlij*vij_mat(2,2);
466 
467  K_matrix(3*j,3*j)+=invlij*vij_mat(0,0);
468  K_matrix(3*j,3*j+1)+=invlij*vij_mat(0,1);
469  K_matrix(3*j,3*j+2)+=invlij*vij_mat(0,2);
470  K_matrix(3*j+1,3*j)+=invlij*vij_mat(1,0);
471  K_matrix(3*j+1,3*j+1)+=invlij*vij_mat(1,1);
472  K_matrix(3*j+1,3*j+2)+=invlij*vij_mat(1,2);
473  K_matrix(3*j+2,3*j)+=invlij*vij_mat(2,0);
474  K_matrix(3*j+2,3*j+1)+=invlij*vij_mat(2,1);
475  K_matrix(3*j+2,3*j+2)+=invlij*vij_mat(2,2);
476 
477  }

480  // edge 1p p1
481  array_1d<double,3> v24;
482  array_1d<double,3> v23;
483  array_1d<double,3> n1p;
484  double xp,yp,zp;
485  double area=0.0;
486  triangle_area(x[1],x[2],x[3],y[1],y[2],y[3],z[1],z[2],z[3],area);
487 
488  v24[0]=x[3]-x[1];
489  v24[1]=y[3]-y[1];
490  v24[2]=z[3]-z[1];
491  v24=v24/sqrt(pow(v24[0],2)+pow(v24[1],2)+pow(v24[2],2));
492  v23[0]=x[2]-x[1];
493  v23[1]=y[2]-y[1];
494  v23[2]=z[2]-z[1];
495  v23=v23/sqrt(pow(v23[0],2)+pow(v23[1],2)+pow(v23[2],2));
496

498  n1p[1]=v24[2]*v23[0]-v24[0]*v23[3];
499  n1p[2]=v24[0]*v23[1]-v24[1]*v23[0];
500  n1p=n1p/sqrt(pow(n1p[0],2)+pow(n1p[1],2)+pow(n1p[2],2));
501 
502  boost::numeric::ublas::bounded_matrix<double,3,3> n1p_mat;
503 
504  n1p_mat(0,0)=pow(n1p[0],2);
505  n1p_mat(0,1)=(n1p[0]*n1p[1]);
506  n1p_mat(0,2)=(n1p[0]*n1p[2]);
507  n1p_mat(1,0)=(n1p[1]*n1p[0]);
508  n1p_mat(1,1)=pow(n1p[1],2);
509  n1p_mat(1,2)=(n1p[1]*n1p[2]);
510  n1p_mat(2,0)=(n1p[2]*n1p[0]);
511  n1p_mat(2,1)=(n1p[2]*n1p[1]);
512  n1p_mat(2,2)=pow(n1p[2],2);
513 
514  xp=x[0]-(n1p_mat(0,0)*(x[0]-x[1])+n1p_mat(0,1)*(y[0]-y[1])+n1p_mat(0,2)*(z[0]-z[1]));
515  yp=y[0]-(n1p_mat(1,0)*(x[0]-x[1])+n1p_mat(1,1)*(y[0]-y[1])+n1p_mat(1,2)*(z[0]-z[1]));
516  zp=z[0]-(n1p_mat(2,0)*(x[0]-x[1])+n1p_mat(2,1)*(y[0]-y[1])+n1p_mat(2,2)*(z[0]-z[1]));
517 
518  boost::numeric::ublas::bounded_matrix<double,3,3> v1p_mat;
519  v1p_mat(0,0)=pow((xp-x[0]),2);
520  v1p_mat(0,1)=((xp-x[0])*(yp-y[0]));
521  v1p_mat(0,2)=((xp-x[0])*(zp-z[0]));
522  v1p_mat(1,0)=((xp-x[0])*(yp-y[0]));
523  v1p_mat(1,1)=pow((yp-y[0]),2);
524  v1p_mat(1,2)=((yp-y[0])*(zp-z[0]));
525  v1p_mat(2,0)=((xp-x[0])*(zp-z[0]));
526  v1p_mat(2,1)=((yp-y[0])*(zp-z[0]));
527  v1p_mat(2,2)=pow((zp-z[0]),2);
