225  static IndexType AssignDofsEquationIds(DofsArrayType & rDofs)
226  {
227  SizeType free_id = 0;
228  SizeType fix_id = rDofs.size();
229 
230  for (typename DofsArrayType::iterator dof_iterator = rDofs.begin(); dof_iterator != rDofs.end(); ++dof_iterator)
231  if (dof_iterator->IsFixed())
232  dof_iterator->SetEquationId(--fix_id);
233  else
234  dof_iterator->SetEquationId(free_id++);
235 
236  return fix_id;

273  static void AssembleMatrix(SystemMatrixType& rSystemMatrix, LocalMatrixType const& LocalMatrix, DofsArrayType const& Dofs)
274  {
275  KRATOS_TRY_LEVEL_4
276 
277  SizeType global_size = TSystemSpace::Size1(rSystemMatrix);
278 
279  SizeType local_size = TLocalSpace::Size1(LocalMatrix);
280 
281  for(SizeType i_local=0; i_local<local_size; i_local++)
282  {
283  SizeType i_global= Dofs[i_local].EquationId();
284  if ( i_global < global_size)
285  {
286  for (SizeType j_local=0; j_local<local_size; j_local++)
287  {
288  SizeType j_global= Dofs[j_local].EquationId();
289  if ( j_global < global_size )
290  rSystemMatrix(i_global,j_global) += LocalMatrix(i_local, j_local);
291  }
292  }
293  }
294  KRATOS_CATCH_LEVEL_4("")
295  }

548  void ApplyDirichletConditions()
549  {
550  KRATOS_TRY_LEVEL_4
551  if(mDirichletSize == 0)
552  return;
553  SystemVectorType xd(mDirichletSize);
554  SystemVectorType dirichlet_contribution(mEquationSystemSize);
555 
556  //std::cout << "mDirichletSize : " << mDirichletSize << std::endl;
557 
558  for(typename DofsArrayType::iterator i = mDofs.begin() ; i != mDofs.end() ; ++i)
559  if(i->IsFixed())
560  xd[i->EquationId() - mEquationSystemSize] = (*i)();
561 
562  //chapuza!! should be done in just one function like multiply and add!!
563  TSystemSpace::Mult(mDirichletMatrix, xd,dirichlet_contribution );
564 
565  mRightHandSide -= dirichlet_contribution;
566 
567  KRATOS_CATCH_LEVEL_4("")
568  }

570  void Resize(SizeType NewSize)
