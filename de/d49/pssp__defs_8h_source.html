50 /*
51  * *************************************************
52  * Global data structures used in LU factorization
53  * *************************************************
54  *
55  * nsuper: number of supernodes = nsuper+1, numbered between 0 and nsuper.
56  *
57  * (supno, xsup, xsup_end):
58  * supno[i] is the supernode number to which column i belongs;
59  * xsup[s] points to the first column of supernode s;
60  * xsup_end[s] points to one past the last column of supernode s.
61  * Example: supno 0 1 2 2 3 3 3 4 4 4 4 4 (n=12)
62  * xsup 0 1 2 4 7
63  * xsup_end 1 2 4 7 12
64  * Note: dfs will be performed on supernode rep. relative to the new
65  * row pivoting ordering
66  *
67  * (lsub, xlsub, xlsub_end):
68  * lsub[*] contains the compressed subscripts of the supernodes;
69  * xlsub[j] points to the starting location of the j-th column in
70  * lsub[*];
71  * xlsub_end[j] points to one past the ending location of the j-th
72  * column in lsub[*].
73  * Storage: original row subscripts in A.
74  *
75  * During the course of sparse LU factorization, we also use
76  * (lsub, xlsub, xlsub_end, xprune) to represent symmetrically
77  * pruned graph. Contention will occur when one processor is
78  * performing DFS on supernode S, while another processor is pruning
79  * supernode S. We use the following data structure to deal with
80  * this problem. Suppose each supernode contains columns {s,s+1,...,t},
81  * with first column s and last column t.
82  *
83  * (1) if t > s, only the subscript sets for column s and column t
84  * are stored. Column t represents pruned adjacency structure.
85  *
86  * --------------------------------------------
87  * lsub[*] ... | col s | col t | ...
88  * --------------------------------------------
89  * ^ ^ ^
90  * xlsub[s] xlsub_end[s] xlsub_end[s+1]
91  * xlsub[s+1] :
92  * : :
93  * : xlsub_end[t]
94  * xlsub[t] xprune[t]
95  * xprune[s]
96  *
97  * (2) if t == s, i.e., a singleton supernode, the subscript set
98  * is stored twice:
99  *
100  * --------------------------------------
101  * lsub[*] ... | s | s | ...
102  * --------------------------------------
103  * ^ ^ ^
104  * xlsub[s] xlsub_end[s] xprune[s]
105  *
106  * There are two subscript sets for each supernode, the last column
107  * structures (for pruning) will be removed after the numerical LU
108  * factorization phase:
109  * o lsub[j], j = xlsub[s], ..., xlsub_end[s]-1
110  * is the structure of column s (i.e. structure of this supernode).
111  * It is used for the storage of numerical values.
112  * o lsub[j], j = xlsub[t], ..., xlsub_end[t]-1
113  * is the structure of the last column t of this supernode.
114  * It is for the purpose of symmetric pruning. Therefore, the
115  * structural subscripts can be rearranged without making physical
116  * interchanges among the numerical values.
117  *
118  * DFS will traverse the first subscript set if the supernode
119  * has not been pruned; otherwise it will traverse the second
120  * subscript list, i.e., the part of the pruned graph.
121  *
122  * (lusup, xlusup, xlusup_end):
123  * lusup[*] contains the numerical values of the supernodes;
124  * xlusup[j] points to the starting location of the j-th column in
125  * storage vector lusup[*];
126  * xlusup_end[j] points to one past the ending location of the j-th
127  * column in lusup[*].
128  * Each supernode is stored in column-major, consistent with Fortran
129  * two-dimensional array storage.
130  *
131  * (ucol, usub, xusub, xusub_end):
132  * ucol[*] stores the numerical values of the U-columns above the
133  * supernodes.
134  * usub[k] stores the row subscripts of nonzeros ucol[k];
135  * xusub[j] points to the starting location of column j in ucol/usub[];
136  * xusub_end[j] points to one past the ending location column j in
137  * ucol/usub[].
138  * Storage: new row subscripts; that is indexed intp PA.
139  *
140  */
141 typedef struct {
142  int *xsup; /* supernode and column mapping */
143  int *xsup_end;
144  int *supno;
145  int *lsub; /* compressed L subscripts */
146  int *xlsub;
147  int *xlsub_end;
148  float *lusup; /* L supernodes */
149  int *xlusup;
150  int *xlusup_end;
151  float *ucol; /* U columns */
152  int *usub;
153  int *xusub;
154  int *xusub_end;
155  int nsuper; /* current supernode number */
156  int nextl; /* next position in lsub[] */
157  int nextu; /* next position in usub[]/ucol[] */
158  int nextlu; /* next position in lusup[] */
159  int nzlmax; /* current max size of lsub[] */
160  int nzumax; /* " " " ucol[] */
161  int nzlumax; /* " " " lusup[] */
162  /* ---------------------------------------------------------------
163  * Memory managemant for L supernodes
164  */
165  int *map_in_sup; /* size n+1 - the address offset of each column
166  * in lusup[*], which is divided into regions
167  * by the supernodes of Householder matrix H.
168  * If column k starts a supernode in H,
169  * map_in_sup[k] is the next open position in
170  * lusup[*]; otherwise map_in_sup[k] gives the
171  * offset (negative) to the leading column
172  * of the supernode in H.
173  */
174  int dynamic_snode_bound;
175  /* --------------------------------------------------------------- */
176 } GlobalLU_t;
