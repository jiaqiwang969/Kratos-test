353  double Solve()
354  {
355  KRATOS_TRY
356 
357 
358  //pointers needed in the solution
359  typename TSchemeType::Pointer pScheme = GetScheme();
360  typename TBuilderAndSolverType::Pointer pBuilderAndSolver = GetBuilderAndSolver();
361 
362  //int solstep = pCurrentProcessInfo.GetCurrentSolutionStep();
363  DofsArrayType& rDofSet = pBuilderAndSolver->GetDofSet();
364 
365  //OPERATIONS THAT SHOULD BE DONE ONCE - internal check to avoid repetitions
366  //if the operations needed were already performed this does nothing
367  if(mInitializeWasPerformed == false)
368  Initialize();
369 
370  //set up the system, operation performed just once unless it is required
371  //to reform the dof set at each iteration
372  if (pBuilderAndSolver->GetDofSetIsInitializedFlag() == false ||
373  mReformDofSetAtEachStep == true )
374  {
375  //setting up the list of the DOFs to be solved
376  pBuilderAndSolver->SetUpDofSet(pScheme,BaseType::GetModelPart());
377 
378  //shaping correctly the system
379  pBuilderAndSolver->SetUpSystem(BaseType::GetModelPart());
380  }
381 
382  //prints informations about the current time
383  if (this->GetEchoLevel()!=0)
384  {
385  std::cout << " " << std::endl;
386  std::cout << "CurrentTime = " << BaseType::GetModelPart().GetProcessInfo()[TIME] << std::endl;
387  }
388 
389  //updates the database with a prediction of the solution
390  Predict();
391 
392  //initialize solution step
393  if (mSolutionStepIsInitialized == false)
394  InitializeSolutionStep();
395 
396  TSystemMatrixType& mA = *mpA;
397  TSystemVectorType& mDx = *mpDx;
398  TSystemVectorType& mb = *mpb;
399 
400 
401 
402  //initializing the parameters of the Newton-Raphson cicle
403  unsigned int iteration_number=1;
404  BaseType::GetModelPart().GetProcessInfo()[NL_ITERATION_NUMBER] = iteration_number;
405 // BaseType::GetModelPart().GetProcessInfo().SetNonLinearIterationNumber(iteration_number);
406  bool is_converged = false;
407  bool ResidualIsUpdated = false;
408  pScheme->InitializeNonLinIteration(BaseType::GetModelPart(),mA,mDx,mb);
409  is_converged = mpConvergenceCriteria->PreCriteria(BaseType::GetModelPart(),rDofSet,mA,mDx,mb);
410 
411  //function to perform the building and the solving phase.
412  if(BaseType::mRebuildLevel >1 || BaseType::mStiffnessMatrixIsBuilt == false)
413  {
414  TSparseSpace::SetToZero(mA);
415  TSparseSpace::SetToZero(mDx);
416  TSparseSpace::SetToZero(mb);
417 
418  //pBuilderAndSolver->BuildAndSolve(pScheme,BaseType::GetModelPart(),mA,mDx,mb);
419  pBuilderAndSolver->Build(pScheme,BaseType::GetModelPart(),mA,mb);
420 
421  pBuilderAndSolver->BuildRHS(pScheme,BaseType::GetModelPart(),mb);
422 
423  pBuilderAndSolver->SystemSolve(mA,mDx,mb);
424 
425  }
426  else
427  {
428  TSparseSpace::SetToZero(mDx); //mDx=0.00;
429  TSparseSpace::SetToZero(mb);
430 
431  pBuilderAndSolver->BuildRHSAndSolve(pScheme,BaseType::GetModelPart(),mA,mDx,mb);
432  }
433 
434  if (this->GetEchoLevel()==3) //if it is needed to print the debug info
435  {
436  std::cout << "After first system solution" << std::endl;
437  std::cout << "SystemMatrix = " << mA << std::endl;
438  std::cout << "solution obtained = " << mDx << std::endl;
439  std::cout << "RHS = " << mb << std::endl;
440  }
441 
442  //update results
443  rDofSet = pBuilderAndSolver->GetDofSet();
444  pScheme->Update(BaseType::GetModelPart(),rDofSet,mA,mDx,mb);
445 
446  //move the mesh if needed
447  if(BaseType::MoveMeshFlag() == true) BaseType::MoveMesh();
448 
449  pScheme->FinalizeNonLinIteration(BaseType::GetModelPart(),mA,mDx,mb);
450 
451  if (is_converged==true)
452  {
453  //initialisation of the convergence criteria
454  rDofSet = pBuilderAndSolver->GetDofSet();
455  mpConvergenceCriteria->InitializeSolutionStep(BaseType::GetModelPart(),rDofSet,mA,mDx,mb);
456 
457  if (mpConvergenceCriteria->GetActualizeRHSflag() == true)
458  {
459  TSparseSpace::SetToZero(mb);
460 
461  pBuilderAndSolver->BuildRHS(pScheme,BaseType::GetModelPart(),mb);
462  }
463 
464  is_converged = mpConvergenceCriteria->PostCriteria(BaseType::GetModelPart(),rDofSet,mA,mDx,mb);
465  }
466 
467 
468  //Iteration Cicle... performed only for NonLinearProblems
469  while( is_converged == false &&
470  iteration_number++<mMaxIterationNumber)
471  {
472  //setting the number of iteration
473  BaseType::GetModelPart().GetProcessInfo()[NL_ITERATION_NUMBER] = iteration_number;
474 
475  pScheme->InitializeNonLinIteration(BaseType::GetModelPart(),mA,mDx,mb);
476 
477 
478  is_converged = mpConvergenceCriteria->PreCriteria(BaseType::GetModelPart(),rDofSet,mA,mDx,mb);
479 
480  //call the linear system solver to find the correction mDx for the
481  //it is not called if there is no system to solve
482  if (SparseSpaceType::Size(mDx)!=0)
483  {
484  if(BaseType::mRebuildLevel >1 || BaseType::mStiffnessMatrixIsBuilt == false)
485  {
486  //mA = 0.00;
487  TSparseSpace::SetToZero(mA);
488  TSparseSpace::SetToZero(mDx);
489  TSparseSpace::SetToZero(mb);
490 
491  //pBuilderAndSolver->BuildAndSolve(pScheme,BaseType::GetModelPart(),mA,mDx,mb);
492  pBuilderAndSolver->Build(pScheme,BaseType::GetModelPart(),mA,mb);
493  pBuilderAndSolver->BuildRHS(pScheme,BaseType::GetModelPart(),mb);
494  pBuilderAndSolver->SystemSolve(mA,mDx,mb);
495 
496 
497  }
498  else
499  {
500  TSparseSpace::SetToZero(mDx);
501  TSparseSpace::SetToZero(mb);
502 
503  pBuilderAndSolver->BuildRHSAndSolve(pScheme,BaseType::GetModelPart(),mA,mDx,mb);
504  }
505  }
506  else
507  {
508  std::cout << "ATTENTION: no free DOFs!! " << std::endl;
509  }
510 
511 
512  //Updating the results stored in the database
513  rDofSet = pBuilderAndSolver->GetDofSet();
514  pScheme->Update(BaseType::GetModelPart(),rDofSet,mA,mDx,mb);
515 
516  //move the mesh if needed
517  if(BaseType::MoveMeshFlag() == true) BaseType::MoveMesh();
518 
519  pScheme->FinalizeNonLinIteration(BaseType::GetModelPart(),mA,mDx,mb);
520 
521  ResidualIsUpdated = false;
522 
523  if (is_converged==true)
524  {
525 
526  if (mpConvergenceCriteria->GetActualizeRHSflag() == true)
527  {
528  TSparseSpace::SetToZero(mb);
529 
530  pBuilderAndSolver->BuildRHS(pScheme,BaseType::GetModelPart(),mb);
531  ResidualIsUpdated = true;
532  //std::cout << "mb is calculated" << std::endl;
533  }
534 
535  is_converged = mpConvergenceCriteria->PostCriteria(BaseType::GetModelPart(),rDofSet,mA,mDx,mb);
536  }
537  }
538 
539 
540  //plots a warning if the maximum number of iterations is exceeded
541  if (iteration_number>=mMaxIterationNumber)
542  MaxIterationsExceeded();
543 
544  //recalculate residual if needed
545  // (note that some convergence criteria need it to be recalculated)
546  if (ResidualIsUpdated==false)
547  {
548  TSparseSpace::SetToZero(mb);
549 
550  pBuilderAndSolver->BuildRHS(pScheme,BaseType::GetModelPart(),mb);
551 
552  //std::cout << "mb is calculated" << std::endl;
553  }
554 
555  //calculate reactions if required
556  if (mCalculateReactionsFlag ==true)
557  {
558  pBuilderAndSolver->CalculateReactions(pScheme,BaseType::GetModelPart(),mA,mDx,mb);
559  }
560 
561  //Finalisation of the solution step,
562  //operations to be done after achieving convergence, for example the
563  //Final Residual Vector (mb) has to be saved in there
564  //to avoid error accumulation
565  pScheme->FinalizeSolutionStep(BaseType::GetModelPart(),mA,mDx,mb);
566  pBuilderAndSolver->FinalizeSolutionStep(BaseType::GetModelPart(),mA,mDx,mb);
567 
568  //Cleaning memory after the solution
569  pScheme->Clean();
570 
571  //reset flags for next step
572  mSolutionStepIsInitialized = false;
573 
574  if (mReformDofSetAtEachStep == true) //deallocate the systemvectors
575  {
576  //TSparseSpace::Clear(mA);
577  //TSparseSpace::Clear(mDx);
578  //TSparseSpace::Clear(mb);
579 
580  TSparseSpace::ClearData(mA);
581  TSparseSpace::ClearData(mDx);
582  TSparseSpace::ClearData(mb);
583  }
584 
585  return 0.00;
586 
587  KRATOS_CATCH("")
588 
589  }
