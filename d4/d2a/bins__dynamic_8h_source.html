138  void CalculateBoundingBox() {
139  for(SizeType i = 0 ; i < TDimension ; i++){
140  mMinPoint[i] = (**mPointBegin)[i];
141  mMaxPoint[i] = (**mPointBegin)[i];
142  }
143  for(IteratorType Point = mPointBegin ; Point != mPointEnd ; Point++)
144  for(SizeType i = 0 ; i < TDimension ; i++){
145  if( (**Point)[i] < mMinPoint[i] ) mMinPoint[i] = (**Point)[i];
146  if( (**Point)[i] > mMaxPoint[i] ) mMaxPoint[i] = (**Point)[i];

152  void CalculateCellSize() {
153 
154  CoordinateType delta[TDimension];
155  CoordinateType alpha[TDimension];
156  CoordinateType mult_delta = 1.00;
157  for(SizeType i = 0 ; i < TDimension ; i++) {
158  delta[i] = mMaxPoint[i] - mMinPoint[i];
159  delta[i] = (delta[i] == 0.00) ? 1.00 : delta[i];
160  }
161 
162  alpha[0] = 1.00;
163  for(SizeType i = 1 ; i < TDimension ; i++){
164  alpha[i] = delta[i] / delta[0];
165  mult_delta *= alpha[i];
166  }
167 
168  mN[0] = static_cast<SizeType>( pow(static_cast<CoordinateType>(std::distance(mPointBegin,mPointEnd)/mult_delta), 1.00/3.00)+1 );
169 
170  for(SizeType i = 1 ; i < TDimension ; i++)
171  mN[i] = static_cast<SizeType>(alpha[i] * mN[0] + 1);
172 
173  for(SizeType i = 0 ; i < TDimension ; i++){
174  mCellSize[i] = delta[i] / mN[i];
175  mInvCellSize[i] = 1.00 / mCellSize[i];
176  }
177  }
178

181  void CalculateCellSize( CoordinateType BoxSize ) {
182  for(SizeType i = 0 ; i < TDimension ; i++){
183  mCellSize[i] = BoxSize;
184  mInvCellSize[i] = 1.00 / mCellSize[i];
185  mN[i] = static_cast<SizeType>( (mMaxPoint[i]-mMinPoint[i]) / mCellSize[i]) + 1;
186  }
187  }
