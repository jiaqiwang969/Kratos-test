228  unsigned int send_buffer_size = nodal_data_size * rThisModelPart.NumberOfNodes(ModelPart::Kratos_Ownership_Size);

232  double* receive_buffer = new double[receive_buffer_size];

237 
238  // Filling the buffer

242  for(unsigned int i = 0 ; i < nodal_data_size ; i++)

244 
245  if(position > send_buffer_size)
246  std::cout << rank << " Error in estimating send buffer size...." << std::endl;
