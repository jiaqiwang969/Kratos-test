661  virtual CoordinatesArrayType& PointLocalCoordinates( CoordinatesArrayType& rResult,
662  const CoordinatesArrayType& rPoint )
663  {
664  Matrix J = ZeroMatrix( LocalSpaceDimension(), LocalSpaceDimension() );
665  if( rResult.size() != LocalSpaceDimension() )
666  rResult.resize( LocalSpaceDimension() );
667  //starting with xi = 0
668  rResult = ZeroVector( LocalSpaceDimension() );
669  Vector DeltaXi = ZeroVector( LocalSpaceDimension() );
670  CoordinatesArrayType CurrentGlobalCoords( ZeroVector(3) );
671  //Newton iteration:
672  double tol = 1.0e-8;
673  int maxiter = 1000;
674  for( int k=0; k<maxiter; k++ )
675  {
676  CurrentGlobalCoords = ZeroVector(3);
677  GlobalCoordinates( CurrentGlobalCoords, rResult );
678  noalias( CurrentGlobalCoords ) = rPoint - CurrentGlobalCoords;
679  InverseOfJacobian( J, rResult );
680  noalias(DeltaXi) = prod(J,CurrentGlobalCoords);
681  noalias( rResult ) += DeltaXi;
682  if( MathUtils<double>::Norm3(DeltaXi) > 30 )
683  {
684  break;
685  }
686  if( MathUtils<double>::Norm3(DeltaXi) < tol )
687  {
688  break;
689  }
690  }
691  return( rResult );
692  }
