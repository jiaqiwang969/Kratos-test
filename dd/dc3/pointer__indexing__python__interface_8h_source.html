58  // pointer_indexing_suite class. This class is the protocol class for
59  // the management of C++ containers intended to be integrated
60  // to Python. The objective is make a C++ container look and
61  // feel and behave exactly as we'd expect a Python container.
62  // By default indexed elements are returned by proxy. This can be
63  // disabled by supplying *true* in the NoProxy template parameter.
64  //
65  // Derived classes provide the hooks needed by the pointer_indexing_suite
66  // to do its job:
67  //
68  // static data_type&
69  // get_item(Container& container, index_type i);
70  //
71  // static object
72  // get_slice(Container& container, index_type from, index_type to);
73  //
74  // static void
75  // set_item(Container& container, index_type i, data_type const& v);
76  //
77  // static void
78  // set_slice(
79  // Container& container, index_type from,
80  // index_type to, data_type const& v
81  // );
82  //
83  // template <class Iter>
84  // static void
85  // set_slice(Container& container, index_type from,
86  // index_type to, Iter first, Iter last
87  // );
88  //
89  // static void
90  // delete_item(Container& container, index_type i);
91  //
92  // static void
93  // delete_slice(Container& container, index_type from, index_type to);
94  //
95  // static size_t
96  // size(Container& container);
97  //
98  // template <class T>
99  // static bool
100  // contains(Container& container, T const& val);
101  //
102  // static index_type
103  // convert_index(Container& container, PyObject* i);
104  //
105  // static index_type
106  // adjust_index(index_type current, index_type from,
107  // index_type to, size_type len
108  // );
109  //
110  // Most of these policies are self explanatory. convert_index and
111  // adjust_index, however, deserves some explanation.
112  //
113  // convert_index converts an Python index into a C++ index that the
114  // container can handle. For instance, negative indexes in Python, by
115  // convention, indexes from the right (e.g. C[-1] indexes the rightmost
116  // element in C). convert_index should handle the necessary conversion
117  // for the C++ container (e.g. convert -1 to C.size()-1). convert_index
118  // should also be able to convert the type of the index (A dynamic Python
119  // type) to the actual type that the C++ container expects.
120  //
121  // When a container expands or contracts, held indexes to its elements
122  // must be adjusted to follow the movement of data. For instance, if
123  // we erase 3 elements, starting from index 0 from a 5 element vector,
124  // what used to be at index 4 will now be at index 1:
125  //
126  // [a][b][c][d][e] ---> [d][e]
127  // ^ ^
128  // 4 1
129  //
130  // adjust_index takes care of the adjustment. Given a current index,
131  // the function should return the adjusted index when data in the
132  // container at index from..to is replaced by *len* elements.
133  //
