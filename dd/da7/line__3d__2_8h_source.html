395  virtual JacobiansType& Jacobian(JacobiansType& rResult, IntegrationMethod ThisMethod) const
396  {
397  Matrix jacobian(3,1);
398  jacobian(0,0)=(this->GetPoint(1).X()-this->GetPoint(0).X())*0.5; //on the Gauss points (J is constant at each element)
399  jacobian(1,0)=(this->GetPoint(1).Y()-this->GetPoint(0).Y())*0.5;
400  jacobian(2,0)=(this->GetPoint(1).Z()-this->GetPoint(0).Z())*0.5;
401 
402  if(rResult.size() != BaseType::IntegrationPointsNumber(ThisMethod))
403  {
404  // KLUDGE: While there is a bug in ublas vector resize, I have to put this beside resizing!!
405  JacobiansType temp(BaseType::IntegrationPointsNumber(ThisMethod));
406  rResult.swap(temp);
407  }
408  std::fill(rResult.begin(), rResult.end(), jacobian);
409  return rResult;
410  }

713  static Matrix CalculateShapeFunctionsIntegrationPointsValues(typename BaseType::IntegrationMethod ThisMethod)
714  {
715  const IntegrationPointsContainerType& all_integration_points = AllIntegrationPoints();
716  const IntegrationPointsArrayType& IntegrationPoints = all_integration_points[ThisMethod];
717  int integration_points_number = IntegrationPoints.size();
718  Matrix N(integration_points_number,2);
719  for(int it_gp = 0; it_gp < integration_points_number; it_gp++)
720  {
721  double e = IntegrationPoints[it_gp].X();
722  N(it_gp,0) = 0.5*(1-e);
723  N(it_gp,1) = 0.5*(1+e);
724  }
725 
726  return N;
727  }
