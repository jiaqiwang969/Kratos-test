162  static void InvertMatrix3(const MatrixType& InputMatrix,
163  MatrixType& InvertedMatrix,
164  TDataType& InputMatrixDet) //VERIFIED!!!
165  {
166  KRATOS_TRY
167  if(InvertedMatrix.size1() != 3 || InvertedMatrix.size2() != 3)
168  InvertedMatrix.resize(3,3);
169 
170  //filling the inverted matrix with the algebraic complements
171  //first column
172  InvertedMatrix(0,0) = InputMatrix(1,1)*InputMatrix(2,2) - InputMatrix(1,2)*InputMatrix(2,1);
173  InvertedMatrix(1,0) = -InputMatrix(1,0)*InputMatrix(2,2) + InputMatrix(1,2)*InputMatrix(2,0);
174  InvertedMatrix(2,0) = InputMatrix(1,0)*InputMatrix(2,1) - InputMatrix(1,1)*InputMatrix(2,0);
175 
176  //second column
177  InvertedMatrix(0,1) = -InputMatrix(0,1)*InputMatrix(2,2) + InputMatrix(0,2)*InputMatrix(2,1);
178  InvertedMatrix(1,1) = InputMatrix(0,0)*InputMatrix(2,2) - InputMatrix(0,2)*InputMatrix(2,0);
179  InvertedMatrix(2,1) = -InputMatrix(0,0)*InputMatrix(2,1) + InputMatrix(0,1)*InputMatrix(2,0);
180 
181  //third column
182  InvertedMatrix(0,2) = InputMatrix(0,1)*InputMatrix(1,2) - InputMatrix(0,2)*InputMatrix(1,1);
183  InvertedMatrix(1,2) = -InputMatrix(0,0)*InputMatrix(1,2) + InputMatrix(0,2)*InputMatrix(1,0);
184  InvertedMatrix(2,2) = InputMatrix(0,0)*InputMatrix(1,1) - InputMatrix(0,1)*InputMatrix(1,0);
185 
186  //calculation of determinant (of the input matrix)
187  InputMatrixDet = InputMatrix(0,0)*InvertedMatrix(0,0) + InputMatrix(0,1)*InvertedMatrix(1,0) + InputMatrix(0,2)*InvertedMatrix(2,0);
188 
189  //finalizing the calculation of the inverted matrix
190  InvertedMatrix /= InputMatrixDet;
191  KRATOS_CATCH("")
192  }
