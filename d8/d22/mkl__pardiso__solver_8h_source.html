144  typedef boost::numeric::bindings::traits::vector_traits<VectorType> mbtraits;
145  typedef typename matraits::value_type val_t;
146 
147  MKL_INT n = matraits::size1 (rA);

149  assert (n == mbtraits::size1 (rB));
150  assert (n == mbtraits::size1 (rX));
151 
155  double* a = matraits::value_storage(rA);
156 
160  int *index1_vector = new (std::nothrow) int[rA.index1_data().size()];
161  int *index2_vector = new (std::nothrow) int[rA.index2_data().size()];
162  std::cout << "Size of the problem: " << n << std::endl;
163  std::cout << "Size of index1_vector: " << rA.index1_data().size() << std::endl;
164  std::cout << "Size of index2_vector: " << rA.index2_data().size() << std::endl;
165  for( int i = 0; i < rA.index1_data().size(); i++ )
166  {
167  index1_vector[i] = (int)(rA.index1_data()[i])+1;
168  }
169  for( int i = 0; i < rA.index2_data().size(); i++ )
170  {
171  index2_vector[i] = (int)(rA.index2_data()[i])+1;
172  }
173 
174  MKL_INT mtype = 11; /* Real nonsymmetric matrix */
175  /* RHS and solution vectors. */
176  double *b = mbtraits::storage(rB);
177  double *x = mbtraits::storage(rX);
178 
179  MKL_INT nrhs = 1; /* Number of right hand sides. */
180  /* Internal solver memory pointer pt, */
181  /* 32-bit: int pt[64]; 64-bit: long int pt[64] */
182  /* or void *pt[64] should be OK on both architectures */
183  void *pt[64];
184  /* Pardiso control parameters. */
185  MKL_INT iparm[64];
186  MKL_INT maxfct, mnum, phase, error, msglvl;
187  /* Auxiliary variables. */
188  MKL_INT i;
189  double ddum; /* Double dummy */
190  MKL_INT idum; /* Integer dummy. */
191  /* -------------------------------------------------------------------- */
192  /* .. Setup Pardiso control parameters. */
193  /* -------------------------------------------------------------------- */
194  for (i = 0; i < 64; i++) {
195  iparm[i] = 0;
196  }
197  iparm[0] = 1; /* No solver default */
198  iparm[1] = 2; /* Fill-in reordering from METIS */
199  /* Numbers of processors, value of OMP_NUM_THREADS */
200  iparm[2] = omp_get_max_threads();
201  std::cout << "number of threads: " << iparm[2] << std::endl;
202  if( mRefinements > 0 )
203  iparm[3] = 1; /* iterative-direct algorithm */
204  else
205  iparm[3] = 0; /* no iterative-direct algorithm */
206  iparm[4] = 0; /* No user fill-in reducing permutation */
207  iparm[5] = 0; /* Write solution into x */
208  iparm[6] = 0; /* Not in use */
209  iparm[7] = mRefinements; /* Max numbers of iterative refinement steps */
210  iparm[8] = 0; /* Not in use */
211  iparm[9] = 13; /* Perturb the pivot elements with 1E-13 */
212  iparm[10] = 1; /* Use nonsymmetric permutation and scaling MPS */
213  iparm[11] = 0; /* Not in use */
214  iparm[12] = 0; /* Not in use */
215  iparm[13] = 0; /* Output: Number of perturbed pivots */
216  iparm[14] = 0; /* Not in use */
217  iparm[15] = 0; /* Not in use */
218  iparm[16] = 0; /* Not in use */
219  iparm[17] = -1; /* Output: Number of nonzeros in the factor LU */
220  iparm[18] = -1; /* Output: Mflops for LU factorization */
221  iparm[19] = 0; /* Output: Numbers of CG Iterations */
222  maxfct = 1; /* Maximum number of numerical factorizations. */
223  mnum = 1; /* Which factorization to use. */
224  msglvl = 0; /* Print statistical information in file */
225  error = 0; /* Initialize error flag */
226  /* -------------------------------------------------------------------- */
227  /* .. Initialize the internal solver memory pointer. This is only */
228  /* necessary for the FIRST call of the PARDISO solver. */
229  /* -------------------------------------------------------------------- */
230  for (i = 0; i < 64; i++) {
231  pt[i] = 0;
232  }
233  /* -------------------------------------------------------------------- */
234  /* .. Reordering and Symbolic Factorization. This step also allocates */
235  /* all memory that is necessary for the factorization. */
236  /* -------------------------------------------------------------------- */
237 
238  phase = 11;
239  PARDISO (pt, &maxfct, &mnum, &mtype, &phase,
240  &n, a, index1_vector, index2_vector, &idum, &nrhs,
241  iparm, &msglvl, &ddum, &ddum, &error);
242 
243  if (error != 0) {

245  exit(1);
246  }
247  std::cout << "Reordering completed ... " << std::endl;
248  //printf("\nNumber of nonzeros in factors = %d", iparm[17]);
249  //printf("\nNumber of factorization MFLOPS = %d", iparm[18]);
250  /* -------------------------------------------------------------------- */
251  /* .. Numerical factorization. */
252  /* -------------------------------------------------------------------- */
253  phase = 22;
254  PARDISO (pt, &maxfct, &mnum, &mtype, &phase,
255  &n, a, index1_vector, index2_vector, &idum, &nrhs,
256  iparm, &msglvl, &ddum, &ddum, &error);
257  if (error != 0) {
258  std::cout << "ERROR during numerical factorization: " << error << std::endl;
259  exit(2);
260  }
261  std::cout << "Factorization completed ... " << std::endl;
262  /* -------------------------------------------------------------------- */
263  /* .. Back substitution and iterative refinement. */
264  /* -------------------------------------------------------------------- */
265  phase = 33;
266  iparm[7] = 2; /* Max numbers of iterative refinement steps. */
267  /* Set right hand side to one. */
268  //for (i = 0; i < n; i++) {
269  // b[i] = 1;
270  //}
271  PARDISO (pt, &maxfct, &mnum, &mtype, &phase,
272  &n, a, index1_vector, index2_vector, &idum, &nrhs,
273  iparm, &msglvl, b, x, &error);
274  if (error != 0) {
275  std::cout << "ERROR during solution: " << error << std::endl;
276  exit(3);
277  }
278  /* -------------------------------------------------------------------- */
279  /* .. Termination and release of memory. */
280  /* -------------------------------------------------------------------- */
281  phase = -1; /* Release internal memory. */
282  PARDISO (pt, &maxfct, &mnum, &mtype, &phase,
283  &n, &ddum, index1_vector, index2_vector, &idum, &nrhs,
284  iparm, &msglvl, &ddum, &ddum, &error);
285  delete [] index1_vector;
286  delete [] index2_vector;

289  return true;
