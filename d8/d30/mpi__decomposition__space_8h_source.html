382  template<class TOtherMatrixType, class TEquationIdVectorType>
383  inline static void AssembleLHS(
384  MatrixType& A,
385  TOtherMatrixType& LHS_Contribution,
386  TEquationIdVectorType& EquationId
387  )
388  {
389  unsigned int system_size = A.size1();
390  unsigned int local_size = LHS_Contribution.size1();
391 
392  for (unsigned int i_local=0; i_local<local_size; i_local++)
393  {
394  unsigned int i_global=EquationId[i_local];
395  if ( i_global < system_size )
396  {
397  for (unsigned int j_local=0; j_local<local_size; j_local++)
398  {
399  unsigned int j_global=EquationId[j_local];
400  if ( j_global < system_size )
401  A(i_global,j_global) += LHS_Contribution(i_local,j_local);
402  }
403  }
404  }
405  }
