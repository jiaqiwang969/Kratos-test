220  bool IterativeSolve(SparseMatrixType& rA, VectorType& rX, VectorType& rB)
221  {
222  const int size = TSparseSpaceType::Size(rX);
223 
224 
225  int info_count = 0;
226  BaseType::mIterationsNumber = 0;
227 
228 
229  BaseType::mBNorm = TSparseSpaceType::TwoNorm(rB);
230 
231 
232  double errtol = BaseType::mBNorm*BaseType::GetTolerance();
233 
234 
235  VectorType r(rB);
236  VectorType w(r);
237 
238 
239  VectorType y1(r);
240  VectorType y2(size);
241  for(typename VectorType::iterator ity2 = y2.begin(); ity2 != y2.end(); ity2++) *ity2 = 0.00;
242 
243 
244  VectorType d(size);
245  for(typename VectorType::iterator itd = d.begin(); itd != d.end(); itd++) *itd = 0.00;
246 
247 
248  VectorType v(size);
249  PreconditionedMult(rA, y1, v);
250 
251 
252  VectorType u1(v);
253  VectorType u2(size);
254  for(typename VectorType::iterator itu2 = u2.begin(); itu2 != u2.end(); itu2++) *itu2 = 0.00;
255 
256 
257  double theta = 0.00;
258  double eta = 0.00;
259  double sigma = 0.00;
260  double alpha = 0.00;
261  double c = 0.00;
262 
263 
264  double tau = TSparseSpaceType::TwoNorm(r);
265  double rho = tau*tau;
266  double rhon = 0.00;
267  double beta = 0.00;
268 
269 
270  int m = 0;
271 
272 
273 
274  while(BaseType::mIterationsNumber < BaseType::GetMaxIterationsNumber() )
275  {
276 
277 
278  BaseType::mIterationsNumber++;
279 
280 
281  sigma = TSparseSpaceType::Dot(r,v);
282 
283  if (sigma == 0.00) break;
284 
285  alpha = rho/sigma;
286 
287 
288 
290 
291 
292  m = 2*BaseType::mIterationsNumber-1;
293 
294 
295  //w=w-alpha*u1;
296  TSparseSpaceType::ScaleAndAdd(-alpha, u1, 1.00, w);
297 
298 
299  //d=y1+(theta*theta*eta/alpha)*d;
300  TSparseSpaceType::ScaleAndAdd(1.00, y1, (theta*theta*eta/alpha), d);
301 
302  theta = TSparseSpaceType::TwoNorm(w)/tau;
303  c = 1.00/sqrt(1.00+theta*theta);
304  tau = tau*theta*c;
305  eta = c*c*alpha;
306 
307 
308  //x=x+eta*d;
309  TSparseSpaceType::ScaleAndAdd(eta, d, 1.00, rX);
310 
311  BaseType::mResidualNorm = tau*sqrt((double)(m+1));
312 
313  if (BaseType::mResidualNorm <= errtol) break;
314 
315 
316 
318 
319 
320  m = 2*BaseType::mIterationsNumber;
321 
322 
323  //y2=y1-alpha*v;
324  TSparseSpaceType::ScaleAndAdd(1.00, y1, -alpha, v, y2);
325 
326 
327  //u2=A*y2;
328  PreconditionedMult(rA,y2,u2);
329 
330  //w=w-alpha*u2;
331  TSparseSpaceType::ScaleAndAdd(-alpha, u2, 1.00, w);
332 
333  //d=y2+(theta*theta*eta/alpha)*d;
334  TSparseSpaceType::ScaleAndAdd(1.00, y2, (theta*theta*eta/alpha), d);
335 
336 
337  theta = TSparseSpaceType::TwoNorm(w)/tau;
338  c = 1.00/sqrt(1+theta*theta);
339  tau = tau*theta*c;
340  eta = c*c*alpha;
341 
342 
343  //x=x+eta*d;
344  TSparseSpaceType::ScaleAndAdd(eta, d, 1.00, rX);
345 
346 
347  BaseType::mResidualNorm = tau*sqrt((double)(m+1));
348 
349  if (BaseType::mResidualNorm <= errtol) break;
350 
351 
352 
354 
355 
356 
357  if (rho == 0.00) break;
358 
359  rhon = TSparseSpaceType::Dot(r,w);
360  beta = rhon/rho;
361  rho = rhon;
362 
363 
364  //y1= w + beta*y2;
365  TSparseSpaceType::ScaleAndAdd(1.00, w, beta, y2, y1);
366 
367 
368  //u1 = A*y1;
369  PreconditionedMult(rA,y1,u1);
370 
371  //v=u1+beta*(u2+beta*v);
372  TSparseSpaceType::ScaleAndAdd(1.00, u2, beta, v);
373  TSparseSpaceType::ScaleAndAdd(1.00, u1, beta, v);
374 
375 
376  //Print iteration info
377  info_count++;
378  if (info_count==100)
379  {
380  std::cout<<"it = "<<BaseType::mIterationsNumber<<" res = "<<(BaseType::mResidualNorm/BaseType::mBNorm)<<std::endl;
381  info_count=0;
382  }

385  }
