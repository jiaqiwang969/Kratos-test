83  Element const& rReferenceElement,
84  Condition const& rReferenceBoundaryCondition
85  )

89  //assigning the nodes to the new model part
90  DestinationModelPart.Nodes().clear();
91  DestinationModelPart.Nodes() = OriginModelPart.Nodes();
92 
93  //generating the elements
94  int id = 1;
95  Properties::Pointer properties = OriginModelPart.GetMesh().pGetProperties(1);
96  for(ModelPart::ElementsContainerType::iterator iii = OriginModelPart.ElementsBegin(); iii != OriginModelPart.ElementsEnd(); iii++)
97  {
98  Geometry< Node<3> >& geom = iii->GetGeometry();
99  Element::Pointer p_element = rReferenceElement.Create(id, geom ,properties);
100  DestinationModelPart.Elements().push_back(p_element);
101  id = id + 1;
102  }
103  std::cout << "Elements are generated" << std::endl;
104 
105  //generating the conditions
106  id = 1;
107  for(ModelPart::ConditionsContainerType::iterator iii = OriginModelPart.ConditionsBegin(); iii != OriginModelPart.ConditionsEnd(); iii++)
108  {
109  Geometry< Node<3> >& geom = iii->GetGeometry();
110  double nfree_surf = 0;
111  for(unsigned int k = 0; k<geom.size(); k++)
112  nfree_surf += geom[k].FastGetSolutionStepValue(IS_FREE_SURFACE);
113 
114  if(nfree_surf > 1)
115  {
116  Condition::Pointer p_condition = rReferenceBoundaryCondition.Create(id, geom,properties);
117  DestinationModelPart.Conditions().push_back(p_condition);
118  id = id + 1;
119  }
120  }
121  std::cout << "Conditions are generated" << std::endl;
122 
123  KRATOS_CATCH("");
124  }
