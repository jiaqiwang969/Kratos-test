148  void Deactivate( ModelPart& model_part, int from_level, int to_level )
149  {
150  KRATOS_TRY;
151 
152  //first step: reactivate all elements and conditions
153  ReactivateAll( model_part );
154  //second step: deactivate elements and conditions to be deactivated currently
155  // identify elements to be deactivated
156  std::vector<int> elements_to_be_deleted;
157  for ( ElementsArrayType::ptr_iterator it=model_part.Elements().ptr_begin();
158  it!=model_part.Elements().ptr_end(); ++it)
159  {
160  //if element is to be deactivated
161  if( ( (*it)->GetValue( ACTIVATION_LEVEL ) > from_level
162  && (*it)->GetValue( ACTIVATION_LEVEL ) < to_level)
163  || ( (*it)->GetValue( ACTIVATION_LEVEL ) < 0 )
164  )
165  {
166  elements_to_be_deleted.push_back((*it)->Id());
167  mDeactivatedElements.push_back( *(it) );
168  // mDeactivatedElements.push_back( *(it.base()) );
169  //deactivate associated conditions (unless they are contact master)
170  Condition Cond = model_part.Conditions()[(*it)->Id()];
171  if( Cond.GetValue( IS_CONTACT_MASTER ) == false )
172  {
173  mDeactivatedConditions.push_back( Cond );
174  model_part.RemoveCondition( Cond );
175  }
176  }
177  }
178  // deactivate identified elements
179  for( std::vector<int>::iterator it=elements_to_be_deleted.begin();
180  it != elements_to_be_deleted.end(); it++ )
181  {
182  model_part.RemoveElement( *it );
183  }
184  //re-sort the elements
185  model_part.Elements().Sort();
186  model_part.Conditions().Sort();
187 
188  KRATOS_CATCH("")
189  }

195  void ReactivateAll( ModelPart& model_part )
196  {
197  for( ElementsArrayType::iterator it = mDeactivatedElements.begin();
198  it != mDeactivatedElements.end(); it++ )
199  {
200  model_part.AddElement( *(it.base()));
201  }
202  for( ConditionsArrayType::iterator jt = mDeactivatedConditions.begin();
203  jt != mDeactivatedConditions.end(); jt++ )
204  {
205  model_part.AddCondition( *(jt.base()));
206  }
207  model_part.Elements().Sort();
208  model_part.Conditions().Sort();
209 
210  mDeactivatedElements.clear();
211  mDeactivatedConditions.clear();
