451  //returns false if it should be removed
452  bool AlphaShape(double alpha_param, Geometry<Node<3> >& pgeom)
453  //bool AlphaShape(double alpha_param, Triangle2D<Node<3> >& pgeom)
454  {
455  KRATOS_TRY
456 
457 
458  double x0 = pgeom[0].X();
459  double x1 = pgeom[1].X();
460  double x2 = pgeom[2].X();
461 
462  double y0 = pgeom[0].Y();
463  double y1 = pgeom[1].Y();
464  double y2 = pgeom[2].Y();
465 
466  mJ(0,0)=2.0*(x1-x0); mJ(0,1)=2.0*(y1-y0);
467  mJ(1,0)=2.0*(x2-x0); mJ(1,1)=2.0*(y2-y0);
468 
469 
470  double detJ = mJ(0,0)*mJ(1,1)-mJ(0,1)*mJ(1,0);
471 
472  mJinv(0,0) = mJ(1,1); mJinv(0,1) = -mJ(0,1);
473  mJinv(1,0) = -mJ(1,0); mJinv(1,1) = mJ(0,0);
474 
475  bounded_matrix<double,2,2> check;
476 
477 
478  if(detJ < 1e-12)
479  {
480  //std::cout << "detJ = " << detJ << std::endl;
482  pgeom[0].GetSolutionStepValue(IS_BOUNDARY) = 1;
483  pgeom[1].GetSolutionStepValue(IS_BOUNDARY) = 1;
484  pgeom[2].GetSolutionStepValue(IS_BOUNDARY) = 1;
485  return false;
486  }
487 
488  else
489  {
490 
491  double x0_2 = x0*x0 + y0*y0;
492  double x1_2 = x1*x1 + y1*y1;
493  double x2_2 = x2*x2 + y2*y2;
494 
495  //finalizing the calculation of the inverted matrix
496  //std::cout<<"MATR INV"<<MatrixInverse(mJ)<<std::endl;
497  mJinv /= detJ;
498  //calculating the RHS
499  mRhs[0] = (x1_2 - x0_2);
500  mRhs[1] = (x2_2 - x0_2);
501 
502  //calculate position of the center
503  noalias(mC) = prod(mJinv,mRhs);
504 
505  double radius = sqrt(pow(mC[0]-x0,2)+pow(mC[1]-y0,2));
506 
507  //calculate average h
508  double h;
509  h = pgeom[0].FastGetSolutionStepValue(NODAL_H);
510  h += pgeom[1].FastGetSolutionStepValue(NODAL_H);
511  h += pgeom[2].FastGetSolutionStepValue(NODAL_H);
512  h *= 0.333333333;
513  if (radius < h*alpha_param)
514  {
515  return true;
516  }
517  else
518  {
519  return false;
520  }
521  }
