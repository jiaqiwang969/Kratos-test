165  // The power iteration algorithm
166  void Solve(SparseMatrixType& K,
167  SparseMatrixType& M,
168  DenseVectorType& Eigenvalues,
169  DenseMatrixType& Eigenvectors)
170  {
171 
172  using boost::numeric::ublas::trans;
173 
174  SizeType size = K.size1();
175  SizeType max_iteration = BaseType::GetMaxIterationsNumber();
176  double tolerance = BaseType::GetTolerance();
177 
178  VectorType x = ZeroVector(size);
179  VectorType y = ZeroVector(size);
180 
181  RandomInitialize(y);
182 
183  if(Eigenvalues.size() < 1)
184  Eigenvalues.resize(1,0.00);
185 
186 
187  // Starting with first step
188  double beta = 0.00;
189  double ro = 0.00;
190  double old_ro = Eigenvalues[0];
191  std::cout << "iteration beta \t\t ro \t\t convergence norm" << std::endl;
192  for(SizeType i = 0 ; i < max_iteration ; i++)
193  {
194  //K*x = y
195  mpLinearSolver->Solve(K,x,y);
196 
197  ro = inner_prod(y,x);
198 
199  //y = M*x
200  noalias(y) = prod(M,x);
201 
202  beta = inner_prod(x, y);
203  if(beta <= 0.00)
204  KRATOS_ERROR(std::invalid_argument, "M is not Positive-definite", "");
205 
206  ro = ro / beta;
207  beta = sqrt(beta);
208 
209  double inverse_of_beta = 1.00 / beta;
210 
211  y *= inverse_of_beta;
212 
213  if(ro == 0.00)
214  KRATOS_ERROR(std::runtime_error, "Perpendicular eigenvector to M", "");
215 
216  double convergence_norm = fabs((ro - old_ro) / ro);
217 
218  std::cout << i << " \t " << beta << " \t " << ro << " \t " << convergence_norm << std::endl;
219  //std::cout << "i = " << i << ": beta = " << beta << ", ro = " << ro << ", convergence norm = " << convergence_norm << std::endl;
220 
221  if(convergence_norm < tolerance)
222  break;
223 
224  old_ro = ro;
225 
226 
227 
228  }
229 
230 KRATOS_WATCH(ro);
231 //KRATOS_WATCH(y);
232 
233  Eigenvalues[0] = ro;
234 
235  if((Eigenvectors.size1() < 1) || (Eigenvectors.size2() < size))
236  Eigenvectors.resize(1,size);
237 
238  for(SizeType i = 0 ; i < size ; i++)
239  Eigenvectors(0,i) = y[i];
240  }
