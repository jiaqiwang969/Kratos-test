881  virtual double ShapeFunctionValue( IndexType ShapeFunctionIndex,
882  const CoordinatesArrayType& rPoint) const
883  {
884 
885  double thirdCoord= 1-rPoint[0]-rPoint[1];
886  switch( ShapeFunctionIndex )
887  {
888  case 0:
889  return( thirdCoord*(2*thirdCoord-1) );
890  case 1:
891  return( rPoint[0]*(2*rPoint[0]-1) );
892  case 2:
893  return( rPoint[1]*(2*rPoint[1]-1) );
894  case 3:
895  return( 4*thirdCoord*rPoint[0]);
896  case 4:
897  return( 4*rPoint[0]*rPoint[1] );
898  case 5:
899  return( 4*rPoint[1]*thirdCoord);
900 
901  default:
902  KRATOS_ERROR(std::logic_error,
903  "Wrong index of shape function!" ,
904  *this);
905  }
906  return 0;

925  virtual ShapeFunctionsGradientsType& ShapeFunctionsIntegrationPointsGradients(

1099  virtual Matrix& ShapeFunctionsLocalGradients( Matrix& rResult,
1100  const CoordinatesArrayType& rPoint ) const
1101  {
1102  rResult.resize(6,2);
1103  double thirdCoord= 1-rPoint[0]-rPoint[1];
1104  double thirdCoord_DX= -1;
1105  double thirdCoord_DY= -1;
1106 
1107  noalias(rResult) = ZeroMatrix(6,2);
1108  rResult(0,0) = (4*thirdCoord-1)*thirdCoord_DX;
1109  rResult(0,1) = (4*thirdCoord-1)*thirdCoord_DY;
1110  rResult(1,0) = 4*rPoint[0]-1;
1111  rResult(1,1) = 0;
1112  rResult(2,0) = 0;
