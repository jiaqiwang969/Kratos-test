795  virtual Matrix& InverseOfJacobian( Matrix& rResult,
796  IndexType IntegrationPointIndex,
797  IntegrationMethod ThisMethod ) const
798  {
799  //current jacobian
800  Matrix tempMatrix = ZeroMatrix(2,2);
801  tempMatrix = Jacobian( tempMatrix, IntegrationPointIndex, ThisMethod );
802  //determinant of jacobian
803  double det_j = DeterminantOfJacobian( IntegrationPointIndex, ThisMethod );
804  //checking for singularity
805  if(det_j == 0.00)
806  KRATOS_ERROR(std::runtime_error,
807  "Zero determinant of jacobian during inversion of matrix!" ,
808  *this);
809  //setting up result matrix
810  rResult.resize(2,2);
811  //filling matrix
812  rResult(0,0) = (tempMatrix(1,1))/(det_j);
813  rResult(1,0) = -(tempMatrix(1,0))/(det_j);
814  rResult(0,1) = -(tempMatrix(0,1))/(det_j);
815  rResult(1,1) = (tempMatrix(0,0))/(det_j);
816  return rResult;
817  }

966  CalculateShapeFunctionsIntegrationPointsLocalGradients(ThisMethod);
967  //getting the inverse jacobian matrices

970  //loop over all integration points
971  for( unsigned int pnt=0; pnt<integration_points_number; pnt++ )

975  {

978  rResult[pnt](i,j) =
979  (locG[pnt](i,0)*invJ[pnt](j,0))
