481 // //
482 // // Purpose:
483 // //
484 // // MGMRES applies the restarted GMRES iteration to a linear system.
485 // //
486 // // Discussion:
487 // //
488 // // The linear system A*X=B is solved iteratively.
489 // //
490 // // The matrix A is assumed to be sparse. To save on storage, only
491 // // the nonzero entries of A are stored. For instance, the K-th nonzero
492 // // entry in the matrix is stored by:
493 // //
494 // // A(K) = value of entry,
495 // // IA(K) = row of entry,
496 // // JA(K) = column of entry.
497 // //
498 // // The "matrices" H and V are treated as one-dimensional vectors
499 // // which store the matrix data in row major form.
500 // //
501 // // This requires that references to H[I][J] be replaced by references
502 // // to H[I*MR+J] and references to V[I][J] by V[I*N+J].
503 // //
504 // // Modified:
505 // //
506 // // 30 October 2006
507 // //
508 // // Author:
509 // //
510 // // Lili Ju
511 // //
512 // // C++ translation by John Burkardt
513 // //
514 // // Reference:
515 // //
516 // // Richard Barrett, Michael Berry, Tony Chan, James Demmel,
517 // // June Donato, Jack Dongarra, Victor Eijkhout, Roidan Pozo,
518 // // Charles Romine, Henk van der Vorst,
519 // // Templates for the Solution of Linear Systems:
520 // // Building Blocks for Iterative Methods,
521 // // SIAM, 1994.
522 // //
523 // // Yousef Saad,
524 // // Iterative Methods for Sparse Linear Systems,
525 // // Second Edition,
526 // // SIAM, 20003,
527 // // ISBN: 0898715342.
528 // //
529 // // Parameters:
530 // //
531 // // Input, double A[NZ_NUM], the matrix values.
532 // //
533 // // Input, int IA[NZ_NUM], JA[NZ_NUM], the row and column indices
534 // // of the matrix values.
535 // //
536 // // Input/output, double X[N]; on input, an approximation to
537 // // the solution. On output, an improved approximation.
538 // //
539 // // Input, double RHS[N], the right hand side of the linear system.
540 // //
541 // // Input, int N, the order of the linear system.
542 // //
543 // // Input, int NZ_NUM, the number of nonzero matrix values.
544 // //
545 // // Input, int ITRMAX, the maximum number of (outer) iterations to take.
546 // //
547 // // Input, int MR, the maximum number of (inner) iterations to take.
548 // // MR must be less than N.
549 // //
550 // // Input, double TOL, a tolerance.
551 // //
552  bool mgmres ( long double a[], int ia[], int ja[],long double x[],long double rhs[],
553  int n, int nz_num, int itrmax, int mr, long double tol )
554  {
555  long double av;
556  long double *c;
557  long double delta = 1.0e-3;
558  long double *g;
559  long double *h;
560  long double htmp;
561  int itr;
562  int i;
563  int j;
564  int k;
565  long double mu;
566  long double *r;
567  long double rho;
568  long double rho_tol;
569  long double *s;
570  long double *v;
571  long double *y;
572  bool isConverged= false;
573 
574  c = new long double[mr+1];
575  g = new long double[mr+1];
576  h = new long double[(mr+1)*mr];
577  r = new long double[n];
578  s = new long double[mr+1];
579  v = new long double[(mr+1)*n];
580  y = new long double[mr+1];
581 
582  for (itr = 0; itr < itrmax; itr++ )
583  {
584  ax ( a, ia, ja, x, r, n, nz_num );
585 
586  for ( i = 0; i < n; i++ )
587  {
588  r[i] = rhs[i] - r[i];
589  }
590 
591  rho = sqrt ( dot_product ( r, r, n ) );
592 
593  if ( itr == 0 )
594  {
595  rho_tol = rho * tol;
596  }
597 
598  for (i = 0; i < n; i++)
599  {
600  v[0*n+i] = r[i] / rho;
601  }
602 
603  for ( i = 0; i <= mr; i++ )
604  {
605  g[i] = 0.0;
606  for ( j = 0; j < mr; j++ )
607  {
608  h[i*mr+j] = 0.0;
609  }
610  }
611 
612  g[0] = rho;
613  k = 0;
614 
615  while ( ( rho_tol < rho ) && ( k < mr ) )
616  {
617  ax ( a, ia, ja, v+k*n, v+(k+1)*n, n, nz_num );
618  av = sqrt ( dot_product ( v+(k+1)*n, v+(k+1)*n, n ) );
619 
620  for ( j = 0; j <= k; j++ )
621  {
622  h[j*mr+k] = dot_product ( v+(k+1)*n, v+j*n, n );
623  for ( i = 0; i < n; i++ )
624  {
625  v[(k+1)*n+i] = v[(k+1)*n+i] - h[j*mr+k] * v[j*n+i];
626  }
627  }
628 
629  h[(k+1)*mr+k] = sqrt ( dot_product ( v+(k+1)*n, v+(k+1)*n, n ) );
630 
631  if ( ( av + delta * h[(k+1)*mr+k] ) == av )
632  {
633  for ( j = 0; j <= k; j++ )
634  {
635  htmp = dot_product ( v+(k+1)*n, v+j*n, n );
636  h[j*mr+k] = h[j*mr+k] + htmp;
637  for ( i = 0; i < n; i++ )
638  {
639  v[(k+1)*n+i] = v[(k+1)*n+i] - htmp * v[j*n+i];
640  }
641  }
642  h[(k+1)*mr+k] = sqrt ( dot_product ( v+(k+1)*n, v+(k+1)*n, n ) );
643  }
644 
645  for ( i = 0; i < n; i++ )
646  {
647  v[(k+1)*n+i] = v[(k+1)*n+i] / h[(k+1)*mr+k];
648  }
649 
650  if ( 0 < k )
651  {
652  for ( i = 0; i <= k+1; i++ )
653  {
654  y[i] = h[i*mr+k];
655  }
656  for ( j = 0; j < k; j++ )
657  {
658  mult_givens ( c[j], s[j], j, y );
659  }
660  for ( i = 0; i <= k+1; i++ )
661  {
662  h[i*mr+k] = y[i];
663  }
664  }
665  mu = sqrt ( pow ( h[k*mr+k], 2 ) + pow ( h[(k+1)*mr+k], 2 ) );
666  c[k] = h[k*mr+k] / mu;
667  s[k] = -h[(k+1)*mr+k] / mu;
668  h[k*mr+k] = c[k] * h[k*mr+k] - s[k] * h[(k+1)*mr+k];
669  h[(k+1)*mr+k] = 0;
670  mult_givens ( c[k], s[k], k, g );
671  rho = fabs ( g[k] );
672  k = k + 1;
673  }
674  k = k-1;
675  y[k] = g[k] / h[k*mr+k];
676 
677  for ( i = k-1; 0 <= i; i-- )
678  {
679  y[i] = g[i];
680  for ( j = k; i < j; j-- )
681  {
682  y[i] = y[i] - h[i*mr+j] * y[j];
683  }
684  y[i] = y[i] / h[i*mr+i];
685  }
686 
687  for ( i = 0; i < n; i++ )
688  {
689  for ( j = 0; j < k; j++ )
690  {
691  x[i] = x[i] + v[j*n+i] * y[j];
692  }
693  }
694 
695  if ( rho <= rho_tol )
696  {
697  isConverged= true;
698  break;
699  }
700  }
701 
702  BaseType::SetIterationsNumber(( itr + 1 ) * mr);
703  BaseType::SetResidualNorm(rho);
704 
705  delete [] c;
706  delete [] g;
707  delete [] h;
708  delete [] r;
709  delete [] s;
710  delete [] v;
711  delete [] y;
712 
713  return isConverged;
714  }

746 // //
747 // // Parameters:
748 // //
749 // // Input, double C, S, the cosine and sine of a Givens
750 // // rotation.
751 // //
752 // // Input, int K, indicates the location of the first vector entry.
753 // //
754 // // Input/output, double G[K+2], the vector to be modified. On output,
755 // // the Givens rotation has been applied to entries G(K) and G(K+1).

757  void mult_givens ( long double c,long double s, int k, long double g[] )
758  {
759  long double g1;
760  long double g2;
761 
762  g1 = c * g[k] - s * g[k+1];
763  g2 = s * g[k] + c * g[k+1];
764 
765  g[k] = g1;
766  g[k+1] = g2;
767 
768  return;
769  }
