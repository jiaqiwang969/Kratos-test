116  //input mesh
117  in.numberofpoints = ThisModelPart.Nodes().size();
118  in.pointlist = (REAL *) malloc(in.numberofpoints * 2 * sizeof(REAL));
119  //writing the points coordinates in a vector
120  ModelPart::NodesContainerType::iterator nodes_begin = ThisModelPart.NodesBegin();
121  for(unsigned int i = 0; i<ThisModelPart.Nodes().size(); i++)
122  {
123  int base = i*2;
124  //int base = ((nodes_begin + i)->Id() - 1 ) * 2;
125 
126  //from now on it is consecutive
127  (nodes_begin + i)->Id() = i+1;
128 
129  in.pointlist[base] = (nodes_begin + i)->X();
130  in.pointlist[base+1] = (nodes_begin + i)->Y();
131  }
132 
133  in.numberoftriangles = ThisModelPart.Elements().size();
134  in.trianglelist = (int*) malloc(in.numberoftriangles * 3 * sizeof(int));
135  //copying the elements in the input file
136  ModelPart::ElementsContainerType::iterator elem_begin = ThisModelPart.ElementsBegin();
137  for(unsigned int el = 0; el<ThisModelPart.Elements().size(); el++)
138  {
139  int base = el * 3;
140

142 
143 
144  in.trianglelist[base] = geom[0].Id();

146  in.trianglelist[base+2] = geom[2].Id();
147  }
148 
149  //clearing elements

151 
152  KRATOS_WATCH(in.numberofsegments);

155  KRATOS_WATCH(in.numberofholes);
156 
157  //read and regenerate the existing mesh ... to generate the boundaries
158  char options[] = "rcEBQ";
159  triangulate(options, &in, &mid, &vorout);
160 
161  //free the memory used in the first step
162  free( in.pointlist );
163  free( in.trianglelist );
164 
165  //uses the boundary list generated at the previous step to generate the "skin"
166  mid.numberoftriangles = 0;
167  free(mid.trianglelist);

169  triangulate(regeneration_option, &mid, &out, &vorout);
170 
171  KRATOS_WATCH(out.numberofsegments);
172  KRATOS_WATCH(out.numberofpoints);
173  KRATOS_WATCH(out.numberoftriangles);
174  KRATOS_WATCH(out.numberofholes);
175 
176  //free the memory used in the intermediate step
177  free( mid.pointlist );

183 
184  //generate Kratos triangle

186 
187  //note that the node list can not be changed starting from here
188  ModelPart::NodesContainerType& ModelNodes = ThisModelPart.Nodes();
189 
190  //generate kratos elements (conditions are not touched)

192  {

199  temp.push_back( *((ModelNodes).find( out.trianglelist[base+2]).base() ) );
200 
201  Element::Pointer p_element = rReferenceElement.Create(id, temp, properties);
202  ThisModelPart.Elements().push_back(p_element);
203  }
