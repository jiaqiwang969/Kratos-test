142  ResidualBasedNewmarkScheme(double mDissipationRadius ): Scheme<TSparseSpace,TDenseSpace>()
143  {
144  mElementalDofList.resize(5);
145 
146  mAlpha= mDissipationRadius/(1+mDissipationRadius);
147 
148  mAlpha_m= (2*mDissipationRadius-1)/(mDissipationRadius+1);
149 
150  mBeta= (1+mAlpha-mAlpha_m)*(1+mAlpha-mAlpha_m)/4;
151 
152  mGamma= 0.5+mAlpha-mAlpha_m;
153 
154  std::cout << "using the Newmark Time Integration Scheme with radius= "<< mDissipationRadius <<" alpha_f= "<<mAlpha <<" alpha_m= "<<mAlpha_m<<" beta= "<<mBeta<<" gamma= "<<mGamma<< std::endl;

242  void InitializeNonLinIteration(ModelPart& r_model_part,
243  TSystemMatrixType& A,
244  TSystemVectorType& Dx,
245  TSystemVectorType& b )
246  {
247  KRATOS_TRY
248  ProcessInfo CurrentProcessInfo= r_model_part.GetProcessInfo();
249  //Update nodal values and nodal velocities at mAlpha
250  for(ModelPart::NodeIterator i = r_model_part.NodesBegin() ;
251  i != r_model_part.NodesEnd() ; i++)
252  {
253  if( i->HasDofFor(DISPLACEMENT_X) )
254  {
255  i->GetSolutionStepValue(ACCELERATION_EINS_X)
256  = 1/(mBeta*CurrentProcessInfo[DELTA_TIME]
257  *CurrentProcessInfo[DELTA_TIME])
258  * (i->GetSolutionStepValue(DISPLACEMENT_EINS_X)
259  -i->GetSolutionStepValue(DISPLACEMENT_NULL_X))
260  -1/(mBeta*CurrentProcessInfo[DELTA_TIME])
261  *i->GetSolutionStepValue(DISPLACEMENT_NULL_DT_X)
262  -(1-2*mBeta)/(2*mBeta)*i->GetSolutionStepValue(ACCELERATION_NULL_X);
263 
264  i->GetSolutionStepValue(DISPLACEMENT_EINS_DT_X)
265  = (i->GetSolutionStepValue(DISPLACEMENT_EINS_X)
266  -i->GetSolutionStepValue(DISPLACEMENT_NULL_X))
267  *mGamma/(mBeta*CurrentProcessInfo[DELTA_TIME])
268  -(mGamma-mBeta)/mBeta*(i->GetSolutionStepValue(DISPLACEMENT_NULL_DT_X))
269  -(mGamma-2*mBeta)/(2*mBeta)*CurrentProcessInfo[DELTA_TIME]
270  *(i->GetSolutionStepValue(ACCELERATION_NULL_X));
271 
272  i->GetSolutionStepValue(ACCELERATION_X)
273  = mAlpha_m*i->GetSolutionStepValue(ACCELERATION_NULL_X)
274  +(1.0-mAlpha_m)*i->GetSolutionStepValue(ACCELERATION_EINS_X);
275 
276  i->GetSolutionStepValue(DISPLACEMENT_DT_X)
277  = mAlpha*i->GetSolutionStepValue(DISPLACEMENT_NULL_DT_X)
278  +(1.0-mAlpha)*i->GetSolutionStepValue(DISPLACEMENT_EINS_DT_X);
279 
280  i->GetSolutionStepValue(DISPLACEMENT_X)
281  = mAlpha*i->GetSolutionStepValue(DISPLACEMENT_NULL_X)
282  +(1.0-mAlpha)*i->GetSolutionStepValue(DISPLACEMENT_EINS_X);
283 
284  }
285  if( i->HasDofFor(DISPLACEMENT_Y) )
286  {
287  i->GetSolutionStepValue(ACCELERATION_EINS_Y)
288  = 1/(mBeta*CurrentProcessInfo[DELTA_TIME]*CurrentProcessInfo[DELTA_TIME])
289  * (i->GetSolutionStepValue(DISPLACEMENT_EINS_Y)
290  -i->GetSolutionStepValue(DISPLACEMENT_NULL_Y))
291  -1/(mBeta*CurrentProcessInfo[DELTA_TIME])
292  *i->GetSolutionStepValue(DISPLACEMENT_NULL_DT_Y)
293  -(1-2*mBeta)/(2*mBeta)*
294  i->GetSolutionStepValue(ACCELERATION_NULL_Y);
295 
296  i->GetSolutionStepValue(DISPLACEMENT_EINS_DT_Y)
297  = (i->GetSolutionStepValue(DISPLACEMENT_EINS_Y)
298  -i->GetSolutionStepValue(DISPLACEMENT_NULL_Y))
299  *mGamma/(mBeta*CurrentProcessInfo[DELTA_TIME])
300  -(mGamma-mBeta)/mBeta*(i->GetSolutionStepValue(DISPLACEMENT_NULL_DT_Y))
301  -(mGamma-2*mBeta)/(2*mBeta)*CurrentProcessInfo[DELTA_TIME]
302  *(i->GetSolutionStepValue(ACCELERATION_NULL_Y));
303 
304  i->GetSolutionStepValue(ACCELERATION_Y)
305  = mAlpha_m*i->GetSolutionStepValue(ACCELERATION_NULL_Y)
306  +(1.0-mAlpha_m)*i->GetSolutionStepValue(ACCELERATION_EINS_Y);
307 
308  i->GetSolutionStepValue(DISPLACEMENT_DT_Y)
309  = mAlpha*i->GetSolutionStepValue(DISPLACEMENT_NULL_DT_Y)
310  +(1.0-mAlpha)*i->GetSolutionStepValue(DISPLACEMENT_EINS_DT_Y);
311 
312  i->GetSolutionStepValue(DISPLACEMENT_Y)
313  = mAlpha*i->GetSolutionStepValue(DISPLACEMENT_NULL_Y)+(1.0-mAlpha)*
314  i->GetSolutionStepValue(DISPLACEMENT_EINS_Y);
315  }
316  if( i->HasDofFor(DISPLACEMENT_Z) )
317  {
318  i->GetSolutionStepValue(ACCELERATION_EINS_Z)
319  = 1/(mBeta*CurrentProcessInfo[DELTA_TIME]*CurrentProcessInfo[DELTA_TIME])
320  * (i->GetSolutionStepValue(DISPLACEMENT_EINS_Z)
321  -i->GetSolutionStepValue(DISPLACEMENT_NULL_Z))
322  -1/(mBeta*CurrentProcessInfo[DELTA_TIME])
323  *i->GetSolutionStepValue(DISPLACEMENT_NULL_DT_Z)
324  -(1-2*mBeta)/(2*mBeta)*i->GetSolutionStepValue(ACCELERATION_NULL_Z);
325 
326  i->GetSolutionStepValue(DISPLACEMENT_EINS_DT_Z)
327  = (i->GetSolutionStepValue(DISPLACEMENT_EINS_Z)
328  -i->GetSolutionStepValue(DISPLACEMENT_NULL_Z))
329  *mGamma/(mBeta*CurrentProcessInfo[DELTA_TIME])-(mGamma-mBeta)/mBeta*
330  (i->GetSolutionStepValue(DISPLACEMENT_NULL_DT_Z))
331  -(mGamma-2*mBeta)/(2*mBeta)*CurrentProcessInfo[DELTA_TIME]
332  *(i->GetSolutionStepValue(ACCELERATION_NULL_Z));
333 
334  i->GetSolutionStepValue(ACCELERATION_Z)
335  = mAlpha_m*i->GetSolutionStepValue(ACCELERATION_NULL_Z)
336  +(1.0-mAlpha_m)*i->GetSolutionStepValue(ACCELERATION_EINS_Z);
337 
338  i->GetSolutionStepValue(DISPLACEMENT_DT_Z)
339  = mAlpha*i->GetSolutionStepValue(DISPLACEMENT_NULL_DT_Z)
340  +(1.0-mAlpha)*i->GetSolutionStepValue(DISPLACEMENT_EINS_DT_Z);
341 
342  i->GetSolutionStepValue(DISPLACEMENT_Z)
343  = mAlpha*i->GetSolutionStepValue(DISPLACEMENT_NULL_Z)
344  +(1.0-mAlpha)*i->GetSolutionStepValue(DISPLACEMENT_EINS_Z);
345  }
346  if( i->HasDofFor(WATER_PRESSURE) )
347  {
348  i->GetSolutionStepValue(WATER_PRESSURE_EINS_ACCELERATION)
349  = 1/(mBeta*CurrentProcessInfo[DELTA_TIME]*CurrentProcessInfo[DELTA_TIME])
350  * (i->GetSolutionStepValue(WATER_PRESSURE_EINS)
351  -i->GetSolutionStepValue(WATER_PRESSURE_NULL))
352  -1/(mBeta*CurrentProcessInfo[DELTA_TIME])
353  *i->GetSolutionStepValue(WATER_PRESSURE_NULL_DT)
354  -(1-2*mBeta)/(2*mBeta)*i->GetSolutionStepValue(WATER_PRESSURE_NULL_ACCELERATION);
355 
356  i->GetSolutionStepValue(WATER_PRESSURE_EINS_DT)
357  = (i->GetSolutionStepValue(WATER_PRESSURE_EINS)
358  -i->GetSolutionStepValue(WATER_PRESSURE_NULL))
359  *mGamma/(mBeta*CurrentProcessInfo[DELTA_TIME])
360  -(mGamma-mBeta)/mBeta*(i->GetSolutionStepValue(WATER_PRESSURE_NULL_DT))
361  -(mGamma-2*mBeta)/(2*mBeta)*CurrentProcessInfo[DELTA_TIME]
362  *(i->GetSolutionStepValue(WATER_PRESSURE_NULL_ACCELERATION));
363 
364  i->GetSolutionStepValue(WATER_PRESSURE_ACCELERATION)
365  = mAlpha_m*i->GetSolutionStepValue(WATER_PRESSURE_NULL_ACCELERATION)
366  +(1.0-mAlpha_m)*i->GetSolutionStepValue(WATER_PRESSURE_EINS_ACCELERATION);
367 
368  i->GetSolutionStepValue(WATER_PRESSURE_DT)
369  = mAlpha*i->GetSolutionStepValue(WATER_PRESSURE_NULL_DT)
370  +(1.0-mAlpha)*i->GetSolutionStepValue(WATER_PRESSURE_EINS_DT);
371 
372  i->GetSolutionStepValue(WATER_PRESSURE)
373  = mAlpha*i->GetSolutionStepValue(WATER_PRESSURE_NULL)
374  +(1.0-mAlpha)*i->GetSolutionStepValue(WATER_PRESSURE_EINS);
375  }
376  if( i->HasDofFor(AIR_PRESSURE) )
377  {
378  i->GetSolutionStepValue(AIR_PRESSURE_EINS_ACCELERATION)
379  = 1/(mBeta*CurrentProcessInfo[DELTA_TIME]*CurrentProcessInfo[DELTA_TIME])
380  * (i->GetSolutionStepValue(AIR_PRESSURE_EINS)
381  -i->GetSolutionStepValue(AIR_PRESSURE_NULL))
382  -1/(mBeta*CurrentProcessInfo[DELTA_TIME])
383  *i->GetSolutionStepValue(AIR_PRESSURE_NULL_DT)
384  -(1-2*mBeta)/(2*mBeta)*i->GetSolutionStepValue(AIR_PRESSURE_NULL_ACCELERATION);
385 
386  i->GetSolutionStepValue(AIR_PRESSURE_EINS_DT)
387  = (i->GetSolutionStepValue(AIR_PRESSURE_EINS)
388  -i->GetSolutionStepValue(AIR_PRESSURE_NULL))
389  *mGamma/(mBeta*CurrentProcessInfo[DELTA_TIME])
390  -(mGamma-mBeta)/mBeta*
391  (i->GetSolutionStepValue(AIR_PRESSURE_NULL_DT))
392  -(mGamma-2*mBeta)/(2*mBeta)*CurrentProcessInfo[DELTA_TIME]
393  *(i->GetSolutionStepValue(AIR_PRESSURE_NULL_ACCELERATION));
394 
395  i->GetSolutionStepValue(AIR_PRESSURE_ACCELERATION)
396  = mAlpha_m*i->GetSolutionStepValue(AIR_PRESSURE_NULL_ACCELERATION)
397  +(1.0-mAlpha_m)*i->GetSolutionStepValue(AIR_PRESSURE_EINS_ACCELERATION);
398 
399  i->GetSolutionStepValue(AIR_PRESSURE_DT)
400  = mAlpha*i->GetSolutionStepValue(AIR_PRESSURE_NULL_DT)
401  +(1.0-mAlpha)*i->GetSolutionStepValue(AIR_PRESSURE_EINS_DT);
402 
403  i->GetSolutionStepValue(AIR_PRESSURE)
404  = mAlpha*i->GetSolutionStepValue(AIR_PRESSURE_NULL)
405  +(1.0-mAlpha)*i->GetSolutionStepValue(AIR_PRESSURE_EINS);
406  }
407  }
408 
409  //For total Lagrangian
410  for(ModelPart::NodeIterator i = r_model_part.NodesBegin() ;
411  i != r_model_part.NodesEnd() ; ++i)
412  {
413  (i)->X() = (i)->X0() + i->GetSolutionStepValue(DISPLACEMENT_X);
414  (i)->Y() = (i)->Y0() + i->GetSolutionStepValue(DISPLACEMENT_Y);
415  (i)->Z() = (i)->Z0() + i->GetSolutionStepValue(DISPLACEMENT_Z);
416  }
417  KRATOS_CATCH("")
418  }
