148  //
149 
150  double havg = 0.00;
151  double h_nodes = 0;
152 
153  for(ModelPart::NodesContainerType::iterator in = rNodes.begin(); in!=rNodes.end(); in++)
154  {
155  if((in->GetValue(NEIGHBOUR_NODES)).size() != 0)
156  {
157  double xc = in->X();
158  double yc = in->Y();
159  double zc = in->Z();
160 
161  double h = 1000.0;
162  for( WeakPointerVector< Node<3> >::iterator i = in->GetValue(NEIGHBOUR_NODES).begin();
163  i != in->GetValue(NEIGHBOUR_NODES).end(); i++)
164  {
165  double x = i->X();
166  double y = i->Y();
167  double z = i->Z();
168  double l = (x-xc)*(x-xc);
169  l += (y-yc)*(y-yc);
170  l += (z-zc)*(z-zc);
171 
172  //if(l>h) h = l;
173  if(l<h) h = l;
174  }
175  h = sqrt(h);
176  havg += h;
177  h_nodes += 1;
178 
179  in->FastGetSolutionStepValue(NODAL_H) = h;
180  }
181  }
182 
183  //calculate average h
184  if(h_nodes == 0)
185  KRATOS_ERROR(std::logic_error,"no node has neighbours!!!!","");
186  havg /= h_nodes;
187 
188  //set the h to the average to the nodes without neighours
189  for(ModelPart::NodesContainerType::iterator in = rNodes.begin(); in!=rNodes.end(); in++)
190  {
191  if((in->GetValue(NEIGHBOUR_NODES)).size() == 0)
192  {
193  in->FastGetSolutionStepValue(NODAL_H) = havg;
194  }
195  }
