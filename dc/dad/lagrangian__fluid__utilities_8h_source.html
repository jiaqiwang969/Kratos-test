81  double EstimateDeltaTime(double dt_min, double dt_max, ModelPart::NodesContainerType& rNodes)
82  {
83  KRATOS_TRY
84 
85  array_1d<double,3> xc, dist;
86  double deltatime = 1000.0;
87  array_1d<double,3> vaux;
88 
89  for(ModelPart::NodesContainerType::iterator in = rNodes.begin(); in!=rNodes.end(); in++)
90  {
91  const array_1d<double,3>& vc = in->FastGetSolutionStepValue(VELOCITY);
92 
93  if((in->GetValue(NEIGHBOUR_NODES)).size() != 0)
94  {
95  xc[0] = in->X();
96  xc[1] = in->Y();
97  xc[2] = in->Z();
98 
99  double dt_node = 0.00;
100  for( WeakPointerVector< Node<3> >::iterator i = in->GetValue(NEIGHBOUR_NODES).begin();
101  i != in->GetValue(NEIGHBOUR_NODES).end(); i++)
102  {
103  dist[0] = i->X();
104  dist[1] = i->Y();
105  dist[2] = i->Z();
106  dist -= xc;
107 
108  noalias(vaux) = i->FastGetSolutionStepValue(VELOCITY);
109  noalias(vaux) -= vc;
110  double vnorm2 = inner_prod(vaux,vaux);
111 
112  if(vnorm2 != 0.0)
113  {
114  double nom = inner_prod(dist,vaux);
115  double dt_side = fabs(nom/vnorm2);
116  if(dt_side > dt_node)
117  dt_node = dt_side;
118  else if(dt_node < deltatime)
119  deltatime = dt_node;
120  }
121  }
122  }
123  }
124 
125  if(deltatime > dt_max) deltatime = dt_max;
126  else if (deltatime < dt_min) deltatime = dt_min;
127 
128  return deltatime;
129 
130  KRATOS_CATCH("")
131  }
