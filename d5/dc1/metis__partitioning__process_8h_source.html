331  int GraphColoring(GraphType& rDomainsGraph, GraphType& rDomainsColoredGraph)
332  {
333  int max_color = 0;
334  // Initializing the coloered graph. -1 means no connection
335  rDomainsColoredGraph = ScalarMatrix(mNumberOfPartitions, mNumberOfPartitions, -1.00);
336 
337  // Start coloring...
338  for(SizeType i = 0 ; i < rDomainsGraph.size1() ; i++) // for each domain
339  for(SizeType j = i + 1 ; j < rDomainsGraph.size2() ; j++) // finding neighbor domains
340  if(rDomainsGraph(i,j) != 0.00) // domain i has interface with domain j
341  for(SizeType color = 0 ; color < rDomainsColoredGraph.size2() ; color++) // finding color
342  if((rDomainsColoredGraph(i,color) == -1.00) && (rDomainsColoredGraph(j,color) == -1.00)) // the first unused color
343  {
344  rDomainsColoredGraph(i,color) = j;
345  rDomainsColoredGraph(j,color) = i;
346  if(max_color < static_cast<int>(color + 1))
347  max_color = color + 1;
348  break;
349  }
350  return max_color;
351  }

459  void CallingMetis(SizeType NumberOfNodes, SizeType NumberOfElements, IO::ConnectivitiesContainerType& ElementsConnectivities, idxtype* NPart, idxtype* EPart)
460  {
461  int rank = GetRank();
462 
463  // calculating total size of connectivity vector
464  int connectivity_size = 0;
465  for(IO::ConnectivitiesContainerType::iterator i_connectivities = ElementsConnectivities.begin() ;
466  i_connectivities != ElementsConnectivities.end() ; i_connectivities++)
467  connectivity_size += i_connectivities->size();
468 
469  int number_of_element_nodes = ElementsConnectivities.begin()->size(); // here assuming that all elements are the same!!
470 
471  int ne = NumberOfElements;
472  int nn = NumberOfNodes;
473 
474 
475 
476 
477  int etype;
478  if(number_of_element_nodes == 3) // triangles
479  etype = 1;
480  else if(number_of_element_nodes == 4) // tetrahedra or quadilateral
481  {
482  if(mDimension == 2) // quadilateral
483  etype = 4;
484  else // tetrahedra
485  etype = 2;
486  }
487  else if(number_of_element_nodes == 8) // hexahedra
488  etype = 3;
489  else
490  KRATOS_ERROR(std::invalid_argument, "invalid element type with number of nodes : ", number_of_element_nodes);
491 
492 
493 
494  int numflag = 0;
495  int number_of_partitions = static_cast<int>(mNumberOfPartitions);
496  int edgecut;
497 
498  idxtype* elmnts = new idxtype[connectivity_size];
499 
500  mLogFile << rank << " : Preparing Data for metis..." << std::endl;
501  int i = 0;
502  // Creating the elmnts array for Metis
503  for(IO::ConnectivitiesContainerType::iterator i_connectivities = ElementsConnectivities.begin() ;
504  i_connectivities != ElementsConnectivities.end() ; i_connectivities++)
505  for(unsigned int j = 0 ; j < i_connectivities->size() ; j++)
506  elmnts[i++] = (*i_connectivities)[j] - 1; // transforming to zero base indexing
507 
508  mLogFile << rank << " : Calling metis..." << std::endl;
509  // Calling Metis to partition
510  METIS_PartMeshDual(&ne, &nn, elmnts, &etype, &numflag, &number_of_partitions, &edgecut, EPart, NPart);
511  mLogFile << rank << " : Metis Finished!!!" << std::endl;
512  mLogFile << rank << " : edgecut = " << edgecut << std::endl;
513 
514  delete[] elmnts;
515 
516  }
