434  double CalculateNormDisp(ModelPart& r_model_part)
435  {
436  double result=0;
437  for(ModelPart::NodeIterator i = r_model_part.NodesBegin() ;
438  i != r_model_part.NodesEnd() ; ++i)
439  {
440  array_1d<double,3>& OldDisp = (i)->GetSolutionStepValue(DISPLACEMENT_NULL);
441  array_1d<double,3>& CurrentDisp = (i)->GetSolutionStepValue(DISPLACEMENT);
442 
443 
444  if( (i->pGetDof(DISPLACEMENT_X))->IsFixed() == false )
445  result += (CurrentDisp[0]-OldDisp[0])*(CurrentDisp[0]-OldDisp[0]);
446  if( i->pGetDof(DISPLACEMENT_Y)->IsFixed() == false )
447  result += (CurrentDisp[1]-OldDisp[1])*(CurrentDisp[1]-OldDisp[1]);
448  if( i->HasDofFor(DISPLACEMENT_Z))
449  if( i->pGetDof(DISPLACEMENT_Z)->IsFixed() == false )
450  result += (CurrentDisp[2]-OldDisp[2])*(CurrentDisp[2]-OldDisp[2]);
451  }
452  return result;
453  }
