167  array_1d<double,3> An;
168  if(dimension == 2)
169  {
170  boost::numeric::ublas::bounded_matrix<double,3,2> DN_DX;
171  array_1d<double,3> N;
172  double Volume;
173  for(ElementsArrayType::iterator it = rElements.begin(); it !=rElements.end(); it++)
174  {
175  Element::GeometryType& geom = it->GetGeometry();
176  GeometryUtils::CalculateGeometryData(geom, DN_DX, N, Volume);
177 
178  for(unsigned int i = 0; i<geom.size(); i++)
179  {
180  array_1d<double,3>& normal = geom[i].FastGetSolutionStepValue(NORMAL);
181  for(unsigned int j=0; j<2; j++)
