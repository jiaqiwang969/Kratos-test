410  bool IsConverged( int step , int lastRealCondition)

432  for( IndexType i=0; i<(*it)->GetGeometry().IntegrationPoints().size();

434  {
435 // KRATOS_WATCH((*it)->GetValue( LAMBDAS )[i]);

438  if((*it)->GetValue( GAPS )[i]>0)
439  {
440  energy_contact += ((*it)->GetValue( GAPS )[i])
441  * ((*it)->GetValue( GAPS )[i])
442  *0.5*((*it)->GetValue(PENALTY)[i]);
443  absolute += (*it)->GetValue( LAMBDAS )[i];
444  ratio += (*it)->GetValue( DELTA_LAMBDAS )[i];
445  Index++;
446  }
447  }
448  }
449  }
450 
451  if( BaseType::GetModelPart().GetProperties(1)[FRICTION_COEFFICIENT] > 0.0 )
452  {
453  friction=true;
454 
455  for (typename ConditionsArrayType::ptr_iterator it=ConditionsArray.ptr_begin() + lastRealCondition;
456  it!=ConditionsArray.ptr_end(); ++it)
457  {
458  int i=(*it)->GetValue(CONTACT_SLAVE_INTEGRATION_POINT_INDEX);
459  if((*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue(GAPS)[i] > 0)
460  {
461  Matrix m = (*it)->GetValue(CONTACT_LINK_M);
462  Matrix mInv = ZeroMatrix(2);
463  double det=0.0;
464  MathUtils<double>::InvertMatrix2( m, mInv, det );
465 
466  absolute_friction +=
467  sqrt((*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue( LAMBDAS_T )(i,0)
468  *mInv(0,0)*(*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue( LAMBDAS_T )(i,0)

473  +(*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue( LAMBDAS_T )(i,1)

475 
476  ratio_friction += sqrt((*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue( LAMBDAS_T )(i,0)
477  *mInv(0,0)*(*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue( DELTA_LAMBDAS_T )(i,0)
478  +(*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue( DELTA_LAMBDAS_T )(i,0)
479  *mInv(0,1)*(*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue( DELTA_LAMBDAS_T )(i,1)
480  +(*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue( DELTA_LAMBDAS_T )(i,1)
481  *mInv(1,0)*(*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue( DELTA_LAMBDAS_T )(i,0)
482  +(*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue( DELTA_LAMBDAS_T )(i,1)
483  *mInv(1,1)*(*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue(DELTA_LAMBDAS_T )(i,1));
484 
485  energy_friction += 0.5*
486  ((*it)->GetValue(CONTACT_LINK_SLAVE)->GetValue(PENALTY_T)[i])*
487  ((m(0,0)
488  * ((*it)->GetValue( MASTER_CONTACT_CURRENT_LOCAL_POINT)[0]
489  - (*it)->GetValue( MASTER_CONTACT_LAST_CURRENT_LOCAL_POINT)[0])
490  + m(0,1)
491  * ((*it)->GetValue( MASTER_CONTACT_CURRENT_LOCAL_POINT)[1]
492  - (*it)->GetValue( MASTER_CONTACT_LAST_CURRENT_LOCAL_POINT)[1]))*
493  (m(0,0)
494  * ((*it)->GetValue( MASTER_CONTACT_CURRENT_LOCAL_POINT)[0]
495  - (*it)->GetValue( MASTER_CONTACT_LAST_CURRENT_LOCAL_POINT)[0])
496  + m(0,1)
497  * ((*it)->GetValue( MASTER_CONTACT_CURRENT_LOCAL_POINT)[1]
498  - (*it)->GetValue( MASTER_CONTACT_LAST_CURRENT_LOCAL_POINT)[1]))
499  +
500  (m(1,0)
501  * ((*it)->GetValue( MASTER_CONTACT_CURRENT_LOCAL_POINT)[0]
502  - (*it)->GetValue( MASTER_CONTACT_LAST_CURRENT_LOCAL_POINT)[0])
503  + m(1,1)
504  * ((*it)->GetValue( MASTER_CONTACT_CURRENT_LOCAL_POINT)[1]

513  }
514  (*it)->GetValue( MASTER_CONTACT_LAST_CURRENT_LOCAL_POINT)
515  = (*it)->GetValue( MASTER_CONTACT_CURRENT_LOCAL_POINT);
516  }
517  }
518  if( this->mEchoLevel > 2 )

924  bool ClosestPoint( Condition::Pointer& Surface,
925  GeometryType::CoordinatesArrayType& rResultGlobal,
926  GeometryType::CoordinatesArrayType& rResultLocal,
927  const GeometryType::CoordinatesArrayType& rSlaveContactGlobalPoint,
928  const GeometryType::CoordinatesArrayType& rCandidateGlobal
929  )
930  {
931  double Xi1 = 0.0;
932  double Xi2 = 0.0;
933  double deltaXi1 = 0.0;
934  double deltaXi2 = 0.0;
935  Matrix localCoords;
936 // KRATOS_WATCH( Surface->GetValue( IS_CONTACT_MASTER ) );
937 // KRATOS_WATCH( Surface->GetGeometry() );
938  localCoords = (Surface->GetGeometry()).PointsLocalCoordinates( localCoords );

940  for( unsigned int n=0; n<Surface->GetGeometry().PointsNumber(); n++ )
941  {
942  if( fabs(rCandidateGlobal[0]-Surface->GetGeometry().GetPoint(n).X()) < 1e-7
943  && fabs(rCandidateGlobal[1]-Surface->GetGeometry().GetPoint(n).Y()) < 1e-7
944  && fabs(rCandidateGlobal[2]-Surface->GetGeometry().GetPoint(n).Z()) < 1e-7
945  )
946  {
947  Xi1 = localCoords(n,0);
948  Xi2 = localCoords(n,1);
949  break;
950  }
951  }
952  //setting up LocalPoint
953  rResultLocal[0] = Xi1;
954  rResultLocal[1] = Xi2;
955  rResultLocal[2] = 0.0;
956  //setting up rResult
957  rResultGlobal = rCandidateGlobal;
958 // KRATOS_WATCH( rCandidateGlobal );
959  //searching for orthogonal projection
960  for( int k=0; k<1000; k++ )
961  {
962  //setting up tangential vectors
963  Vector t1 = ZeroVector(3);//first tangential vector
964  Vector t2 = ZeroVector(3);//second tangential vector
965  //derivatives of tangential vectors
966  Vector dt11 = ZeroVector(3);
967  Vector dt12 = ZeroVector(3);
968  Vector dt21 = ZeroVector(3);
969  Vector dt22 = ZeroVector(3);
970 
971  //retrieving first order derivatives in current solution point
972  Matrix DN = ZeroMatrix(Surface->GetGeometry().PointsNumber(),2);
973  Surface->GetGeometry().ShapeFunctionsLocalGradients( DN, rResultLocal );
974  //retrieving second order derivatives in current solution point
975  GeometryType::ShapeFunctionsSecondDerivativesType D2N;
976  Surface->GetGeometry().ShapeFunctionsSecondDerivatives( D2N, rResultLocal );
977  for( unsigned int n=0; n<Surface->GetGeometry().PointsNumber(); n++ )
978  {
979  //contribution to tangential vectors
980  t1[0] += Surface->GetGeometry().GetPoint(n).X()*DN(n,0);
981  t1[1] += Surface->GetGeometry().GetPoint(n).Y()*DN(n,0);
982  t1[2] += Surface->GetGeometry().GetPoint(n).Z()*DN(n,0);

984  t2[1] += Surface->GetGeometry().GetPoint(n).Y()*DN(n,1);
985  t2[2] += Surface->GetGeometry().GetPoint(n).Z()*DN(n,1);
986  //contribution to derivatives of tangential vectors
987  dt11[0] += Surface->GetGeometry().GetPoint(n).X()*D2N[n](0,0);
988  dt11[1] += Surface->GetGeometry().GetPoint(n).Y()*D2N[n](0,0);
989  dt11[2] += Surface->GetGeometry().GetPoint(n).Z()*D2N[n](0,0);
990  dt12[0] += Surface->GetGeometry().GetPoint(n).X()*D2N[n](0,1);
991  dt12[1] += Surface->GetGeometry().GetPoint(n).Y()*D2N[n](0,1);
992  dt12[2] += Surface->GetGeometry().GetPoint(n).Z()*D2N[n](0,1);
993  dt21[0] += Surface->GetGeometry().GetPoint(n).X()*D2N[n](1,0);
994  dt21[1] += Surface->GetGeometry().GetPoint(n).Y()*D2N[n](1,0);
995  dt21[2] += Surface->GetGeometry().GetPoint(n).Z()*D2N[n](1,0);
996  dt22[0] += Surface->GetGeometry().GetPoint(n).X()*D2N[n](1,1);
997  dt22[1] += Surface->GetGeometry().GetPoint(n).Y()*D2N[n](1,1);
998  dt22[2] += Surface->GetGeometry().GetPoint(n).Z()*D2N[n](1,1);
999  }
1000  //defining auxiliary terms
1001  double A1 = ((rSlaveContactGlobalPoint[0]-rResultGlobal[0])*t1[0])
1002  +((rSlaveContactGlobalPoint[1]-rResultGlobal[1])*t1[1])
1003  +((rSlaveContactGlobalPoint[2]-rResultGlobal[2])*t1[2]);
1004  double A2 = ((rSlaveContactGlobalPoint[0]-rResultGlobal[0])*t2[0])
1005  +((rSlaveContactGlobalPoint[1]-rResultGlobal[1])*t2[1])
1006  +((rSlaveContactGlobalPoint[2]-rResultGlobal[2])*t2[2]);
1007  double B11 = (-t1[0]*t1[0]-t1[1]*t1[1]-t1[2]*t1[2])
1008  + ((rSlaveContactGlobalPoint[0]-rResultGlobal[0])*dt11[0])
1009  +((rSlaveContactGlobalPoint[1]-rResultGlobal[1])*dt11[1])
1010  +((rSlaveContactGlobalPoint[2]-rResultGlobal[2])*dt11[2]);
1011  double B12 = (-t2[0]*t1[0]-t2[1]*t1[1]-t2[2]*t1[2])
1012  + ((rSlaveContactGlobalPoint[0]-rResultGlobal[0])*dt12[0])
1013  +((rSlaveContactGlobalPoint[1]-rResultGlobal[1])*dt12[1])
1014  +((rSlaveContactGlobalPoint[2]-rResultGlobal[2])*dt12[2]);
1015  double B21 = (-t1[0]*t2[0]-t1[1]*t2[1]-t1[2]*t2[2])
1016  + ((rSlaveContactGlobalPoint[0]-rResultGlobal[0])*dt21[0])
1017  +((rSlaveContactGlobalPoint[1]-rResultGlobal[1])*dt21[1])
1018  +((rSlaveContactGlobalPoint[2]-rResultGlobal[2])*dt21[2]);
1019  double B22 = (-t2[0]*t2[0]-t2[1]*t2[1]-t2[2]*t2[2])
1020  + ((rSlaveContactGlobalPoint[0]-rResultGlobal[0])*dt22[0])
1021  +((rSlaveContactGlobalPoint[1]-rResultGlobal[1])*dt22[1])
1022  +((rSlaveContactGlobalPoint[2]-rResultGlobal[2])*dt22[2]);
1023  //calculating update for Xi
1024  deltaXi1 = -A1*B22/(B11*B22-B12*B21)+A2*B12/(B11*B22-B12*B21);
1025  deltaXi2 = A2*B21/(B11*B22-B12*B21)-A2*B11/(B11*B22-B12*B21);
1026  //updating Xi
1027  Xi1 += deltaXi1;
1028  Xi2 += deltaXi2;
1029  //updating LocalPoint
1030  rResultLocal[0] = Xi1;
1031  rResultLocal[1] = Xi2;
1032  //updating rResult
1033  rResultGlobal = ZeroVector( 3 );
1034  rResultGlobal = Surface->GetGeometry().GlobalCoordinates( rResultGlobal, rResultLocal );
1035  if( fabs(deltaXi1) < 1e-7 && fabs(deltaXi2) < 1e-7 )
1036  {
1037  //check whether contact point lies within elementary boundaries
1038  if( fabs(Xi1) <= 1.0 && fabs(Xi2) <= 1.0 )
1039  {
1040 // std::cout << "found matching point after " << k << " iteratons" << std::endl;
1041  return true;
1042  }
1043  else
1044  {
1045 // std::cout << "no matching point inside this element" << std::endl;
1046  return false;
1047  }
1048  }
1049  }
1050  return false;
