451  void InitializeSolutionStep(
452  ModelPart& r_model_part,
453  TSystemMatrixType& A,
454  TSystemVectorType& Dx,
455  TSystemVectorType& b
456  )
457  {
458  ProcessInfo& CurrentProcessInfo = r_model_part.GetProcessInfo();
459 
460  Scheme<TSparseSpace,TDenseSpace>::InitializeSolutionStep(r_model_part,A,Dx,b);
461 
462  double DeltaTime = CurrentProcessInfo[DELTA_TIME];
463 
464  if(DeltaTime == 0)
465  KRATOS_ERROR(std::logic_error,"detected delta_time = 0 in the Bossak Scheme ... check if the time step is created correctly for the current model part","");
466 
467  //initializing constants
468  ma0 = 1.0/(mGammaNewmark*DeltaTime);
469  ma1 = DeltaTime*mBetaNewmark/mGammaNewmark;
470  ma2 = (-1+mGammaNewmark)/mGammaNewmark;
471  ma3 = DeltaTime*(1-mBetaNewmark/mGammaNewmark);
472  ma4 = pow(DeltaTime,2)*(-2*mBetaNewmark+mGammaNewmark)/(2*mGammaNewmark);
473  ma5 = DeltaTime*mBetaNewmark/mGammaNewmark;
474  mam = (1.0-mAlphaBossak)/(mGammaNewmark*DeltaTime);
475  }
