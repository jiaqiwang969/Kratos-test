691  inline void CalculateCenterAndSearchRadius(const double x0, const double y0,
692  const double x1, const double y1,
693  const double x2, const double y2,
694  double& xc, double& yc, double& R
695  )
696  {
697  xc = 0.3333333333333333333*(x0+x1+x2);
698  yc = 0.3333333333333333333*(y0+y1+y2);
699 
700  double R1 = (xc-x0)*(xc-x0) + (yc-y0)*(yc-y0);
701  double R2 = (xc-x1)*(xc-x1) + (yc-y1)*(yc-y1);
702  double R3 = (xc-x2)*(xc-x2) + (yc-y2)*(yc-y2);
703 
704  R = R1;
705  if(R2 > R) R = R2;
706  if(R3 > R) R = R3;
707 
708  R = sqrt(R);
709  }

711  inline double CalculateVol( const double x0, const double y0,
712  const double x1, const double y1,
713  const double x2, const double y2
714  )
715  {
716  return 0.5*( (x1-x0)*(y2-y0)- (y1-y0)*(x2-x0) );
717  }
