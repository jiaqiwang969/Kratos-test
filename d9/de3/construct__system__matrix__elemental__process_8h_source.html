131  virtual void Execute()
132  {
133  std::size_t equation_size = TSystemSpaceType::Size1(mSystemMatrix);
134  std::vector<std::vector<std::size_t> > indices(equation_size);
135  std::vector<std::vector<std::size_t> > dirichlet_indices(TSystemSpaceType::Size1(mDirichletMatrix));
136 
137  ProcessInfo process_info;
138  for(ElementsContainerType::iterator i_element = mElements.begin() ; i_element != mElements.end() ; i_element++)
139  {
140  Element::EquationIdVectorType ids(3,0);
141 
142  i_element->EquationIdVector(ids, process_info);
143  for(std::size_t i = 0 ; i < 3 ; i++)
144  if(ids[i] < equation_size)
145  for(std::size_t j = 0 ; j < 3 ; j++)
146  if(ids[j] < equation_size)
147  indices[ids[i]].push_back(ids[j]);
148  else
149  dirichlet_indices[ids[i]].push_back(ids[j]- equation_size);
150 
151  }
152  for(std::size_t i = 0 ; i < indices.size() ; i++)
153  {
154  std::vector<std::size_t>& row_indices = indices[i];
155  std::vector<std::size_t>& dirichlet_row_indices = dirichlet_indices[i];
156  std::sort(row_indices.begin(), row_indices.end());
157  std::unique(row_indices.begin(), row_indices.end());
158  std::sort(dirichlet_row_indices.begin(), dirichlet_row_indices.end());
159  std::unique(dirichlet_row_indices.begin(), dirichlet_row_indices.end());
160  for(std::size_t j = 0 ; j < row_indices.size() ; j++)
161  mSystemMatrix()(i,row_indices[j]) = 0.00;
162  for(std::size_t j = 0 ; j < dirichlet_row_indices.size() ; j++)
163  mDirichletMatrix()(i,dirichlet_row_indices[j]) = 0.00;
164  }
165  for(std::size_t i = 0 ; i < indices.size() ; i++) // To make put at least one element per each row. Pooyan.
166  mDirichletMatrix()(i,0) = 0.00;
167  }
