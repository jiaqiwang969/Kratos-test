436  if (t <= mLength + 0.000001 && t >= -0.000001){
437  //point is above cylinder wall

453  Point<3> ClosestPoint(Point<3>& point){
454  Vector distance= point-mM;
455  double t= MathUtils<double>::Dot3(distance,
456  r3_normalized);
457  double s1= MathUtils<double>::Dot3(distance,
458  r1_normalized);
459  double s2= MathUtils<double>::Dot3(distance,
460  r2_normalized);
461 
462  Point<3> closestpoint;
463  double rad = sqrt(s1*s1+s2*s2);
464 
465  if (mLength < t){
466  //outside cylidner cap
467  if (radius < rad){
468  //outside cylinder wall and cap
469  Vector helpvector = s1 * r1_normalized + s2 * r2_normalized;
470  helpvector = helpvector / MathUtils<double>::Norm(helpvector);
471  closestpoint = mM + radius * (helpvector) + mLength*r3_normalized;
472  }
473  else
474  //only ouside cylinder cap
475  closestpoint = point + (mLength-t)*r3_normalized;
476  }
477  else if (radius < rad){
478  //only outside cylinder wall
479  Vector helpvector = s1 * r1_normalized + s2 * r2_normalized;
480  helpvector = helpvector / MathUtils<double>::Norm(helpvector);
481  closestpoint = mM + radius * (helpvector) + t*r3_normalized;
482  }
483  else{
484  //node is inside cylidner
485  if(fabs(mLength-t) < fabs(radius-rad)){
486  //node is closer to cap than on wall
487  closestpoint = point + (mLength-t)*r3_normalized;
488  }
489  else{
490  //node is closer on wall than on cap
491  Vector helpvector = s1 * r1_normalized + s2 * r2_normalized;
492  helpvector = helpvector / MathUtils<double>::Norm(helpvector);
493  closestpoint = mM + radius * (helpvector) + t*r3_normalized;
494  }
495  }
496  return closestpoint;
