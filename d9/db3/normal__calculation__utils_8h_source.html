145  //this function calculates the "area normal" (vector oriented as the normal
146  //with a dimension proportional to the area. This is done on the base of the
147  //Conditions provided
148  //which should be intended as the surface elements of the area of interest.
149  void CalculateOnSimplex(
150  ConditionsArrayType& rConditions,
151  int dimension)
152  {
153  KRATOS_TRY
154 
155  //resetting the normals
156  array_1d<double,3> zero = Vector(3);
157  noalias(zero) = ZeroVector(3);
158 
159  for(ConditionsArrayType::iterator it = rConditions.begin();
160  it !=rConditions.end(); it++)
161  {
162  Element::GeometryType& rNodes = it->GetGeometry();
163  for(unsigned int in = 0; in<rNodes.size(); in++)
164  noalias((rNodes[in]).GetSolutionStepValue(NORMAL)) = zero;
165  }
166 
167 
168  //calculating the normals and storing on the conditions
169  array_1d<double,3> An;
170  if(dimension == 2)
171  {
172  for(ConditionsArrayType::iterator it = rConditions.begin();
173  it !=rConditions.end(); it++)
174  {
175  CalculateNormal2D(it,An);
176  }
177  }
178  else if(dimension == 3)
179  {
180  array_1d<double,3> v1;
181  array_1d<double,3> v2;
182  for(ConditionsArrayType::iterator it = rConditions.begin();
183  it !=rConditions.end(); it++)
184  {
185  //calculate the normal on the given condition
186  CalculateNormal3D(it,An,v1,v2);
187  }
188  }
189 
190  //adding the normals to the nodes
191  for(ConditionsArrayType::iterator it = rConditions.begin();
192  it !=rConditions.end(); it++)
193  {
194  Geometry<Node<3> >& pGeometry = (it)->GetGeometry();
195  double coeff = 1.00/pGeometry.size();
196  const array_1d<double,3>& An = it->GetValue(NORMAL);
197 
198  for(unsigned int i = 0; i<pGeometry.size(); i++)
199  {
200  noalias(pGeometry[i].FastGetSolutionStepValue(NORMAL)) += coeff * An;
201  }
202  }
203 
204 
205  KRATOS_CATCH("")
206 
207  }
