164 
165 
166  double detJ = msJ(0,0)*msJ(1,1)-msJ(0,1)*msJ(1,0);
167 
168  msJinv(0,0) = msJ(1,1); msJinv(0,1) = -msJ(0,1);

172 
173 
174  if(detJ < 1e-12)
175  {
176  //std::cout << "detJ = " << detJ << std::endl;
178  pgeom[0].GetSolutionStepValue(IS_BOUNDARY) = 1;
179  pgeom[1].GetSolutionStepValue(IS_BOUNDARY) = 1;

182  }
183 
184  else

186
