132  static void* convertible(PyObject* obj_ptr)
133  {
134  using namespace boost::python;
135  using boost::python::allow_null; // works around gcc 2.96 bug
136  {
137  // Restriction to list, tuple, iter, xrange until
138  // Boost.Python overload resolution is enhanced.
139  if (!( PyList_Check(obj_ptr)
140  || PyTuple_Check(obj_ptr)
141  || PyIter_Check(obj_ptr)
142  || PyRange_Check(obj_ptr))) return 0;
143  }
144  handle<> obj_iter(allow_null(PyObject_GetIter(obj_ptr)));
145  if (!obj_iter.get()) { // must be convertible to an iterator
146  PyErr_Clear();
147  return 0;
148  }
149  if (true) {
150  int obj_size = PyObject_Length(obj_ptr);
151  if (obj_size < 0) { // must be a measurable sequence
152  PyErr_Clear();
153  return 0;
154  }
155  /* if (!TConversionPolicyType::check_size( */
156  /* boost::type<TTContainerTypeType>(), obj_size)) return 0; */
157  bool is_range = PyRange_Check(obj_ptr);
158  std::size_t i=0;
159  for(;;i++) {
160  handle<> py_elem_hdl(allow_null(PyIter_Next(obj_iter.get())));
161  if (PyErr_Occurred()) {
162  PyErr_Clear();
163  return 0;
164  }
165  if (!py_elem_hdl.get()) break; // end of iteration
166  object py_elem_obj(py_elem_hdl);
167  extract<vector<data_type> > elem_proxy(py_elem_obj);
168  if (!elem_proxy.check()) return 0;
169  if (is_range) break; // in a range all elements are of the same type
170  }
171  if (!is_range) assert(i == obj_size);
172  }
173  return obj_ptr;
174  }
