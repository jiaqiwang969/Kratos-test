137  //generate a list of new nodes refining accordingly to the REFINE_FLAG
138  void Execute()
139  {
140  KRATOS_TRY
141 
142  double nu = mr_model_part.GetMesh().GetProperties(1)[VISCOSITY];
143  double density = mr_model_part.GetMesh().GetProperties(1)[DENSITY];
144 
145  array_1d<double,3> aux;
146  array_1d<double,3> u_tilda;
147  double minimum_energy = 0.0001;
148  double h;
149 
150  for(ModelPart::NodeIterator in = mr_model_part.NodesBegin() ;
151  in != mr_model_part.NodesEnd() ; ++in)
152  {
153  const array_1d<double,3>& v = in->FastGetSolutionStepValue(VELOCITY);
154  const array_1d<double,3>& w = in->FastGetSolutionStepValue(MESH_VELOCITY);
155  noalias(aux) = v; noalias(aux)-=w;
156  const double nodal_area = in->FastGetSolutionStepValue(NODAL_AREA);
157  const double pressure = in->FastGetSolutionStepValue(PRESSURE);
158 
159  //calculate tau
160  double norm_u = sqrt( aux[0]*aux[0] + aux[1]*aux[1] + aux[2]*aux[2] );
161  if(mdomain_size == 2)
162  h = sqrt(nodal_area);
163  else if (mdomain_size == 3)
164  h = pow(nodal_area,0.3333333333333);
165 
166  double tau = 1.00 / ( 4.0*nu/(h*h) + 2.0*norm_u/h );
167 
168 //double aa = pow(4.0*nu/(h*h),2);
169 //double bb = pow(2.0*norm_u/(h),2);
170 //tau = sqrt(1.0/(aa+bb));
171 
172  //ESTIMATE SUBSCALE VELOCITY
173  noalias(u_tilda) = in->FastGetSolutionStepValue(PRESS_PROJ);
174  noalias(u_tilda) += in->FastGetSolutionStepValue(CONV_PROJ);
175  u_tilda *= tau;
176 
177  //calculate total fluid energy on the node
178  double energy_tot = 0.5*density*inner_prod(v,v) + fabs(pressure);
179  double energy_tilda = 0.5*density*inner_prod(u_tilda,u_tilda);
180 
181  //calculate the ratio (error)
182  double ratio = 0.0;
183  //if the energy is 0 this comparison makes no sense
184  if(energy_tot > minimum_energy)
185  ratio = energy_tilda/energy_tot;
186  //ratio = energy_tilda;
187 
188  //saving the ratio as error
189  in->FastGetSolutionStepValue(ERROR_RATIO) = ratio;
190  }
191 
192 
193  //decide which elements to refine
194  for(ModelPart::ElementIterator ie = mr_model_part.ElementsBegin() ;
195  ie != mr_model_part.ElementsEnd() ; ++ie)
196  {
197  Geometry<Node<3> >& geom = ie->GetGeometry();
198 
199  //calculate avg error
200  double avg_err = 0.00;
201  for(int i=0; i<geom.size(); i++)
202  {
203  avg_err += geom[i].FastGetSolutionStepValue(ERROR_RATIO);
204  }
205  avg_err /= double(geom.size());
206 
207  if(avg_err > mlimit_ratio)
208  ie->GetValue(REFINE_FLAG) = 1;
209 
210  }
211 
212 
213  KRATOS_CATCH("")
214  }
