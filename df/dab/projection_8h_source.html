480  inline void CalculateCenterAndSearchRadius(const double x0, const double y0,
481  const double x1, const double y1,
482  const double x2, const double y2,
483  double& xc, double& yc, double& R
484  )
485  {
486  xc = 0.3333333333333333333*(x0+x1+x2);
487  yc = 0.3333333333333333333*(y0+y1+y2);
488 
489  double R1 = (xc-x0)*(xc-x0) + (yc-y0)*(yc-y0);
490  double R2 = (xc-x1)*(xc-x1) + (yc-y1)*(yc-y1);
491  double R3 = (xc-x2)*(xc-x2) + (yc-y2)*(yc-y2);
492 
493  R = R1;
494  if(R2 > R) R = R2;
495  if(R3 > R) R = R3;
496 
497  R = 1.01 * sqrt(R);
498  }

508  inline bool CalculatePosition( const double x0, const double y0,
509  const double x1, const double y1,
510  const double x2, const double y2,
511  const double xc, const double yc,
512  array_1d<double,3>& N
513  )
514  {
515  double area = CalculateVol(x0,y0,x1,y1,x2,y2);
516  double inv_area = 0.0;
517  if(area == 0.0)
518  {
519  KRATOS_ERROR(std::logic_error,"element with zero area found","");
520  }
521  else
522  {
523  inv_area = 1.0 / area;
524  }
525 
526 
527  N[0] = CalculateVol(x1,y1,x2,y2,xc,yc) * inv_area;
528  N[1] = CalculateVol(x2,y2,x0,y0,xc,yc) * inv_area;
529  N[2] = CalculateVol(x0,y0,x1,y1,xc,yc) * inv_area;
530 
531 
532  if(N[0] >= 0.0 && N[1] >= 0.0 && N[2] >= 0.0 && N[0] <=1.0 && N[1]<= 1.0 && N[2] <= 1.0) //if the xc yc is inside the triangle return true
533  return true;
