308  bool IterativeSolve(SparseMatrixType& rA, VectorType& rX, VectorType& rB)
309  {
310  const int size = TSparseSpaceType::Size(rX);
311 
312  BaseType::mIterationsNumber = 0;
313 
314  VectorType r(size);
315 
316  PreconditionedMult(rA,rX,r);
317  TSparseSpaceType::ScaleAndAdd(1.00, rB, -1.00, r);
318 
319  BaseType::mBNorm = TSparseSpaceType::TwoNorm(rB);
320 
321  VectorType p(r);
322  VectorType s(size);
323  VectorType q(size);
324 
325  VectorType rs(r);
326  VectorType qs(size);
327 
328  double roh0 = TSparseSpaceType::Dot(r, rs);
329  double roh1 = roh0;
330  double alpha = 0.00;
331  double beta = 0.00;
332  double omega = 0.00;
333 
334 // if(roh0 < 1e-30) //we start from the real solution
335 // return BaseType::IsConverged();
336 
337  do
338  {
339  PreconditionedMult(rA,p,q);
340 
341  alpha = roh0 / TSparseSpaceType::Dot(rs,q);
342 
343  TSparseSpaceType::ScaleAndAdd(1.00, r, -alpha, q, s);
344 
345  PreconditionedMult(rA,s,qs);
346 
347  omega = TSparseSpaceType::Dot(qs,qs);
348 
349  //if(omega == 0.00)
350  if(fabs(omega) <= 1.0e-30)
351  break;
352 
353  omega = TSparseSpaceType::Dot(qs,s) / omega;
354 
355  TSparseSpaceType::ScaleAndAdd(alpha, p, 1.00, rX);
356  TSparseSpaceType::ScaleAndAdd(omega, s, 1.00, rX);
357  TSparseSpaceType::ScaleAndAdd(-omega, qs, 1.00, s, r);
358 
359  roh1 = TSparseSpaceType::Dot(r,rs);
360 
361  //if((roh0 == 0.00) || (omega == 0.00))
362  if((fabs(roh0) <= 1.0e-30) || (fabs(omega) <= 1.0e-30))
363  break;
364 
365  beta = (roh1 * alpha) / (roh0 * omega);
366 
367  TSparseSpaceType::ScaleAndAdd(1.00, p, -omega, q);
368  TSparseSpaceType::ScaleAndAdd(1.00, r, beta, q, p);
369 
370  roh0 = roh1;
371 
372  BaseType::mResidualNorm =TSparseSpaceType::TwoNorm(r);
373  BaseType::mIterationsNumber++;
374 
375  } while(BaseType::IterationNeeded());
376 
377  return BaseType::IsConverged();
378  }
